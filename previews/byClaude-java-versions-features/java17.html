package java17;

/**
 * <b>Sealed Classes and Interfaces</b>
 * <p>
 * Sealed classes restrict which classes may extend or implement them via the {@code permits}
 * clause. Each permitted subtype must be declared {@code final} (no further subclassing),
 * {@code sealed} (further restricted), or {@code non-sealed} (open for anyone to extend).
 * Sealed hierarchies enable exhaustive pattern matching and give library authors tight control
 * over type hierarchies. Previewed in Java 15 (JEP 360) and Java 16 (JEP 397);
 * finalized in Java 17 (JEP 409).
 *
 * @see <a href="https://openjdk.org/jeps/409">JEP 409 – Sealed Classes</a>
 */
public class SealedClasses {

    // ── Before: open abstract class — any code can subclass it ───────────────

    static abstract class OpenShape {
        abstract double area();
    }
    // Anyone can add: class Hexagon extends OpenShape { ... }
    // A switch over OpenShape subtypes always needs a defensive default clause.

    // ── After: sealed hierarchy — only permitted subtypes allowed ────────────

    sealed interface Shape permits Circle, Rectangle, Triangle {}

    /** final — cannot be subclassed further */
    static final class Circle implements Shape {
        private final double radius;
        Circle(double radius) { this.radius = radius; }
        double area() { return Math.PI * radius * radius; }
        double radius() { return radius; }
        @Override public String toString() { return "Circle(r=" + radius + ")"; }
    }

    /** final — cannot be subclassed further */
    static final class Rectangle implements Shape {
        private final double width, height;
        Rectangle(double width, double height) { this.width = width; this.height = height; }
        double area() { return width * height; }
        @Override public String toString() { return "Rectangle(" + width + "x" + height + ")"; }
    }

    /**
     * non-sealed — opens Triangle for arbitrary subclassing.
     * This is an escape hatch when a permitted type must remain extensible.
     */
    static non-sealed class Triangle implements Shape {
        private final double base, height;
        Triangle(double base, double height) { this.base = base; this.height = height; }
        double area() { return 0.5 * base * height; }
        @Override public String toString() { return "Triangle(b=" + base + ", h=" + height + ")"; }
    }

    /** Allowed because Triangle is non-sealed */
    static class RightTriangle extends Triangle {
        RightTriangle(double base, double height) { super(base, height); }
        @Override public String toString() { return "RightTriangle(super=" + super.toString() + ")"; }
    }

    public static void main(String[] args) {
        System.out.println("=== Before Java 17: open abstract class ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 17: sealed interface ===");
        demonstrateAfter();
    }

    /** Before Java 17: open hierarchy — defensive default always needed */
    static void demonstrateBefore() {
        OpenShape shape = new OpenShape() {
            @Override double area() { return 42.0; }
        };
        // Must handle unknown subtypes with a defensive else/default
        String desc;
        if (shape instanceof OpenShape s) {
            desc = "Some shape with area " + s.area();
        } else {
            desc = "Unknown"; // required even though logically unreachable
        }
        System.out.println(desc); // Output: Some shape with area 42.0
    }

    /** Java 17: sealed hierarchy — the compiler knows all permitted subtypes */
    static void demonstrateAfter() {
        Shape[] shapes = {
                new Circle(5),
                new Rectangle(4, 5),
                new Triangle(6, 8),
                new RightTriangle(3, 4) // non-sealed subtype works fine
        };

        for (Shape shape : shapes) {
            // Pattern matching instanceof — no default needed when handling all sealed subtypes
            // (exhaustive switch is available from Java 21; here we use if/else)
            String desc;
            if (shape instanceof Circle c) {
                desc = c + " → area = " + String.format("%.2f", c.area());
            } else if (shape instanceof Rectangle r) {
                desc = r + " → area = " + String.format("%.2f", r.area());
            } else if (shape instanceof Triangle t) {
                // Covers Triangle AND its non-sealed subclass RightTriangle
                desc = t + " → area = " + String.format("%.2f", t.area());
            } else {
                // This branch is logically unreachable for the sealed interface,
                // but the compiler still requires it in if/else chains (not in switch).
                desc = "Unknown shape";
            }
            System.out.println(desc);
        }
        // Output:
        // Circle(r=5.0) → area = 78.54
        // Rectangle(4.0x5.0) → area = 20.00
        // Triangle(b=6.0, h=8.0) → area = 24.00
        // RightTriangle(super=Triangle(b=3.0, h=4.0)) → area = 6.00

        // Compiler prevents unknown subtypes — this would NOT compile:
        // class Hexagon implements Shape {} // error: not listed in permits
    }
}

package java11;

import java.util.Arrays;
import java.util.stream.Collectors;

/**
 * <b>New String Methods</b>
 * <p>
 * Java 11 added several convenience methods to {@code String}:
 * <ul>
 *   <li>{@code isBlank()} — true if the string is empty or contains only whitespace</li>
 *   <li>{@code strip()} / {@code stripLeading()} / {@code stripTrailing()} — Unicode-aware whitespace removal (unlike {@code trim()} which only handles ASCII ≤ U+0020)</li>
 *   <li>{@code lines()} — lazily splits on line terminators, returning a {@code Stream<String>}</li>
 *   <li>{@code repeat(n)} — concatenates the string with itself {@code n} times</li>
 * </ul>
 * Introduced in Java 11.
 *
 * @see <a href="https://docs.oracle.com/en/java/se/11/docs/api/java.base/java/lang/String.html">String Javadoc (Java 11)</a>
 */
public class StringMethods {

    public static void main(String[] args) {
        System.out.println("=== Before Java 11: workarounds ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 11: new String methods ===");
        demonstrateAfter();
    }

    /** Before Java 11: verbose equivalents of the new methods */
    static void demonstrateBefore() {
        // isBlank — using trim().isEmpty()
        String blank = "   ";
        System.out.println("blank check (trim): " + blank.trim().isEmpty()); // Output: true

        // strip — trim() removes only ASCII whitespace (chars ≤ '\u0020')
        // The Unicode en-quad space U+2000 is NOT removed by trim()
        String withUnicode = "\u2000hello\u2000";
        System.out.println("trim keeps Unicode space: '" + withUnicode.trim() + "'"); // Output: ' hello ' (spaces remain)

        // lines — splitting manually
        String multiline = "line1\nline2\nline3";
        String[] lineArray = multiline.split("\\R");
        System.out.println("lines via split: " + Arrays.asList(lineArray)); // Output: [line1, line2, line3]

        // repeat — StringBuilder loop
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 3; i++) sb.append("ha");
        System.out.println("repeat via loop: " + sb); // Output: hahaha
    }

    /** Java 11: clean, expressive string utility methods */
    static void demonstrateAfter() {
        // isBlank()
        System.out.println("\"   \".isBlank(): "  + "   ".isBlank());  // Output: true
        System.out.println("\"hi \".isBlank(): "  + "hi ".isBlank());  // Output: false
        System.out.println("\"\".isBlank(): "      + "".isBlank());     // Output: true

        // strip() — Unicode-aware (also removes U+2000, U+00A0, etc.)
        String padded = "  hello  ";
        System.out.println("strip:         '" + padded.strip()         + "'"); // Output: 'hello'
        System.out.println("stripLeading:  '" + padded.stripLeading()  + "'"); // Output: 'hello  '
        System.out.println("stripTrailing: '" + padded.stripTrailing() + "'"); // Output: '  hello'

        // Unicode whitespace — strip() removes it, trim() does not
        String withUnicode = "\u2000hello\u2000";
        System.out.println("trim  Unicode: '" + withUnicode.trim()  + "'"); // Output: 'hello' on some JVMs, or with spaces
        System.out.println("strip Unicode: '" + withUnicode.strip() + "'"); // Output: 'hello'

        // lines() — returns Stream<String>, handles \n, \r\n, \r
        String text = "first\nsecond\nthird";
        text.lines().forEach(System.out::println);
        // Output:
        // first
        // second
        // third

        // Collect lines into a List
        var lineList = text.lines().collect(Collectors.toList());
        System.out.println("line count: " + lineList.size()); // Output: 3

        // repeat(n)
        System.out.println("ha".repeat(3));   // Output: hahaha
        System.out.println("-".repeat(20));   // Output: --------------------
        System.out.println("ab".repeat(0));   // Output: (empty string)
    }
}

// === Before Java 11: workarounds ===
// blank check (trim): true
// trim keeps Unicode space: '?hello?'
// lines via split: [line1, line2, line3]
// repeat via loop: hahaha
//
// === Java 11: new String methods ===
// "   ".isBlank(): true
// "hi ".isBlank(): false
// "".isBlank(): true
// strip:         'hello'
// stripLeading:  'hello  '
// stripTrailing: '  hello'
// trim  Unicode: '?hello?'
// strip Unicode: 'hello'
// first
// second
// third
// line count: 3

HttpClientApi.java 
package java11;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URI;
import java.net.URL;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.time.Duration;

/**
 * <b>HTTP Client API</b>
 * <p>
 * Java 11 standardised the {@code java.net.http.HttpClient} API (incubated in Java 9 under
 * {@code jdk.incubator.http}). It provides a fluent builder for requests, supports HTTP/1.1
 * and HTTP/2, and offers both synchronous ({@code send}) and asynchronous
 * ({@code sendAsync}) calls. It replaces the verbose {@code HttpURLConnection} API.
 * Finalized in Java 11 (JEP 321).
 *
 * @see <a href="https://openjdk.org/jeps/321">JEP 321 – HTTP Client (Standard)</a>
 */
public class HttpClientApi {

    private static final String URL_STR = "https://httpbin.org/get";

    public static void main(String[] args) {
        System.out.println("=== Before Java 11: HttpURLConnection ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 11: HttpClient API ===");
        demonstrateAfter();
    }

    /** Before Java 11: verbose HttpURLConnection approach */
    static void demonstrateBefore() {
        try {
            URL url = new URL(URL_STR);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            conn.setConnectTimeout(3000);
            conn.setReadTimeout(3000);

            int status = conn.getResponseCode();
            System.out.println("Status (old API): " + status); // Output: 200

            StringBuilder response = new StringBuilder();
            try (BufferedReader reader = new BufferedReader(
                    new InputStreamReader(conn.getInputStream()))) {
                String line;
                while ((line = reader.readLine()) != null) {
                    response.append(line).append('\n');
                }
            }
            System.out.println("Body (first 60 chars): " + response.substring(0, Math.min(60, response.length())));
            conn.disconnect();
        } catch (Exception e) {
            System.out.println("(offline — HttpURLConnection demo skipped: " + e.getMessage() + ")");
            System.out.println("Old API requires: URL, openConnection(), setRequestMethod(), getResponseCode(), BufferedReader loop");
        }
    }

    /** Java 11: fluent HttpClient with sync and async requests */
    static void demonstrateAfter() {
        HttpClient client = HttpClient.newBuilder()
                .connectTimeout(Duration.ofSeconds(3))
                .version(HttpClient.Version.HTTP_2)
                .build();

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(URL_STR))
                .GET()
                .header("Accept", "application/json")
                .timeout(Duration.ofSeconds(3))
                .build();

        // Synchronous send
        try {
            HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
            System.out.println("Status (new API): " + response.statusCode()); // Output: 200
            System.out.println("HTTP version: " + response.version());
            String body = response.body();
            System.out.println("Body (first 60 chars): " + body.substring(0, Math.min(60, body.length())));
        } catch (Exception e) {
            System.out.println("(offline — sync demo skipped: " + e.getMessage() + ")");
        }

        // Asynchronous send — returns CompletableFuture
        System.out.println("--- async request ---");
        try {
            client.sendAsync(request, HttpResponse.BodyHandlers.ofString())
                    .thenApply(HttpResponse::statusCode)
                    .thenAccept(code -> System.out.println("Async status: " + code)) // Output: Async status: 200
                    .join(); // wait for completion in this demo
        } catch (Exception e) {
            System.out.println("(offline — async demo skipped: " + e.getMessage() + ")");
        }

        // POST example (body not sent to httpbin in this demo)
        HttpRequest postRequest = HttpRequest.newBuilder()
                .uri(URI.create("https://httpbin.org/post"))
                .POST(HttpRequest.BodyPublishers.ofString("{\"key\":\"value\"}"))
                .header("Content-Type", "application/json")
                .build();
        System.out.println("POST request built: " + postRequest.method() + " " + postRequest.uri());
        // Output: POST request built: POST https://httpbin.org/post
    }
}

// === Before Java 11: HttpURLConnection ===
// Status (old API): 200
// Body (first 60 chars): {
//   "args": {},
//   "headers": {
//     "Accept": "*/*",   "H
//
// === Java 11: HttpClient API ===
// Status (new API): 200
// HTTP version: HTTP_2
// Body (first 60 chars): {
//   "args": {},
//   "headers": {

// --- async request ---
// Async status: 200

package java11;

import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;

/**
 * <b>Files.readString and Files.writeString</b>
 * <p>
 * Java 11 added {@code Files.readString(Path)} and {@code Files.writeString(Path, CharSequence)}
 * as concise alternatives to reading/writing text files. Both use UTF-8 by default and accept
 * an optional {@code Charset} parameter. They replace multi-step approaches involving byte
 * arrays or {@code BufferedWriter}. Introduced in Java 11.
 *
 * @see <a href="https://docs.oracle.com/en/java/se/11/docs/api/java.base/java/nio/file/Files.html">Files Javadoc (Java 11)</a>
 */
public class FileMethods {

    public static void main(String[] args) {
        System.out.println("=== Before Java 11: verbose file I/O ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 11: Files.readString / writeString ===");
        demonstrateAfter();
    }

    /** Before Java 11: reading and writing files required multiple steps */
    static void demonstrateBefore() {
        Path tmp = null;
        try {
            tmp = Files.createTempFile("demo_before", ".txt");

            // Write: convert String to bytes manually
            String content = "Hello, Java!\nSecond line.";
            Files.write(tmp, content.getBytes(StandardCharsets.UTF_8));

            // Read: load bytes and convert back to String
            byte[] bytes = Files.readAllBytes(tmp);
            String read = new String(bytes, StandardCharsets.UTF_8);
            System.out.println("Read (old): " + read); // Output: Hello, Java!\nSecond line.

            // Alternative write using BufferedWriter
            try (BufferedWriter writer = Files.newBufferedWriter(tmp, StandardCharsets.UTF_8)) {
                writer.write("Overwritten content");
            }
            System.out.println("Written via BufferedWriter");
        } catch (IOException e) {
            System.out.println("Error: " + e.getMessage());
        } finally {
            if (tmp != null) {
                try { Files.deleteIfExists(tmp); } catch (IOException ignored) {}
            }
        }
    }

    /** Java 11: single-method read/write with Files.readString / writeString */
    static void demonstrateAfter() {
        Path tmp = null;
        try {
            tmp = Files.createTempFile("demo_java11", ".txt");

            // Write — one line, UTF-8 by default
            String content = "Hello, Java 11!\nSecond line.\nThird line.";
            Files.writeString(tmp, content);
            System.out.println("File written to: " + tmp.getFileName());

            // Read — one line, UTF-8 by default
            String read = Files.readString(tmp);
            System.out.println("Read content:\n" + read);
            // Output:
            // Hello, Java 11!
            // Second line.
            // Third line.

            // Line count using the Java 11 String.lines() method
            long lineCount = read.lines().count();
            System.out.println("Line count: " + lineCount); // Output: 3

            // Explicit charset
            Files.writeString(tmp, "Café", StandardCharsets.UTF_8);
            String withAccent = Files.readString(tmp, StandardCharsets.UTF_8);
            System.out.println("With accent: " + withAccent); // Output: Café

            // Append using StandardOpenOption
            Files.writeString(tmp, "\nAppended line",
                    java.nio.file.StandardOpenOption.APPEND);
            System.out.println("After append lines: " + Files.readString(tmp).lines().count()); // Output: 2

        } catch (IOException e) {
            System.out.println("Error: " + e.getMessage());
        } finally {
            if (tmp != null) {
                try { Files.deleteIfExists(tmp); } catch (IOException ignored) {}
            }
        }
    }
}

package java16;

import java.util.Objects;

/**
 * <b>Records</b>
 * <p>
 * Records are a concise syntax for declaring immutable data-carrier classes. The compiler
 * auto-generates a canonical constructor, accessor methods (named after the components,
 * not {@code getX()}), and {@code toString}, {@code equals}, and {@code hashCode}
 * implementations. Records are implicitly {@code final} and cannot extend other classes.
 * Custom methods, compact constructors (for validation), and interface implementations are
 * all supported. Previewed in Java 14 (JEP 359) and Java 15 (JEP 384); finalized in
 * Java 16 (JEP 395).
 *
 * @see <a href="https://openjdk.org/jeps/395">JEP 395 – Records</a>
 */
public class Records {

    // ── Before: full POJO boilerplate ────────────────────────────────────────

    static final class PointOld {
        private final int x;
        private final int y;

        PointOld(int x, int y) { this.x = x; this.y = y; }

        public int getX() { return x; }
        public int getY() { return y; }

        @Override
        public boolean equals(Object o) {
            if (this == o) return true;
            if (!(o instanceof PointOld that)) return false;
            return x == that.x && y == that.y;
        }

        @Override
        public int hashCode() { return Objects.hash(x, y); }

        @Override
        public String toString() { return "PointOld{x=" + x + ", y=" + y + "}"; }
    }

    // ── After: records ────────────────────────────────────────────────────────

    /** Basic record — one line replaces ~30 lines of POJO */
    record Point(int x, int y) {
        /** Custom instance method */
        double distance() {
            return Math.sqrt((double) x * x + (double) y * y);
        }
    }

    /** Compact constructor — validation logic without repeating field assignments */
    record PositivePoint(int x, int y) {
        PositivePoint {
            // In a compact constructor, fields are assigned automatically after this block
            if (x < 0 || y < 0) throw new IllegalArgumentException(
                    "Coordinates must be non-negative, got (" + x + ", " + y + ")");
        }
    }

    /** Record implementing an interface */
    interface Shape {
        double area();
    }

    record Circle(double radius) implements Shape {
        // Compact constructor for validation
        Circle {
            if (radius <= 0) throw new IllegalArgumentException("Radius must be positive");
        }

        @Override
        public double area() { return Math.PI * radius * radius; }
    }

    /** Generic record */
    record Pair<A, B>(A first, B second) {}

    public static void main(String[] args) {
        System.out.println("=== Before Java 16: verbose POJO ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 16: records ===");
        demonstrateAfter();
    }

    /** Before Java 16: manual boilerplate for a simple data class */
    static void demonstrateBefore() {
        PointOld p1 = new PointOld(3, 4);
        PointOld p2 = new PointOld(3, 4);
        System.out.println("p1: " + p1);                    // Output: PointOld{x=3, y=4}
        System.out.println("p1.equals(p2): " + p1.equals(p2)); // Output: true
        System.out.println("p1.getX(): " + p1.getX());      // Output: 3
    }

    /** Java 16: records — auto-generated constructor, accessors, equals, hashCode, toString */
    static void demonstrateAfter() {
        // Canonical constructor and auto-generated toString
        Point p1 = new Point(3, 4);
        System.out.println("p1: " + p1); // Output: Point[x=3, y=4]

        // Accessor methods — named after components (no "get" prefix)
        System.out.println("p1.x(): " + p1.x()); // Output: 3
        System.out.println("p1.y(): " + p1.y()); // Output: 4

        // Auto-generated equals and hashCode
        Point p2 = new Point(3, 4);
        Point p3 = new Point(1, 2);
        System.out.println("p1.equals(p2): " + p1.equals(p2)); // Output: true
        System.out.println("p1.equals(p3): " + p1.equals(p3)); // Output: false
        System.out.println("same hash: " + (p1.hashCode() == p2.hashCode())); // Output: true

        // Custom method
        System.out.printf("distance(3,4): %.1f%n", p1.distance()); // Output: 5.0

        // Compact constructor validates input
        try {
            new PositivePoint(-1, 5);
        } catch (IllegalArgumentException e) {
            System.out.println("Validation: " + e.getMessage()); // Output: Coordinates must be non-negative...
        }
        PositivePoint pp = new PositivePoint(2, 3);
        System.out.println("PositivePoint: " + pp); // Output: PositivePoint[x=2, y=3]

        // Record implementing an interface
        Circle c = new Circle(5.0);
        System.out.printf("Circle area: %.2f%n", c.area()); // Output: 78.54

        // Generic record
        Pair<String, Integer> pair = new Pair<>("Hello", 42);
        System.out.println("Pair: " + pair); // Output: Pair[first=Hello, second=42]
        System.out.println("first: " + pair.first() + ", second: " + pair.second());

        // Records are final — cannot be subclassed (compiler error if attempted):
        // class SubPoint extends Point {} // ← compile error
    }
}

package java16;

import java.util.Arrays;
import java.util.List;

/**
 * <b>Pattern Matching for {@code instanceof}</b>
 * <p>
 * Pattern matching for {@code instanceof} combines the type test and the cast into a single
 * expression. When the test succeeds, a <em>pattern variable</em> is bound to the value
 * (already cast) and is in scope where the condition is known to be true.
 * This eliminates the redundant explicit cast that always follows {@code instanceof}.
 * Previewed in Java 14 (JEP 305) and Java 15 (JEP 375); finalized in Java 16 (JEP 394).
 *
 * @see <a href="https://openjdk.org/jeps/394">JEP 394 – Pattern Matching for instanceof</a>
 */
public class PatternMatchingInstanceof {

    public static void main(String[] args) {
        System.out.println("=== Before Java 16: explicit cast after instanceof ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 16: pattern matching instanceof ===");
        demonstrateAfter();
    }

    /** Before Java 16: three separate operations — test, cast, use */
    static void demonstrateBefore() {
        Object obj = "Hello";

        // Test, then cast separately — redundant and verbose
        if (obj instanceof String) {
            String s = (String) obj; // explicit cast required
            System.out.println("Length: " + s.length()); // Output: 5
        }

        // Multi-type dispatch
        List<Object> mixed = Arrays.asList("text", 42, 3.14, true);
        for (Object o : mixed) {
            if (o instanceof String) {
                String s = (String) o;
                System.out.println("String: " + s.toUpperCase());
            } else if (o instanceof Integer) {
                Integer i = (Integer) o;
                System.out.println("Integer doubled: " + (i * 2));
            } else {
                System.out.println("Other: " + o);
            }
        }
        // Output:
        // String: TEXT
        // Integer doubled: 84
        // Other: 3.14
        // Other: true
    }

    /** Java 16: pattern variable bound in the instanceof expression */
    static void demonstrateAfter() {
        Object obj = "Hello";

        // Test + bind in one expression — 's' is in scope inside the if block
        if (obj instanceof String s) {
            System.out.println("Length: " + s.length()); // Output: 5
        }

        // Guard condition with && — 's' only bound when both conditions are true
        if (obj instanceof String s && s.length() > 3) {
            System.out.println("Long string: " + s.toUpperCase()); // Output: HELLO
        }

        // Multi-type dispatch — cleaner, no explicit casts
        List<Object> mixed = Arrays.asList("text", 42, 3.14, true);
        for (Object o : mixed) {
            if (o instanceof String s) {
                System.out.println("String: " + s.toUpperCase());
            } else if (o instanceof Integer i) {
                System.out.println("Integer doubled: " + (i * 2));
            } else if (o instanceof Double d) {
                System.out.println("Double: " + d);
            } else {
                System.out.println("Other: " + o);
            }
        }
        // Output:
        // String: TEXT
        // Integer doubled: 84
        // Double: 3.14
        // Other: true

        // Scope: 's' is NOT in scope in the else branch
        Object maybeString = 99;
        if (maybeString instanceof String s) {
            System.out.println("Is string: " + s);
        } else {
            // 's' is not accessible here — the compiler knows the test failed
            System.out.println("Not a string: " + maybeString); // Output: Not a string: 99
        }

        // Negation + early return — 's' flows into the rest of the method
        printUpperIfString("world");
        printUpperIfString(123);
    }

    static void printUpperIfString(Object obj) {
        if (!(obj instanceof String s)) {
            System.out.println("Not a string — returning");
            return;
        }
        // 's' is in scope here because we know the non-string case returned
        System.out.println("Upper: " + s.toUpperCase()); // Output: WORLD
    }
}

// Console output:
// === Before Java 16: explicit cast after instanceof ===
// Length: 5
// String: TEXT
// Integer doubled: 84
// Other: 3.14
// Other: true
//
// === Java 16: pattern matching instanceof ===
// Length: 5
// Long string: HELLO
// String: TEXT
// Integer doubled: 84
// Double: 3.14
// Other: true
// Not a string: 99
// Upper: WORLD

StreamToList.java 
package java16;

import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.Stream;

/**
 * <b>Stream.toList()</b>
 * <p>
 * {@code Stream.toList()} is a concise terminal operation added in Java 16 that collects
 * stream elements into an <em>unmodifiable</em> {@code List}, replacing the verbose
 * {@code collect(Collectors.toList())} call. Unlike {@code Collectors.toList()} (which returns
 * a mutable {@code ArrayList}), {@code toList()} always returns an unmodifiable list.
 * Introduced in Java 16.
 *
 * @see <a href="https://docs.oracle.com/en/java/se/16/docs/api/java.base/java/util/stream/Stream.html#toList()">Stream.toList() Javadoc</a>
 */
public class StreamToList {

    public static void main(String[] args) {
        System.out.println("=== Before Java 16: Collectors.toList() ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 16: Stream.toList() ===");
        demonstrateAfter();
    }

    /** Before Java 16: verbose collect() calls */
    static void demonstrateBefore() {
        List<String> names = Stream.of("Alice", "Bob", "Charlie")
                .filter(s -> s.length() > 3)
                .collect(Collectors.toList()); // mutable ArrayList returned
        System.out.println("Names: " + names); // Output: [Alice, Charlie]

        // Mutable — add() succeeds
        names.add("Diana");
        System.out.println("After add: " + names); // Output: [Alice, Charlie, Diana]

        // To get an unmodifiable list was even more verbose
        List<String> unmod = Stream.of("X", "Y", "Z")
                .collect(Collectors.toUnmodifiableList());
        System.out.println("Unmodifiable: " + unmod); // Output: [X, Y, Z]
        try {
            unmod.add("W");
        } catch (UnsupportedOperationException e) {
            System.out.println("Unmod add threw: UnsupportedOperationException");
        }
    }

    /** Java 16: Stream.toList() — concise and always unmodifiable */
    static void demonstrateAfter() {
        List<String> names = Stream.of("Alice", "Bob", "Charlie")
                .filter(s -> s.length() > 3)
                .toList(); // shorter, and unmodifiable
        System.out.println("Names: " + names); // Output: [Alice, Charlie]

        // toList() result is unmodifiable
        try {
            names.add("Diana");
        } catch (UnsupportedOperationException e) {
            System.out.println("toList() add threw: UnsupportedOperationException"); // Output: ...
        }

        // Full pipeline example
        record Person(String name, int age) {}
        List<Person> people = List.of(
                new Person("Alice", 30),
                new Person("Bob", 17),
                new Person("Charlie", 25)
        );

        List<String> adults = people.stream()
                .filter(p -> p.age() >= 18)
                .map(Person::name)
                .sorted()
                .toList();
        System.out.println("Adults: " + adults); // Output: [Alice, Charlie]

        // Contrast: collect(toList()) is still mutable
        List<String> mutable = Stream.of("A", "B")
                .collect(Collectors.toList());
        mutable.add("C");
        System.out.println("Mutable after add: " + mutable); // Output: [A, B, C]

        System.out.println("toList() vs collect(toList()): toList() is always unmodifiable");
    }
}

// Console output:
// === Before Java 16: Collectors.toList() ===
// Names: [Alice, Charlie]
// After add: [Alice, Charlie, Diana]
// Unmodifiable: [X, Y, Z]
// Unmod add threw: UnsupportedOperationException
//
// === Java 16: Stream.toList() ===
// Names: [Alice, Charlie]
// toList() add threw: UnsupportedOperationException
// Adults: [Alice, Charlie]
// Mutable after add: [A, B, C]
// toList() vs collect(toList()): toList() is always unmodifiable

package java24;

import java.util.List;
import java.util.Optional;
import java.util.stream.Gatherer;
import java.util.stream.Gatherers;
import java.util.stream.Stream;

/**
 * <b>Stream Gatherers</b>
 * <p>
 * {@code Stream.gather(Gatherer)} is a new intermediate operation that allows building
 * custom, composable stream transformations — filling the gap between built-in operations
 * like {@code map} and {@code filter} and fully custom {@code Collector}s.
 * <p>
 * The JDK ships four built-in gatherers in {@code java.util.stream.Gatherers}:
 * <ul>
 *   <li>{@code windowFixed(n)} — groups elements into non-overlapping lists of size n</li>
 *   <li>{@code windowSliding(n)} — groups elements into overlapping sliding windows of size n</li>
 *   <li>{@code fold(init, fn)} — stateful left-fold (like reduce, but keeps intermediate state for mapping)</li>
 *   <li>{@code scan(init, fn)} — running totals: emits the accumulated value after each element</li>
 *   <li>{@code mapConcurrent(n, fn)} — concurrent map with at most n in-flight virtual threads</li>
 * </ul>
 * Gatherers can also be composed with {@code andThen()} and implemented from scratch.
 * Previewed in Java 22 (JEP 461) and Java 23 (JEP 473); finalized in Java 24 (JEP 485).
 *
 * @see <a href="https://openjdk.org/jeps/485">JEP 485 – Stream Gatherers</a>
 */
public class StreamGatherers {

    public static void main(String[] args) {
        System.out.println("=== Before Java 24: custom intermediate ops require workarounds ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 24: Stream Gatherers ===");
        demonstrateAfter();
    }

    /** Before Java 24: windowing and running totals required external loops or collectors */
    static void demonstrateBefore() {
        List<Integer> numbers = List.of(1, 2, 3, 4, 5, 6, 7, 8);

        // Fixed windows — no built-in: must collect then batch manually
        System.out.println("Fixed windows (manual):");
        List<Integer> list = numbers.stream().toList();
        int windowSize = 3;
        for (int i = 0; i < list.size(); i += windowSize) {
            int end = Math.min(i + windowSize, list.size());
            System.out.println("  " + list.subList(i, end));
        }
        // Output: [1, 2, 3] / [4, 5, 6] / [7, 8]

        // Running total (scan) — must collect to list first, then iterate with index
        System.out.println("Running total (manual):");
        int[] runningTotal = {0};
        numbers.forEach(n -> {
            runningTotal[0] += n;
            System.out.print(runningTotal[0] + " ");
        });
        System.out.println();
        // Output: 1 3 6 10 15 21 28 36
    }

    /** Java 24: built-in and custom gatherers as composable stream operations */
    static void demonstrateAfter() {
        List<Integer> numbers = List.of(1, 2, 3, 4, 5, 6, 7, 8);

        // windowFixed(n) — non-overlapping windows, last window may be smaller
        System.out.println("windowFixed(3):");
        numbers.stream()
                .gather(Gatherers.windowFixed(3))
                .forEach(w -> System.out.println("  " + w));
        // Output: [1, 2, 3] / [4, 5, 6] / [7, 8]

        // windowSliding(n) — overlapping windows, one new element shifts the window
        System.out.println("windowSliding(3):");
        numbers.stream()
                .gather(Gatherers.windowSliding(3))
                .forEach(w -> System.out.println("  " + w));
        // Output: [1,2,3] / [2,3,4] / [3,4,5] / [4,5,6] / [5,6,7] / [6,7,8]

        // scan(identity, accumulator) — running total, emits state after each element
        System.out.println("scan (running total):");
        List<Integer> runningTotals = numbers.stream()
                .gather(Gatherers.scan(() -> 0, Integer::sum))
                .toList();
        System.out.println("  " + runningTotals);
        // Output: [1, 3, 6, 10, 15, 21, 28, 36]

        // fold(identity, accumulator) — like reduce but as an intermediate gatherer;
        // emits a single Optional with the final accumulated value
        System.out.println("fold (sum):");
        Optional<Integer> total = numbers.stream()
                .gather(Gatherers.fold(() -> 0, Integer::sum))
                .findFirst();
        System.out.println("  sum = " + total.orElse(0));
        // Output: sum = 36

        // Composing gatherers with andThen()
        System.out.println("windowFixed(2).andThen(map to sum):");
        numbers.stream()
                .gather(Gatherers.windowFixed(2))           // group into pairs
                .map(w -> w.stream().mapToInt(Integer::intValue).sum()) // sum each pair
                .forEach(s -> System.out.print(s + " "));  // Output: 3 7 11 15
        System.out.println();

        // Custom Gatherer — "takeWhileIncreasing": emit elements while each is > previous
        System.out.println("Custom gatherer — takeWhileIncreasing:");
        Gatherer<Integer, int[], Integer> takeWhileIncreasing = Gatherer.ofSequential(
                () -> new int[]{Integer.MIN_VALUE},          // initializer: prev = MIN_VALUE
                (state, element, downstream) -> {            // integrator
                    if (element > state[0]) {
                        state[0] = element;
                        return downstream.push(element);     // emit and continue
                    }
                    return false;                            // stop the stream
                }
        );

        List<Integer> increasing = Stream.of(1, 3, 5, 4, 6, 2, 7)
                .gather(takeWhileIncreasing)
                .toList();
        System.out.println("  " + increasing); // Output: [1, 3, 5]

        // mapConcurrent(n, fn) — maps elements using virtual threads, at most n in flight
        // Useful for I/O-bound transformations (e.g., HTTP calls, DB lookups)
        System.out.println("mapConcurrent(3, slow fn):");
        List<String> results = Stream.of("a", "b", "c", "d", "e")
                .gather(Gatherers.mapConcurrent(3, s -> {
                    // Simulate latency (e.g. HTTP call)
                    try { Thread.sleep(10); } catch (InterruptedException e) { Thread.currentThread().interrupt(); }
                    return s.toUpperCase();
                }))
                .toList();
        System.out.println("  " + results); // Output: [A, B, C, D, E] (order preserved)
    }
}

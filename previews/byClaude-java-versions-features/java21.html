package java21;

import java.time.Duration;
import java.time.Instant;
import java.util.concurrent.CountDownLatch;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * <b>Virtual Threads</b>
 * <p>
 * Virtual threads are lightweight threads managed by the JVM rather than the OS.
 * A platform thread consumes ~1 MB of stack memory; a virtual thread uses only the heap
 * memory needed for its current stack frame, allowing millions of concurrent threads.
 * They use the same {@code java.lang.Thread} API and are ideal for <em>blocking I/O</em>
 * tasks in high-throughput servers. Anti-pattern: do not pool virtual threads —
 * create one per task and let the JVM schedule them. Previewed in Java 19 (JEP 425)
 * and Java 20 (JEP 436); finalized in Java 21 (JEP 444).
 *
 * @see <a href="https://openjdk.org/jeps/444">JEP 444 – Virtual Threads</a>
 */
public class VirtualThreads {

    public static void main(String[] args) throws InterruptedException {
        System.out.println("=== Before Java 21: platform threads ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 21: virtual threads ===");
        demonstrateAfter();
    }

    /** Before Java 21: platform threads — expensive, limited concurrency */
    static void demonstrateBefore() throws InterruptedException {
        int taskCount = 10;
        CountDownLatch latch = new CountDownLatch(taskCount);

        // Fixed thread pool — OS threads, limited by memory (~1MB stack each)
        try (ExecutorService pool = Executors.newFixedThreadPool(4)) {
            Instant start = Instant.now();
            for (int i = 0; i < taskCount; i++) {
                final int id = i;
                pool.submit(() -> {
                    try {
                        Thread.sleep(100); // simulated blocking I/O
                        System.out.println("  platform task " + id + " on " + Thread.currentThread().getName());
                    } catch (InterruptedException e) {
                        Thread.currentThread().interrupt();
                    } finally {
                        latch.countDown();
                    }
                });
            }
            latch.await(5, TimeUnit.SECONDS);
            System.out.println("Platform threads done in ~" + Duration.between(start, Instant.now()).toMillis() + " ms");
            // 10 tasks, 4 threads, 100ms each → ~300ms (tasks queued, not parallel)
        }
    }

    /** Java 21: virtual threads — one thread per task, scheduled by the JVM */
    static void demonstrateAfter() throws InterruptedException {
        // 1. Single virtual thread
        Thread vThread = Thread.ofVirtual()
                .name("my-vthread")
                .start(() -> {
                    System.out.println("  Running on: " + Thread.currentThread());
                    System.out.println("  isVirtual: " + Thread.currentThread().isVirtual()); // Output: true
                });
        vThread.join();

        // 2. Named virtual thread with a counter
        AtomicInteger counter = new AtomicInteger();
        Thread named = Thread.ofVirtual()
                .name("counter-thread")
                .start(() -> counter.incrementAndGet());
        named.join();
        System.out.println("  Counter: " + counter.get()); // Output: 1

        // 3. One virtual thread per task — scale to 1000 tasks easily
        int taskCount = 1000;
        CountDownLatch latch = new CountDownLatch(taskCount);
        AtomicInteger completed = new AtomicInteger();

        Instant start = Instant.now();
        try (ExecutorService exec = Executors.newVirtualThreadPerTaskExecutor()) {
            for (int i = 0; i < taskCount; i++) {
                exec.submit(() -> {
                    try {
                        Thread.sleep(100); // simulated blocking I/O
                        completed.incrementAndGet();
                    } catch (InterruptedException e) {
                        Thread.currentThread().interrupt();
                    } finally {
                        latch.countDown();
                    }
                });
            }
        } // ExecutorService.close() waits for all tasks (Java 19+ AutoCloseable)

        latch.await(10, TimeUnit.SECONDS);
        long ms = Duration.between(start, Instant.now()).toMillis();
        System.out.println("  1000 virtual-thread tasks done in ~" + ms + " ms");
        // All 1000 tasks run concurrently → ~100 ms total (not 100000 ms)
        System.out.println("  Completed: " + completed.get()); // Output: 1000

        // 4. ThreadFactory for frameworks / executors
        var factory = Thread.ofVirtual().name("vt-", 0).factory();
        Thread t = factory.newThread(() -> System.out.println("  Factory thread: " + Thread.currentThread().getName()));
        t.start();
        t.join();
        // Output: vt-0

        // Anti-pattern note:
        System.out.println("  Anti-pattern: do NOT pool virtual threads (Executors.newFixedThreadPool with virtual threads).");
        System.out.println("  Use newVirtualThreadPerTaskExecutor() instead.");
    }
}

RecordPatterns.java 
package java21;

/**
 * <b>Record Patterns</b>
 * <p>
 * Record patterns extend {@code instanceof} and {@code switch} pattern matching to
 * destructure record values inline. Instead of extracting components via accessor calls
 * after an {@code instanceof} check, the components are bound directly in the pattern:
 * {@code instanceof Point(int x, int y)}. Nested records can be destructured recursively.
 * {@code var} can be used in place of explicit types for inferred component binding.
 * Previewed in Java 19 (JEP 405) and Java 20 (JEP 432); finalized in Java 21 (JEP 440).
 *
 * @see <a href="https://openjdk.org/jeps/440">JEP 440 – Record Patterns</a>
 */
public class RecordPatterns {

    // Supporting records
    record Point(int x, int y) {}
    record Line(Point start, Point end) {}
    record Box<T>(T value) {}
    record ColoredPoint(Point point, String color) {}

    public static void main(String[] args) {
        System.out.println("=== Before Java 21: accessor calls after instanceof ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 21: record patterns ===");
        demonstrateAfter();
    }

    /** Before Java 21: must use accessor methods after the instanceof check */
    static void demonstrateBefore() {
        Object obj = new Point(3, 4);

        // Test type, then call accessors manually
        if (obj instanceof Point p) {
            int x = p.x();
            int y = p.y();
            System.out.println("Point x=" + x + ", y=" + y); // Output: Point x=3, y=4
        }

        // Nested record — deeply verbose accessor chain
        Object line = new Line(new Point(0, 0), new Point(5, 5));
        if (line instanceof Line l) {
            int x1 = l.start().x();
            int y1 = l.start().y();
            int x2 = l.end().x();
            int y2 = l.end().y();
            System.out.println("Line from (" + x1 + "," + y1 + ") to (" + x2 + "," + y2 + ")");
            // Output: Line from (0,0) to (5,5)
        }
    }

    /** Java 21: record patterns — destructure components directly in the pattern */
    static void demonstrateAfter() {
        Object obj = new Point(3, 4);

        // Record pattern: binds x and y directly — no accessor calls
        if (obj instanceof Point(int x, int y)) {
            System.out.println("Point x=" + x + ", y=" + y); // Output: Point x=3, y=4
        }

        // var in record patterns — type is inferred
        if (obj instanceof Point(var x, var y)) {
            System.out.println("Via var: x=" + x + ", y=" + y); // Output: Via var: x=3, y=4
        }

        // Nested record pattern — destructures recursively in one expression
        Object line = new Line(new Point(0, 0), new Point(5, 5));
        if (line instanceof Line(Point(int x1, int y1), Point(int x2, int y2))) {
            System.out.println("Line from (" + x1 + "," + y1 + ") to (" + x2 + "," + y2 + ")");
            // Output: Line from (0,0) to (5,5)
        }

        // Record pattern in switch (Java 21 pattern matching switch)
        Object[] objects = {
                new Point(1, 1),
                new Line(new Point(0, 0), new Point(3, 4)),
                new Box<>("Hello"),
                new ColoredPoint(new Point(2, 3), "red")
        };

        for (Object o : objects) {
            String description = switch (o) {
                case Point(int x, int y)                         -> "Point at (" + x + "," + y + ")";
                case Line(Point(int x1, int y1), Point(int x2, int y2))
                                                                 -> "Line (" + x1 + "," + y1 + ")→(" + x2 + "," + y2 + ")";
                case Box<?>  (var v)                             -> "Box containing: " + v;
                case ColoredPoint(Point(int x, int y), var color)-> color + " point at (" + x + "," + y + ")";
                default                                          -> "Unknown: " + o;
            };
            System.out.println(description);
        }
        // Output:
        // Point at (1,1)
        // Line (0,0)→(3,4)
        // Box containing: Hello
        // red point at (2,3)

        // Guarded record pattern — extra condition on destructured components
        Object p = new Point(5, 5);
        if (p instanceof Point(int x, int y) && x == y) {
            System.out.println("Diagonal point: (" + x + "," + y + ")"); // Output: Diagonal point: (5,5)
        }
    }
}

PatternMatchingSwitch.java 
package java21;

/**
 * <b>Pattern Matching for Switch</b>
 * <p>
 * Pattern matching in {@code switch} allows switching on the <em>type</em> of the selector,
 * with each case arm specifying a type pattern. Key capabilities:
 * <ul>
 *   <li>Type patterns: {@code case String s ->}</li>
 *   <li>Guarded patterns: {@code case String s when s.length() > 5 ->}</li>
 *   <li>Null handling: {@code case null ->} (avoids {@code NullPointerException})</li>
 *   <li>Exhaustive switch over sealed hierarchies — no {@code default} needed</li>
 *   <li>Record patterns inside switch: {@code case Point(int x, int y) ->}</li>
 * </ul>
 * Previewed in Java 17–20; finalized in Java 21 (JEP 441).
 *
 * @see <a href="https://openjdk.org/jeps/441">JEP 441 – Pattern Matching for switch</a>
 */
public class PatternMatchingSwitch {

    // Sealed hierarchy for exhaustiveness demonstration
    sealed interface Shape permits PatternMatchingSwitch.Circle,
            PatternMatchingSwitch.Rectangle, PatternMatchingSwitch.Triangle {}
    record Circle(double radius) implements Shape {}
    record Rectangle(double width, double height) implements Shape {}
    record Point(int x, int y) {}

    // non-sealed to keep the example self-contained
    static non-sealed class Triangle implements Shape {
        final double base, height;
        Triangle(double base, double height) { this.base = base; this.height = height; }
    }

    public static void main(String[] args) {
        System.out.println("=== Before Java 21: if/else instanceof chain ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 21: pattern matching switch ===");
        demonstrateAfter();
    }

    /** Before Java 21: multi-branch if/else for type dispatch */
    static void demonstrateBefore() {
        Object obj = "HelloWorld";

        String result;
        if (obj instanceof Integer i) {
            result = "int: " + i;
        } else if (obj instanceof Long l) {
            result = "long: " + l;
        } else if (obj instanceof Double d) {
            result = "double: " + d;
        } else if (obj instanceof String s && s.length() > 5) {
            result = "long string: " + s;
        } else if (obj instanceof String s) {
            result = "short string: " + s;
        } else {
            result = "other: " + obj;
        }
        System.out.println(result); // Output: long string: HelloWorld
    }

    /** Java 21: switch with type patterns, guards, null, and record patterns */
    static void demonstrateAfter() {
        // Type pattern switch
        for (Object obj : new Object[]{42, 3.14, "Hi", "HelloWorld", true}) {
            String result = switch (obj) {
                case Integer i              -> "int: " + i;
                case Double  d              -> "double: " + d;
                case String  s when s.length() > 5 -> "long string: " + s;
                case String  s              -> "short string: " + s;
                default                     -> "other: " + obj;
            };
            System.out.println(result);
        }
        // Output:
        // int: 42
        // double: 3.14
        // short string: Hi
        // long string: HelloWorld
        // other: true

        // Null handling in switch — avoids NullPointerException
        System.out.println("--- null handling ---");
        for (String s : new String[]{"hello", null, ""}) {
            String desc = switch (s) {
                case null    -> "null value!";
                case ""      -> "empty string";
                default      -> "string: " + s;
            };
            System.out.println(desc);
        }
        // Output:
        // string: hello
        // null value!
        // empty string

        // Exhaustive switch over sealed hierarchy — no default needed
        System.out.println("--- sealed shape areas ---");
        Shape[] shapes = {new Circle(5), new Rectangle(4, 6), new Triangle(3, 8)};
        for (Shape shape : shapes) {
            double area = switch (shape) {
                case Circle(double r)         -> Math.PI * r * r;
                case Rectangle(double w, double h) -> w * h;
                case Triangle t               -> 0.5 * t.base * t.height;
                // No default needed — compiler knows all sealed subtypes
            };
            System.out.printf("%s area = %.2f%n", shape.getClass().getSimpleName(), area);
        }
        // Output:
        // Circle area = 78.54
        // Rectangle area = 24.00
        // Triangle area = 12.00

        // Record pattern in switch with guard
        System.out.println("--- record pattern with guard ---");
        Object[] points = {new Point(0, 0), new Point(3, 3), new Point(1, 5)};
        for (Object obj : points) {
            String desc = switch (obj) {
                case Point(int x, int y) when x == 0 && y == 0 -> "origin";
                case Point(int x, int y) when x == y            -> "diagonal (" + x + "," + y + ")";
                case Point(int x, int y)                        -> "point (" + x + "," + y + ")";
                default                                         -> "not a point";
            };
            System.out.println(desc);
        }
        // Output:
        // origin
        // diagonal (3,3)
        // point (1,5)
    }
}

SequencedCollections.java 
package java21;

import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.LinkedList;
import java.util.List;
import java.util.SequencedCollection;

/**
 * <b>Sequenced Collections</b>
 * <p>
 * Java 21 introduced three new interfaces — {@code SequencedCollection}, {@code SequencedSet},
 * and {@code SequencedMap} — that give all ordered collections a <em>uniform API</em> for
 * accessing first/last elements, adding at head or tail, and obtaining a reversed view.
 * Before Java 21, every collection type used different method names for the same concept
 * ({@code get(0)} vs {@code peekFirst()} vs {@code first()}). Introduced in Java 21 (JEP 431).
 *
 * @see <a href="https://openjdk.org/jeps/431">JEP 431 – Sequenced Collections</a>
 */
public class SequencedCollections {

    public static void main(String[] args) {
        System.out.println("=== Before Java 21: inconsistent first/last APIs ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 21: SequencedCollection uniform API ===");
        demonstrateAfter();
    }

    /** Before Java 21: every collection type had its own way to access first/last */
    static void demonstrateBefore() {
        // List — index-based, verbose and error-prone
        List<String> list = new ArrayList<>(List.of("A", "B", "C", "D"));
        System.out.println("First (List):   " + list.get(0));                    // Output: A
        System.out.println("Last  (List):   " + list.get(list.size() - 1));      // Output: D

        // LinkedList (Deque) — different method names
        var deque = new LinkedList<>(List.of("X", "Y", "Z"));
        System.out.println("First (Deque):  " + deque.peekFirst());              // Output: X
        System.out.println("Last  (Deque):  " + deque.peekLast());               // Output: Z

        // SortedSet — yet another naming convention
        var sorted = new java.util.TreeSet<>(List.of(3, 1, 2));
        System.out.println("First (SortedSet): " + sorted.first());              // Output: 1
        System.out.println("Last  (SortedSet): " + sorted.last());               // Output: 3

        // No reversed() — had to create a copy or iterate manually
        List<String> reversed = new ArrayList<>(list);
        java.util.Collections.reverse(reversed);
        System.out.println("Reversed (copy): " + reversed);                      // Output: [D, C, B, A]
    }

    /** Java 21: uniform SequencedCollection API across all ordered collections */
    static void demonstrateAfter() {
        // ArrayList implements SequencedCollection
        List<String> list = new ArrayList<>(List.of("A", "B", "C", "D"));

        System.out.println("getFirst(): " + list.getFirst()); // Output: A
        System.out.println("getLast():  " + list.getLast());  // Output: D

        // addFirst / addLast
        list.addFirst("Z");
        list.addLast("X");
        System.out.println("After addFirst(Z) + addLast(X): " + list); // Output: [Z, A, B, C, D, X]

        // removeFirst / removeLast
        String removedFirst = list.removeFirst();
        String removedLast  = list.removeLast();
        System.out.println("Removed first: " + removedFirst + ", last: " + removedLast); // Output: Z, X
        System.out.println("After removes: " + list); // Output: [A, B, C, D]

        // reversed() — returns a live reversed VIEW (not a copy)
        SequencedCollection<String> rev = list.reversed();
        System.out.println("Reversed view: " + rev); // Output: [D, C, B, A]

        // Mutations on the reversed view affect the original
        rev.addFirst("LAST"); // addFirst on reversed view = addLast on original
        System.out.println("Original after rev.addFirst(LAST): " + list); // Output: [A, B, C, D, LAST]
        list.removeLast(); // cleanup

        // LinkedList — same uniform API
        LinkedList<Integer> linked = new LinkedList<>(List.of(10, 20, 30));
        System.out.println("\nLinkedList getFirst(): " + linked.getFirst()); // Output: 10
        System.out.println("LinkedList getLast():  " + linked.getLast());   // Output: 30
        linked.addFirst(5);
        System.out.println("LinkedList after addFirst(5): " + linked); // Output: [5, 10, 20, 30]
        System.out.println("LinkedList reversed: " + linked.reversed()); // Output: [30, 20, 10, 5]

        // LinkedHashMap — SequencedMap
        var map = new LinkedHashMap<String, Integer>();
        map.put("one", 1);
        map.put("two", 2);
        map.put("three", 3);

        System.out.println("\nLinkedHashMap firstEntry(): " + map.firstEntry()); // Output: one=1
        System.out.println("LinkedHashMap lastEntry():  " + map.lastEntry());   // Output: three=3
        System.out.println("Sequenced keys: " + map.sequencedKeySet());         // Output: [one, two, three]
        System.out.println("Reversed keys:  " + map.reversed().sequencedKeySet()); // Output: [three, two, one]
    }
}

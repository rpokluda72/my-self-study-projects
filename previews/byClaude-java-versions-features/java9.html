package java9;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;

/**
 * <b>Collection Factory Methods</b>
 * <p>
 * Java 9 added static factory methods {@code List.of()}, {@code Set.of()}, {@code Map.of()},
 * and {@code Map.ofEntries()} that create compact, unmodifiable collections in a single
 * expression. The resulting collections reject {@code null} elements and throw
 * {@code UnsupportedOperationException} on any mutation attempt. Introduced in Java 9 (JEP 269).
 *
 * @see <a href="https://openjdk.org/jeps/269">JEP 269 – Convenience Factory Methods for Collections</a>
 */
public class CollectionFactoryMethods {

    public static void main(String[] args) {
        System.out.println("=== Before Java 9: verbose unmodifiable collections ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 9: collection factory methods ===");
        demonstrateAfter();
    }

    /** Before Java 9: multiple steps to create an unmodifiable collection */
    static void demonstrateBefore() {
        // Unmodifiable list — two steps
        List<String> list = Collections.unmodifiableList(
                new ArrayList<>(Arrays.asList("Apple", "Banana", "Cherry")));
        System.out.println("List: " + list); // Output: [Apple, Banana, Cherry]

        // Unmodifiable set — two steps + extra constructor
        Set<Integer> set = Collections.unmodifiableSet(
                new HashSet<>(Arrays.asList(1, 2, 3)));
        System.out.println("Set contains 2: " + set.contains(2)); // Output: true

        // Unmodifiable map — three steps: create, populate, wrap
        Map<String, Integer> map = new HashMap<>();
        map.put("one", 1);
        map.put("two", 2);
        map = Collections.unmodifiableMap(map);
        System.out.println("Map: " + map); // Output: {one=1, two=2}
    }

    /** Java 9: one-line factory methods returning unmodifiable collections */
    static void demonstrateAfter() {
        // List.of() — ordered, unmodifiable
        List<String> list = List.of("Apple", "Banana", "Cherry");
        System.out.println("List: " + list); // Output: [Apple, Banana, Cherry]

        // Set.of() — unordered, unmodifiable, no duplicates allowed
        Set<Integer> set = Set.of(1, 2, 3);
        System.out.println("Set contains 2: " + set.contains(2)); // Output: true

        // Map.of() — up to 10 key-value pairs inline
        Map<String, Integer> map = Map.of("one", 1, "two", 2, "three", 3);
        System.out.println("Map get 'two': " + map.get("two")); // Output: 2

        // Map.ofEntries() — for more than 10 pairs or improved readability
        Map<String, String> capitals = Map.ofEntries(
                Map.entry("France", "Paris"),
                Map.entry("Germany", "Berlin"),
                Map.entry("Japan", "Tokyo")
        );
        System.out.println("Capital of Japan: " + capitals.get("Japan")); // Output: Tokyo

        // Mutation throws UnsupportedOperationException
        try {
            list.add("Durian");
        } catch (UnsupportedOperationException e) {
            System.out.println("add() threw: " + e.getClass().getSimpleName()); // Output: UnsupportedOperationException
        }

        // null elements are rejected with NullPointerException
        try {
            List.of("a", null, "b");
        } catch (NullPointerException e) {
            System.out.println("null element threw: NullPointerException"); // Output: NullPointerException
        }

        // Tip: to get a mutable copy of a factory-created collection, wrap it:
        List<String> mutable = new ArrayList<>(List.of("X", "Y"));
        mutable.add("Z");
        System.out.println("Mutable copy: " + mutable); // Output: [X, Y, Z]
    }
}

// List: [Apple, Banana, Cherry]
// Set contains 2: true
// Map: {one=1, two=2}

TryWithResourcesEnhancement.java 
package java9;

/**
 * <b>Try-With-Resources Enhancement</b>
 * <p>
 * In Java 7/8, the variable declared in a {@code try}-with-resources header had to be introduced
 * inside the {@code try(...)} parentheses. Java 9 relaxed this: any <em>effectively final</em>
 * {@code AutoCloseable} variable already in scope can be listed directly in the header without
 * re-declaring it. Multiple variables are separated by {@code ;}. Closing order is reversed
 * (last-opened, first-closed). Introduced in Java 9.
 *
 * @see <a href="https://docs.oracle.com/javase/9/language/effective-final-variables-try-resources.htm">Effectively Final Variables in Try-With-Resources</a>
 */
public class TryWithResourcesEnhancement {

    /** Simple AutoCloseable to illustrate resource lifecycle */
    static class Resource implements AutoCloseable {
        private final String name;

        Resource(String name) {
            this.name = name;
            System.out.println(name + " opened");
        }

        void use() {
            System.out.println(name + " in use");
        }

        @Override
        public void close() {
            System.out.println(name + " closed");
        }
    }

    public static void main(String[] args) {
        System.out.println("=== Before Java 9: must re-declare variable in try header ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 9: effectively-final variable used directly ===");
        demonstrateAfter();
    }

    /** Before Java 9: existing variable must be re-assigned to a new variable inside try(...) */
    static void demonstrateBefore() {
        Resource r1 = new Resource("R1");
        // Must declare a new variable 'res' even though r1 is the same object
        try (Resource res = r1) {
            res.use();
        }
        // Output:
        // R1 opened
        // R1 in use
        // R1 closed
    }

    /** Java 9: effectively-final variables used directly in try-with-resources */
    static void demonstrateAfter() {
        Resource r1 = new Resource("R1");
        Resource r2 = new Resource("R2");

        // r1 and r2 are effectively final (never reassigned) — use them directly
        // Multiple resources separated by ; — closed in reverse order: r2, then r1
        try (r1; r2) {
            r1.use();
            r2.use();
        }
        // Output:
        // R1 opened
        // R2 opened
        // R1 in use
        // R2 in use
        // R2 closed   ← reversed closing order
        // R1 closed

        System.out.println("--- single effectively-final variable ---");
        Resource r3 = new Resource("R3");
        try (r3) {
            r3.use();
        }
        // Output:
        // R3 opened
        // R3 in use
        // R3 closed
    }
}

// R1 in use
// R1 closed
//
// === Java 9: effectively-final variable used directly ===
// R1 opened

// R1 in use
// R2 in use
// R2 closed
// R1 closed
// --- single effectively-final variable ---
// R3 opened

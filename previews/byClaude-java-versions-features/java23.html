package java23;

import java.util.List;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * <b>Unnamed Variables and Patterns ({@code _})</b>
 * <p>
 * The underscore {@code _} marks a variable that is intentionally unused, signalling to
 * readers and tools that the value is deliberately discarded. It can be used in:
 * <ul>
 *   <li>Catch clauses: {@code catch (IOException _)}</li>
 *   <li>Enhanced for loops: {@code for (var _ : list)}</li>
 *   <li>Lambda parameters: {@code list.forEach(_ -> count++)}</li>
 *   <li>Pattern matching: {@code case Integer _} or {@code instanceof Point(int x, _)}</li>
 * </ul>
 * Multiple {@code _} declarations are allowed in the same scope (unlike named variables).
 * History: {@code _} was deprecated as an identifier in Java 8, made a compile error in
 * Java 9, previewed as unnamed variable in Java 21 (JEP 443) and Java 22; finalized in
 * Java 23 (JEP 456).
 *
 * @see <a href="https://openjdk.org/jeps/456">JEP 456 – Unnamed Variables and Patterns</a>
 */
public class UnnamedVariables {

    record Point(int x, int y) {}

    public static void main(String[] args) {
        System.out.println("=== Before Java 23: unused variable warnings ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 23: unnamed variable _ ===");
        demonstrateAfter();
    }

    /** Before Java 23: must name variables even when intentionally unused */
    static void demonstrateBefore() {
        // Catch clause — 'e' is unused but must be named; linters warn about it
        try {
            int result = Integer.parseInt("abc");
        } catch (NumberFormatException e) {
            // 'e' is ignored — but we had to give it a name
            System.out.println("Parse failed (exception details ignored)");
        }

        // For-each — element is unused, we only want to count iterations
        List<String> items = List.of("A", "B", "C");
        int count = 0;
        for (String ignored : items) { // 'ignored' is a common naming convention
            count++;
        }
        System.out.println("Count: " + count); // Output: 3

        // Lambda — parameter unused
        AtomicInteger lambdaCount = new AtomicInteger();
        items.forEach(element -> lambdaCount.incrementAndGet()); // 'element' unused
        System.out.println("Lambda count: " + lambdaCount.get()); // Output: 3
    }

    /** Java 23: _ communicates "intentionally unused" to readers and tools */
    static void demonstrateAfter() {
        // Catch clause — _ makes the intent explicit: the exception details don't matter
        try {
            int result = Integer.parseInt("abc");
        } catch (NumberFormatException _) {
            System.out.println("Parse failed (exception details ignored)"); // Output: Parse failed
        }

        // Multiple catch — each _ is independent (same scope, multiple declarations allowed)
        try {
            String s = null;
            s.length();
        } catch (NullPointerException _) {
            System.out.println("NullPointerException caught and ignored");
        } catch (Exception _) {
            System.out.println("Other exception caught and ignored");
        }

        // For-each — only counting, element value not needed
        List<String> items = List.of("A", "B", "C");
        int count = 0;
        for (var _ : items) {
            count++;
        }
        System.out.println("Count: " + count); // Output: 3

        // Lambda — parameter not used
        AtomicInteger lambdaCount = new AtomicInteger();
        items.forEach(_ -> lambdaCount.incrementAndGet());
        System.out.println("Lambda count: " + lambdaCount.get()); // Output: 3

        // Switch — case binds a type but the value is unused
        Object obj = 42;
        String desc = switch (obj) {
            case Integer _ -> "an Integer (value not needed)";
            case String  _ -> "a String (value not needed)";
            default        -> "something else";
        };
        System.out.println(desc); // Output: an Integer (value not needed)

        // Record pattern — extract only x, ignore y
        Object point = new Point(3, 7);
        if (point instanceof Point(int x, int _)) {
            System.out.println("x = " + x + " (y ignored)"); // Output: x = 3 (y ignored)
        }

        // Multiple _ declarations in same scope (not possible with named variables)
        int _ = 1; // unnamed local variable — valid
        String _ = "hello"; // second _ in same scope — also valid
        System.out.println("Multiple _ declarations in same scope: OK");
        // Named variables cannot be re-declared in the same scope, but _ is special.
    }
}

package java14;

/**
 * <b>Switch Expressions</b>
 * <p>
 * Switch expressions turn the {@code switch} construct from a statement into an expression
 * that returns a value. Key improvements:
 * <ul>
 *   <li>Arrow ({@code ->}) syntax eliminates fall-through and the need for {@code break}</li>
 *   <li>Multiple labels share one arm: {@code case "MON", "FRI" ->}</li>
 *   <li>{@code yield} returns a value from a block-body case arm</li>
 *   <li>The compiler enforces exhaustiveness (all cases must be covered)</li>
 * </ul>
 * Previewed in Java 12 (JEP 325) and Java 13 (JEP 354); finalized in Java 14 (JEP 361).
 *
 * @see <a href="https://openjdk.org/jeps/361">JEP 361 – Switch Expressions</a>
 */
public class SwitchExpressions {

    private enum Day { MON, TUE, WED, THU, FRI, SAT, SUN }
    private enum Season { SPRING, SUMMER, AUTUMN, WINTER }

    public static void main(String[] args) {
        System.out.println("=== Before Java 14: switch statement ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 14: switch expression ===");
        demonstrateAfter();
    }

    /** Before Java 14: switch statement — fall-through risk, mutable result variable */
    static void demonstrateBefore() {
        Day day = Day.WED;

        // Result must be assigned before the switch, then mutated inside it
        int numLetters;
        switch (day) {
            case MON:
            case FRI:
            case SUN:
                numLetters = 6;
                break;
            case TUE:
                numLetters = 7;
                break;
            case THU:
            case SAT:
                numLetters = 8;
                break;
            default:
                numLetters = 9; // WED, etc.
                break;
        }
        System.out.println(day + " has " + numLetters + " letters"); // Output: WED has 9 letters

        // Classic fall-through (intentional but error-prone):
        String type;
        switch (day) {
            case SAT:
            case SUN:
                type = "Weekend";
                break;
            default:
                type = "Weekday";
        }
        System.out.println(day + " is a " + type); // Output: WED is a Weekday
    }

    /** Java 14: switch expression — compact, safe, returns a value */
    static void demonstrateAfter() {
        Day day = Day.WED;

        // Arrow syntax — no fall-through, no break, multiple labels share one arm
        int numLetters = switch (day) {
            case MON, FRI, SUN -> 6;
            case TUE            -> 7;
            case THU, SAT       -> 8;
            case WED            -> 9;
        };
        System.out.println(day + " has " + numLetters + " letters"); // Output: WED has 9 letters

        // Block body with yield — used when multiple statements are needed
        String description = switch (day) {
            case MON -> "Start of the work week";
            case FRI -> "End of the work week";
            case SAT, SUN -> "Weekend!";
            default -> {
                String msg = day.name().charAt(0) + day.name().substring(1).toLowerCase();
                yield "Midweek: " + msg; // yield returns from the block
            }
        };
        System.out.println(description); // Output: Midweek: Wed

        // Switch expression with String selector
        String command = "start";
        int code = switch (command) {
            case "start"  -> 1;
            case "stop"   -> 2;
            case "pause"  -> 3;
            default       -> -1;
        };
        System.out.println("Command '" + command + "' code: " + code); // Output: 1

        // Used inline as an argument
        System.out.println("Season: " + getSeason(Day.FRI)); // Output: Season: (based on mapping)

        // Exhaustiveness: switch over enum without default is OK if all values are listed
        for (Season s : Season.values()) {
            String months = switch (s) {
                case SPRING -> "Mar–May";
                case SUMMER -> "Jun–Aug";
                case AUTUMN -> "Sep–Nov";
                case WINTER -> "Dec–Feb";
            };
            System.out.println(s + ": " + months);
        }
        // Output:
        // SPRING: Mar-May
        // SUMMER: Jun-Aug
        // AUTUMN: Sep-Nov
        // WINTER: Dec-Feb
    }

    static String getSeason(Day day) {
        return switch (day) {
            case MON, TUE, WED -> "Winter month (example)";
            case THU, FRI      -> "Spring month (example)";
            case SAT, SUN      -> "Weekend — season varies";
        };
    }
}

// WED has 9 letters
// WED is a Weekday

// WED has 9 letters
// Midweek: Wed

// Season: Spring month (example)
// SPRING: Mar-May

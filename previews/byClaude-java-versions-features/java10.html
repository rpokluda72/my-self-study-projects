package java10;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * <b>Local Variable Type Inference ({@code var})</b>
 * <p>
 * The {@code var} keyword lets the compiler infer the type of a local variable from its
 * initializer, reducing repetition in declarations. Java remains statically typed — the
 * inferred type is fixed at compile time. {@code var} works in local variable declarations,
 * enhanced {@code for} loops, and {@code try}-with-resources. It cannot be used for fields,
 * method parameters, return types, or where the initializer is {@code null} or an array
 * literal without a type. Introduced in Java 10 (JEP 286).
 *
 * @see <a href="https://openjdk.org/jeps/286">JEP 286 – Local-Variable Type Inference</a>
 */
public class LocalVariableTypeInference {

    public static void main(String[] args) {
        System.out.println("=== Before Java 10: explicit type declarations ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 10: var keyword ===");
        demonstrateAfter();
    }

    /** Before Java 10: type is repeated on both sides of the declaration */
    static void demonstrateBefore() {
        ArrayList<String> list = new ArrayList<String>();
        list.add("hello");
        list.add("world");
        System.out.println("List: " + list); // Output: [hello, world]

        Map<String, List<Integer>> map = new HashMap<String, List<Integer>>();
        map.put("primes", new ArrayList<>(List.of(2, 3, 5)));
        System.out.println("Map: " + map); // Output: {primes=[2, 3, 5]}

        for (String item : list) {
            System.out.println("  item: " + item);
        }
    }

    /** Java 10: var — type inferred from the right-hand side */
    static void demonstrateAfter() {
        // Simple types
        var number = 42;           // inferred: int
        var text   = "Hello";      // inferred: String
        var pi     = 3.14;         // inferred: double

        System.out.println("number: " + number + " (" + ((Object) number).getClass().getSimpleName() + ")"); // Output: number: 42 (Integer)
        System.out.println("text: "   + text);  // Output: text: Hello
        System.out.println("pi: "     + pi);    // Output: pi: 3.14

        // Collection declaration — right side still explicit, left side clean
        var list = new ArrayList<String>();
        list.add("hello");
        list.add("world");
        System.out.println("List type: " + list.getClass().getSimpleName()); // Output: ArrayList

        // Nested generics — var especially helpful
        var map = new HashMap<String, List<Integer>>();
        map.put("primes", new ArrayList<>(List.of(2, 3, 5)));
        System.out.println("Map: " + map); // Output: {primes=[2, 3, 5]}

        // var in enhanced for loop
        for (var item : list) {
            System.out.println("  item: " + item); // Output: hello / world
        }

        // Anonymous type capture — var exposes the anonymous class's members
        // (not possible with an explicit declared type)
        var anon = new Object() {
            String name  = "Claude";
            int    version = 4;
        };
        System.out.println("anon.name: "    + anon.name);    // Output: Claude
        System.out.println("anon.version: " + anon.version); // Output: 4

        // Limitations (shown as comments — these would NOT compile):
        // var field = "nope";          // ← not allowed for instance/static fields
        // var nothing;                 // ← no initializer
        // var x = null;               // ← null gives no type info
        // void method(var param) {}    // ← not allowed for method parameters
        System.out.println("(var limitations shown in comments above)");
    }
}

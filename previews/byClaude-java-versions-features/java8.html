package java8;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;
import java.util.function.BiFunction;
import java.util.function.Function;

/**
 * <b>Lambda Expressions</b>
 * <p>
 * Lambda expressions allow writing anonymous functions inline as {@code (parameters) -> body},
 * replacing verbose anonymous class syntax. They are the foundation of Java's functional
 * programming support and work wherever a functional interface is expected.
 * Introduced in Java 8.
 *
 * @see <a href="https://openjdk.org/projects/lambda/">Project Lambda</a>
 */
public class LambdaExpressions {

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: anonymous classes ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: lambda expressions ===");
        demonstrateAfter();
    }

    /** Before Java 8: anonymous classes for single-method interfaces */
    static void demonstrateBefore() {
        List<String> names = new ArrayList<>(Arrays.asList("Banana", "Apple", "Cherry"));

        // Comparator as anonymous class
        Comparator<String> comparator = new Comparator<String>() {
            @Override
            public int compare(String a, String b) {
                return a.compareTo(b);
            }
        };
        Collections.sort(names, comparator);
        System.out.println("Sorted: " + names); // Output: [Apple, Banana, Cherry]

        // Runnable as anonymous class
        Runnable task = new Runnable() {
            @Override
            public void run() {
                System.out.println("Running with anonymous class");
            }
        };
        task.run(); // Output: Running with anonymous class
    }

    /** Java 8: same behaviour expressed with lambda expressions */
    static void demonstrateAfter() {
        List<String> names = new ArrayList<>(Arrays.asList("Banana", "Apple", "Cherry"));

        // Comparator as lambda
        Comparator<String> comparator = (a, b) -> a.compareTo(b);
        Collections.sort(names, comparator);
        System.out.println("Sorted: " + names); // Output: [Apple, Banana, Cherry]

        // Runnable as lambda
        Runnable task = () -> System.out.println("Running with lambda");
        task.run(); // Output: Running with lambda

        // Block-body lambda (multiple statements)
        BiFunction<Integer, Integer, Integer> add = (x, y) -> {
            int sum = x + y;
            System.out.println("sum = " + sum); // Output: sum = 7
            return sum;
        };
        add.apply(3, 4);

        // Expression lambda returning a value
        Function<Integer, Integer> square = x -> x * x;
        System.out.println("square(9) = " + square.apply(9)); // Output: square(9) = 81

        // Lambda passed directly as argument (inline)
        names.sort((a, b) -> b.compareTo(a)); // reverse sort
        System.out.println("Reverse sorted: " + names); // Output: [Cherry, Banana, Apple]
    }
}

package java8;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

/**
 * <b>Stream API</b>
 * <p>
 * The Stream API provides a declarative pipeline model for processing sequences of elements
 * using operations like {@code filter}, {@code map}, {@code reduce}, and {@code collect},
 * without explicit loops. Streams are lazy — intermediate operations execute only when a
 * terminal operation is invoked. Introduced in Java 8 as {@code java.util.stream}.
 *
 * @see <a href="https://docs.oracle.com/en/java/se/8/docs/api/java/util/stream/package-summary.html">java.util.stream package</a>
 */
public class StreamApi {

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: explicit loops ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: Stream API ===");
        demonstrateAfter();
    }

    /** Before Java 8: imperative loops for filtering and transforming collections */
    static void demonstrateBefore() {
        List<String> names = Arrays.asList("Alice", "Bob", "Diana", "Elena", "Ed");

        // Filter names longer than 4 chars, convert to uppercase
        List<String> result = new ArrayList<>();
        for (String name : names) {
            if (name.length() > 4) {
                result.add(name.toUpperCase());
            }
        }
        System.out.println("Filtered names: " + result); // Output: [ALICE, DIANA, ELENA]

        // Count even numbers
        List<Integer> numbers = Arrays.asList(1, 2, 3, 4, 5, 6);
        int evenCount = 0;
        for (int n : numbers) {
            if (n % 2 == 0) evenCount++;
        }
        System.out.println("Even count: " + evenCount); // Output: Even count: 3

        // Sum all numbers
        int sum = 0;
        for (int n : numbers) sum += n;
        System.out.println("Sum: " + sum); // Output: Sum: 21
    }

    /** Java 8: same operations expressed as Stream pipelines */
    static void demonstrateAfter() {
        List<String> names = Arrays.asList("Alice", "Bob", "Diana", "Elena", "Ed");
        List<Integer> numbers = Arrays.asList(1, 2, 3, 4, 5, 6);

        // filter + map + collect
        List<String> result = names.stream()
                .filter(s -> s.length() > 4)
                .map(String::toUpperCase)
                .collect(Collectors.toList());
        System.out.println("Filtered names: " + result); // Output: [ALICE, DIANA, ELENA]

        // count terminal operation
        long evenCount = numbers.stream()
                .filter(n -> n % 2 == 0)
                .count();
        System.out.println("Even count: " + evenCount); // Output: Even count: 3

        // reduce
        int sum = numbers.stream()
                .reduce(0, Integer::sum);
        System.out.println("Sum: " + sum); // Output: Sum: 21

        // sorted + distinct
        List<Integer> dupes = Arrays.asList(3, 1, 4, 1, 5, 9, 2, 6, 5);
        List<Integer> sortedUnique = dupes.stream()
                .distinct()
                .sorted()
                .collect(Collectors.toList());
        System.out.println("Sorted unique: " + sortedUnique); // Output: [1, 2, 3, 4, 5, 6, 9]

        // forEach
        System.out.print("forEach: ");
        names.stream()
                .filter(s -> s.length() <= 3)
                .forEach(s -> System.out.print(s + " ")); // Output: Bob Ed
        System.out.println();

        // anyMatch / allMatch / noneMatch
        boolean anyShort = names.stream().anyMatch(s -> s.length() <= 2);
        boolean allShort = names.stream().allMatch(s -> s.length() <= 5);
        System.out.println("anyMatch length<=2: " + anyShort); // Output: false
        System.out.println("allMatch length<=5: " + allShort); // Output: false

        // findFirst returns Optional
        Optional<String> first = names.stream()
                .filter(s -> s.startsWith("D"))
                .findFirst();
        first.ifPresent(s -> System.out.println("First starting with D: " + s)); // Output: Diana
    }
}

package java8;

import java.util.function.BiFunction;
import java.util.function.Consumer;
import java.util.function.Function;
import java.util.function.Predicate;
import java.util.function.Supplier;

/**
 * <b>Functional Interfaces</b>
 * <p>
 * A functional interface has exactly one abstract method, making it compatible with lambda
 * expressions. The {@code java.util.function} package provides a standard vocabulary of
 * ready-made functional interfaces: {@code Predicate} (test), {@code Function} (transform),
 * {@code Consumer} (consume), {@code Supplier} (produce), and {@code BiFunction} (combine two
 * inputs). They compose via {@code and}, {@code or}, {@code negate}, {@code andThen}, and
 * {@code compose}. Introduced in Java 8.
 *
 * @see <a href="https://docs.oracle.com/javase/8/docs/api/java/util/function/package-summary.html">java.util.function package</a>
 */
public class FunctionalInterfaces {

    // Custom functional interface
    @FunctionalInterface
    interface Transformer<T> {
        T transform(T input);
        // Only one abstract method allowed; default/static methods are permitted
        default Transformer<T> andThen(Transformer<T> next) {
            return input -> next.transform(this.transform(input));
        }
    }

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: custom single-method interfaces ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: java.util.function interfaces ===");
        demonstrateAfter();
    }

    /** Before Java 8: manual single-method interface + anonymous class */
    static void demonstrateBefore() {
        // Had to define a custom interface for every operation
        interface StringChecker {
            boolean check(String s);
        }

        StringChecker isLong = new StringChecker() {
            @Override
            public boolean check(String s) {
                return s.length() > 5;
            }
        };
        System.out.println("isLong(\"Hello\"): " + isLong.check("Hello"));       // Output: false
        System.out.println("isLong(\"HelloWorld\"): " + isLong.check("HelloWorld")); // Output: true
    }

    /** Java 8: standard functional interfaces from java.util.function */
    static void demonstrateAfter() {
        // Predicate<T> — boolean test(T t)
        Predicate<String> isLong = s -> s.length() > 5;
        System.out.println("isLong(\"Hello\"): " + isLong.test("Hello"));       // Output: false
        System.out.println("isLong(\"HelloWorld\"): " + isLong.test("HelloWorld")); // Output: true

        // Predicate composition: and, or, negate
        Predicate<String> startsWithH = s -> s.startsWith("H");
        Predicate<String> longAndH = isLong.and(startsWithH);
        Predicate<String> longOrH = isLong.or(startsWithH);
        System.out.println("long AND starts with H (\"HelloWorld\"): " + longAndH.test("HelloWorld")); // Output: true
        System.out.println("long OR  starts with H (\"Hi\"): " + longOrH.test("Hi"));                 // Output: true
        System.out.println("NOT isLong (\"Hello\"): " + isLong.negate().test("Hello"));               // Output: true

        // Function<T,R> — R apply(T t)
        Function<String, Integer> length = String::length;
        System.out.println("length(\"Hello\"): " + length.apply("Hello")); // Output: 5

        // Function composition: andThen (f then g), compose (g then f)
        Function<Integer, String> intToStr = n -> "Number=" + n;
        Function<String, String> lengthThenStr = length.andThen(intToStr);
        System.out.println("andThen: " + lengthThenStr.apply("Hello")); // Output: Number=5

        // Consumer<T> — void accept(T t)
        Consumer<String> printer = System.out::println;
        printer.accept("Consumer prints: Hello"); // Output: Consumer prints: Hello

        // Consumer chaining with andThen
        Consumer<String> upperPrinter = s -> System.out.println("Upper: " + s.toUpperCase());
        printer.andThen(upperPrinter).accept("world");
        // Output: world
        // Output: Upper: WORLD

        // Supplier<T> — T get()
        Supplier<String> greeting = () -> "Hello, World!";
        System.out.println("Supplier: " + greeting.get()); // Output: Hello, World!

        // BiFunction<T,U,R> — R apply(T t, U u)
        BiFunction<Integer, Integer, Integer> multiply = (a, b) -> a * b;
        System.out.println("3 * 4 = " + multiply.apply(3, 4)); // Output: 3 * 4 = 12

        // Custom @FunctionalInterface with default method
        Transformer<String> upper = String::toUpperCase;
        Transformer<String> exclaim = s -> s + "!";
        Transformer<String> shout = upper.andThen(exclaim);
        System.out.println("shout(\"hello\"): " + shout.transform("hello")); // Output: HELLO!
    }
}

package java8;

import java.util.Arrays;
import java.util.Comparator;
import java.util.List;
import java.util.function.Function;
import java.util.function.Supplier;
import java.util.stream.Collectors;

/**
 * <b>Method References</b>
 * <p>
 * Method references are shorthand lambda expressions that delegate to an existing method using
 * the {@code ::} operator. There are four forms:
 * <ul>
 *   <li><b>Static:</b> {@code ClassName::staticMethod} — e.g. {@code Integer::parseInt}</li>
 *   <li><b>Bound instance:</b> {@code instance::method} — e.g. {@code System.out::println}</li>
 *   <li><b>Unbound instance:</b> {@code ClassName::instanceMethod} — e.g. {@code String::toUpperCase}</li>
 *   <li><b>Constructor:</b> {@code ClassName::new} — e.g. {@code StringBuilder::new}</li>
 * </ul>
 * Introduced in Java 8.
 *
 * @see <a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">Method References tutorial</a>
 */
public class MethodReferences {

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: explicit lambdas ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: method references ===");
        demonstrateAfter();
    }

    /** Before Java 8: explicit lambda expressions for every method delegation */
    static void demonstrateBefore() {
        List<String> words = Arrays.asList("hello", "world", "java");

        // Explicit lambda instead of unbound instance reference
        List<String> upper = words.stream()
                .map(s -> s.toUpperCase())
                .collect(Collectors.toList());
        System.out.println("Uppercase: " + upper); // Output: [HELLO, WORLD, JAVA]

        // Explicit lambda instead of bound instance reference
        words.forEach(s -> System.out.println(s));
        // Output: hello / world / java

        // Explicit lambda instead of static reference
        List<String> numStrs = Arrays.asList("1", "2", "3");
        List<Integer> nums = numStrs.stream()
                .map(s -> Integer.parseInt(s))
                .collect(Collectors.toList());
        System.out.println("Parsed: " + nums); // Output: [1, 2, 3]
    }

    /** Java 8: method references — more readable than equivalent lambdas */
    static void demonstrateAfter() {
        List<String> words = Arrays.asList("hello", "world", "java");

        // Unbound instance reference: ClassName::instanceMethod
        // Equivalent lambda: s -> s.toUpperCase()
        List<String> upper = words.stream()
                .map(String::toUpperCase)
                .collect(Collectors.toList());
        System.out.println("Uppercase: " + upper); // Output: [HELLO, WORLD, JAVA]

        // Bound instance reference: instance::method
        // Equivalent lambda: s -> System.out.println(s)
        words.forEach(System.out::println);
        // Output: hello / world / java

        // Static reference: ClassName::staticMethod
        // Equivalent lambda: s -> Integer.parseInt(s)
        List<String> numStrs = Arrays.asList("1", "2", "3");
        List<Integer> nums = numStrs.stream()
                .map(Integer::parseInt)
                .collect(Collectors.toList());
        System.out.println("Parsed: " + nums); // Output: [1, 2, 3]

        // Static reference: String::valueOf converts int to String
        List<Integer> numbers = Arrays.asList(10, 20, 30);
        List<String> asStrings = numbers.stream()
                .map(String::valueOf)
                .collect(Collectors.toList());
        System.out.println("As strings: " + asStrings); // Output: [10, 20, 30]

        // Constructor reference: ClassName::new
        // Equivalent lambda: s -> new StringBuilder(s)
        Function<String, StringBuilder> sbFactory = StringBuilder::new;
        StringBuilder sb = sbFactory.apply("Hello");
        System.out.println("StringBuilder: " + sb); // Output: Hello

        // Supplier constructor reference (no-arg constructor)
        Supplier<StringBuilder> sbSupplier = StringBuilder::new;
        System.out.println("Empty SB: '" + sbSupplier.get() + "'"); // Output: ''

        // Used with Comparator.comparing — sort by string length
        List<String> sorted = words.stream()
                .sorted(Comparator.comparing(String::length))
                .collect(Collectors.toList());
        System.out.println("Sorted by length: " + sorted); // Output: [java, hello, world]
    }
}

package java8;

/**
 * <b>Default and Static Methods in Interfaces</b>
 * <p>
 * Java 8 allowed interfaces to declare {@code default} methods with a full implementation and
 * {@code static} methods. This lets library authors add new methods to existing interfaces
 * without breaking all implementing classes. Default methods can be overridden; when two
 * interfaces provide the same default method, the implementing class must resolve the conflict
 * explicitly using {@code InterfaceA.super.method()}. Introduced in Java 8.
 *
 * @see <a href="https://docs.oracle.com/javase/tutorial/java/IandI/defaultmethods.html">Default Methods tutorial</a>
 */
public class DefaultMethods {

    // ── Before: shared logic required an external utility class ─────────────

    interface OldVehicle {
        String getBrand();
        // No default methods — every implementor must handle startEngine separately,
        // or rely on an external utility class like VehicleUtils.
    }

    static class VehicleUtils {
        static void startEngine(OldVehicle v) {
            System.out.println("[util] " + v.getBrand() + " engine started");
        }
    }

    // ── After: default and static methods on the interface ──────────────────

    interface Vehicle {
        String getBrand();

        /** Default method — provided by the interface, can be overridden */
        default void startEngine() {
            System.out.println(getBrand() + " engine started");
        }

        default void stopEngine() {
            System.out.println(getBrand() + " engine stopped");
        }

        /** Static factory on the interface — available without an instance */
        static Vehicle of(String brand) {
            return () -> brand; // lambda satisfies the single abstract method
        }
    }

    // Diamond problem: two interfaces both define the same default method
    interface A {
        default String describe() { return "Interface A"; }
    }

    interface B {
        default String describe() { return "Interface B"; }
    }

    // Implementing class must override and disambiguate
    static class AB implements A, B {
        @Override
        public String describe() {
            return A.super.describe(); // explicitly choose A's implementation
        }
    }

    // A class that overrides the default method
    static class ElectricCar implements Vehicle {
        private final String brand;
        ElectricCar(String brand) { this.brand = brand; }

        @Override
        public String getBrand() { return brand; }

        @Override
        public void startEngine() {
            // Electric cars are silent — override the default behaviour
            System.out.println(brand + " motor engaged (no engine noise)");
        }
    }

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: utility class for shared behaviour ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: default and static interface methods ===");
        demonstrateAfter();
    }

    /** Before Java 8: shared logic lives in a separate utility class */
    static void demonstrateBefore() {
        OldVehicle car = () -> "Toyota";
        VehicleUtils.startEngine(car); // Output: [util] Toyota engine started
    }

    /** Java 8: default methods provide shared behaviour directly on the interface */
    static void demonstrateAfter() {
        // Static factory method on the interface
        Vehicle toyota = Vehicle.of("Toyota");
        toyota.startEngine(); // Output: Toyota engine started
        toyota.stopEngine();  // Output: Toyota engine stopped

        // Subclass overrides the default
        Vehicle tesla = new ElectricCar("Tesla");
        tesla.startEngine(); // Output: Tesla motor engaged (no engine noise)
        tesla.stopEngine();  // Output: Tesla engine stopped  (uses default)

        // Diamond problem resolution
        AB ab = new AB();
        System.out.println("Diamond resolved: " + ab.describe()); // Output: Interface A
    }
}

//
// === Java 8: default and static interface methods ===

// Toyota engine stopped
// Tesla motor engaged (no engine noise)

// Diamond resolved: Interface A

package java8;

import java.util.Optional;

/**
 * <b>Optional&lt;T&gt;</b>
 * <p>
 * {@code Optional<T>} is a container that either holds a non-null value or is explicitly empty.
 * It forces callers to handle the absent case, replacing {@code null} returns that silently
 * cause {@code NullPointerException}. Use it as a return type for methods that may produce no
 * result; avoid it as a field or parameter type. Introduced in Java 8.
 *
 * @see <a href="https://docs.oracle.com/javase/8/docs/api/java/util/Optional.html">Optional Javadoc</a>
 */
public class OptionalClass {

    public static void main(String[] args) {
        System.out.println("=== Before Java 8: nullable return values ===");
        demonstrateBefore();
        System.out.println();
        System.out.println("=== Java 8: Optional<T> ===");
        demonstrateAfter();
    }

    // ── Simulated repository ─────────────────────────────────────────────────

    /** Pre-Java-8 style: returns null when not found */
    static String findUserOld(String name) {
        if ("Alice".equals(name)) return "Alice (age 30)";
        return null; // caller must remember to check!
    }

    /** Java 8 style: Optional makes absence explicit in the signature */
    static Optional<String> findUser(String name) {
        if ("Alice".equals(name)) return Optional.of("Alice (age 30)");
        return Optional.empty();
    }

    /** Before Java 8: manual null checks everywhere */
    static void demonstrateBefore() {
        String user = findUserOld("Alice");
        if (user != null) {
            System.out.println("Found: " + user); // Output: Found: Alice (age 30)
        }

        String missing = findUserOld("Bob");
        if (missing != null) {
            System.out.println(missing.toUpperCase());
        } else {
            System.out.println("User not found"); // Output: User not found
        }

        // Easy to forget the null check — NullPointerException risk:
        // String danger = findUserOld("Bob").toUpperCase(); // would throw NPE
    }

    /** Java 8: Optional API for safe, readable absent-value handling */
    static void demonstrateAfter() {
        // Creating Optionals
        Optional<String> present = Optional.of("Alice");
        Optional<String> empty   = Optional.empty();
        Optional<String> nullable = Optional.ofNullable(null); // same as empty()

        System.out.println("present: " + present);   // Output: Optional[Alice]
        System.out.println("empty: "   + empty);     // Output: Optional.empty

        // Checking presence
        System.out.println("isPresent: " + present.isPresent()); // Output: true
        System.out.println("isEmpty:   " + empty.isEmpty());     // Output: true

        // Safe extraction
        System.out.println("orElse: " + empty.orElse("default"));          // Output: default
        System.out.println("orElseGet: " + empty.orElseGet(() -> "lazy")); // Output: lazy

        try {
            empty.orElseThrow(() -> new IllegalStateException("not found"));
        } catch (IllegalStateException e) {
            System.out.println("orElseThrow: " + e.getMessage()); // Output: not found
        }

        // Transformations — do nothing when empty, no NPE
        Optional<String> upper = present.map(String::toUpperCase);
        System.out.println("map: " + upper); // Output: Optional[ALICE]

        Optional<String> filtered = present.filter(s -> s.length() > 10);
        System.out.println("filter (len>10): " + filtered); // Output: Optional.empty

        // flatMap — avoids Optional<Optional<T>>
        Optional<Optional<String>> nested = present.map(s -> Optional.of(s + "!"));
        Optional<String> flat = present.flatMap(s -> Optional.of(s + "!"));
        System.out.println("nested: " + nested); // Output: Optional[Optional[Alice!]]
        System.out.println("flat:   " + flat);   // Output: Optional[Alice!]

        // ifPresent — consume only when value exists
        present.ifPresent(s -> System.out.println("ifPresent: " + s)); // Output: ifPresent: Alice
        empty.ifPresent(s -> System.out.println("never printed"));

        // Full example: safe lookup chain
        findUser("Alice")
                .map(String::toUpperCase)
                .ifPresent(s -> System.out.println("Found: " + s)); // Output: Found: ALICE (AGE 30)

        findUser("Bob")
                .map(String::toUpperCase)
                .ifPresentOrElse(
                        s -> System.out.println("Found: " + s),
                        () -> System.out.println("Bob not found") // Output: Bob not found
                );
    }
}

package test.hash_set;

import java.util.*;

public class TestHashTable {
    public static void main(String[] args) {
//----------Hashtable -------------------------
        Hashtable<Integer,String> ht=new Hashtable<Integer,String>();
        ht.put(101," ajay");
        ht.put(101,"Vijay");
        ht.put(102,"Ravi");
        ht.put(103,"Rahul");
        System.out.println("-------------Hash table--------------");
        for (Map.Entry m:ht.entrySet()) {
            System.out.println(m.getKey()+" "+m.getValue());
        }
        // -------------Hash table--------------
        // 103 Rahul
        // 102 Ravi
        // 101 Vijay

        //----------------HashMap--------------------------------
        HashMap<Integer,String> hm=new HashMap<>();
        hm.put(100,"Amit");
        hm.put(104,"Amit");
        hm.put(100,"Vijay");
        hm.put(102,"Rahul");
        System.out.println("-----------Hash map-----------");
        for (Map.Entry m:hm.entrySet()) {
            System.out.println(m.getKey()+" "+m.getValue());
        }
        // -----------Hash map-----------
        // 100 Vijay
        // 102 Rahul
        // 104 Amit

        //----------------HashSet--------------------------------
        System.out.println("\n -----------HashSet-----------");
        String[] names = { "Gabriel", "John", "Gabriel", "Bob" };
        ArrayList<String> namesArray = new ArrayList<>(List.of(names));
        HashSet<String> hs =new HashSet<>(List.of(names));
        hs.add("Gabriel"); // Duplicate element, will not be added
        HashSet<String> hs2 =new HashSet<>(namesArray);
        System.out.println("hs="+ hs); // hs=[Bob, John, Gabriel]
        System.out.println("hs2="+ hs2); // hs2=[Bob, John, Gabriel]

        System.out.println("- hs.iterator() - iter.hasNext() -");
        for (String h : hs) {
            System.out.println(h);
        }
        // Bob
        // John
        // Gabriel

        System.out.println("- hs.forEach -");
        hs.forEach(System.out::println);
        // Bob
        // John
        // Gabriel

        List<String> sortedList = new ArrayList<>(hs);
        Collections.sort(sortedList);
        System.out.println("sortedList="+ sortedList); // sortedList=[Bob, Gabriel, John]

        HashSet<Game> hsGame =new HashSet<>();
        hsGame.add(new Game("Game A", 2));
        hsGame.add(new Game("Game B", 3));
        hsGame.add(new Game("Game C", 1));
        System.out.println("hsGame="+ hsGame);
        // hsGame=[Game{name='Game B', ranking=3}, Game{name='Game A', ranking=2}, Game{name='Game C', ranking=1}]

        List<Game> sortedGames = new ArrayList<>(hsGame);
        Collections.sort(sortedGames, new Comparator<Game>() {
            @Override
            public int compare(Game g1, Game g2) {
                return Integer.compare(g1.getRanking(), g2.getRanking());
            }
        });
        System.out.println("sortedGames="+ sortedGames);
        // sortedGames=[Game{name='Game C', ranking=1}, Game{name='Game A', ranking=2}, Game{name='Game B', ranking=3}]

        hsGame.add(new Game("Game C", 3));
        System.out.println("after dupl.add : hsGame.size()="+ hsGame.size());
        // after dupl.add : hsGame.size()=4
        System.out.println("hsGame="+ hsGame);
        // hsGame=[Game{name='Game C', ranking=1}, Game{name='Game B', ranking=3}, Game{name='Game A', ranking=2}, Game{name='Game C', ranking=3}]

        HashSet<Game2> hsGame2 =new HashSet<>();
        hsGame2.add(new Game2("Game A", 2));
        hsGame2.add(new Game2("Game B", 3));
        hsGame2.add(new Game2("Game C", 1));
        System.out.println("hsGame2="+ hsGame2);
        // hsGame2=[Game{name='Game B', ranking=3}, Game{name='Game C', ranking=1}, Game{name='Game A', ranking=2}]

        // Sorting HashSet using List
        List<Game2> list = new ArrayList<Game2>(hsGame2);
        Collections.sort(list);
        System.out.println("HashSet sorted using List="+ list);
        // HashSet sorted using List=[Game{name='Game C', ranking=1}, Game{name='Game A', ranking=2}, Game{name='Game B', ranking=3}]

        // Sorting HashSet using TreeSet
        TreeSet<Game2> treeSet2 = new TreeSet<Game2>(hsGame2);
        System.out.println("HashSet sorted using TreeSet="+ treeSet2);
        // HashSet sorted using TreeSet=[Game{name='Game C', ranking=1}, Game{name='Game A', ranking=2}, Game{name='Game B', ranking=3}]


        //----------------TreeSet--------------------------------
        System.out.println("\n -----------TreeSet-----------");
        TreeSet<String> treeSet = new TreeSet<>();
        treeSet.add("banana");
        treeSet.add("apple");
        treeSet.add("cherry");
        // treeSet.add(null); // Will throw NullPointerException
        System.out.println("TreeSet: " + treeSet);
        // TreeSet: [apple, banana, cherry]

        TreeSet<Game2> treeSetGame = new TreeSet<>();
        // public class Game2 implements Comparable<Game2>{
        // @Override public int compareTo(Game2 o) { return this.ranking - o.ranking; }
        treeSetGame.add(new Game2("Game A", 2));
        treeSetGame.add(new Game2("Game B", 3));
        treeSetGame.add(new Game2("Game C", 1));
        System.out.println("treeSetGame: " + treeSetGame);
        // treeSetGame: [Game{name='Game C', ranking=1}, Game{name='Game A', ranking=2}, Game{name='Game B', ranking=3}]
    }
}

package test.hash_set;

import java.util.Arrays;
import java.util.List;
import java.util.StringJoiner;
import java.util.stream.Collectors;

public class TestStringJoiner {
    public static void main(String[] args) {
// https://www.geeksforgeeks.org/java/stringjoiner-class-in-java/
        //https://neesri.medium.com/understanding-java-8-stringjoiner-string-join-and-collectors-joining-5d3b8cdd1551
        //https://mkyong.com/java8/java-8-stringjoiner-example/

        // Create a StringJoiner with a delimiter
        StringJoiner sj = new StringJoiner(", ");

        // Add strings which you need to join
        sj.add("geeks").add("for").add("geeks");
        System.out.println("Joined String: " + sj);
        // Joined String: geeks, for, geeks

        String[] s1 = {"Geeks", "for", "Geeks"};

        // Create StringJoiner with a delimiter
        StringJoiner sjb = new StringJoiner(", ");

        // Adding array elements to StringJoiner
        for (String s2 : s1) {
            sjb.add(s2);
        }
        System.out.println("sjb=" + sjb);
        // sjb=Geeks, for, Geeks

        // add() method
        StringJoiner sj1 = new StringJoiner(", ");
        sj1.add("geeks").add("for").add("geeks");
        System.out.println("StringJoiner with elements: " + sj1);
        // StringJoiner with elements: geeks, for, geeks

        // setEmptyValue() method
        StringJoiner sj2 = new StringJoiner(", ");
        sj2.setEmptyValue("No elements to join");
        System.out.println("Empty StringJoiner: " + sj2);
        // Empty StringJoiner: No elements to join

        // toString() method
        StringJoiner sj3 = new StringJoiner(" | ", "[", "]");
        sj3.add("geeks").add("for").add("geeks");
        System.out.println("StringJoiner with prefix and suffix: " + sj3);
        // StringJoiner with prefix and suffix: [geeks | for | geeks]

        // Checking the length of the joined string
        StringJoiner sj4 = new StringJoiner(", ");
        sj4.add("geeks").add("for").add("geeks");
        System.out.println("Length of joined string: " + sj4.length());
        // Length of joined string: 17

        /** String.join */
        // Using varargs
        String result = String.join(", ", "Java", "Python", "C++");
        System.out.println("result=" + result);
        // result=Java, Python, C++

        // Using a collection
        var languages = Arrays.asList("HTML", "CSS", "JavaScript");
        String result2 = String.join(" | ", languages);
        System.out.println("result2=" + result2);
        // result2=HTML | CSS | JavaScript

        /** Collectors.joining */
        List<String> fruits = Arrays.asList("Mango", "Orange", "Grapes");

        // Join with a delimiter
        String result3 = fruits.stream().collect(Collectors.joining(", "));
        System.out.println("result3=" + result3);
        // result3=Mango, Orange, Grapes

        // Join with delimiter, prefix, and suffix
        String resultWithPrefixSuffix = fruits.stream()
                .collect(Collectors.joining(", ", "{", "}"));
        System.out.println("resultWithPrefixSuffix=" + resultWithPrefixSuffix);
        // resultWithPrefixSuffix={Mango, Orange, Grapes}

        // Join String by a delimiter
        StringJoiner sjc1 = new StringJoiner(",");
        sjc1.add("aaa");
        sjc1.add("bbb");
        sjc1.add("ccc");
        String resultc1 = sjc1.toString(); //aaa,bbb,ccc
        System.out.println("resultc1=" + resultc1);
        // resultc1=aaa,bbb,ccc

        // Join String by a delimiter and starting with a supplied prefix and ending with a supplied suffix
        StringJoiner sjc2 = new StringJoiner("/", "prefix-", "-suffix");
        sj.add("2016");
        sj.add("02");
        sj.add("26");
        String resultc2 = sjc2.toString();
        System.out.println("resultc2=" + resultc2);
        // resultc2=prefix--suffix

        // Join String by a delimiter.
        String resultc3 = String.join("-", "2015", "10", "31" );
        System.out.println("resultc3=" + resultc3);
        // resultc3=2015-10-31

        // Join a List by a delimiter
        List<String> list = Arrays.asList("java", "python", "nodejs", "ruby");
        String resultc4 = String.join(", ", list);
        System.out.println("resultc4=" + resultc4);
        // resultc4=java, python, nodejs, ruby

        // Join List<String> example.
        List<String> listc5 = Arrays.asList("java", "python", "nodejs", "ruby");
        String resultc5 = listc5.stream().map(x -> x).collect(Collectors.joining(" | "));
        System.out.println("resultc5=" + resultc5);
        // resultc5=java | python | nodejs | ruby

        List<Game> listGame = Arrays.asList(
                new Game("Dragon Blaze", 5),
                new Game("Angry Bird", 5),
                new Game("Candy Crush", 5)
        );

        String resultGame = listGame.stream().map(x -> x.getName())
                .collect(Collectors.joining(", ", "{", "}"));
        System.out.println("resultGame=" + resultGame);
        // resultGame={Dragon Blaze, Angry Bird, Candy Crush}
    }

}

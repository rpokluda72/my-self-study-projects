public class TestArrays {

    static void main(String[] args) {
        test1();
        test2();
    }

    static void test1() {
        String[] arrayOfStrings = new String[3];
        arrayOfStrings[0] = "aaa";
        arrayOfStrings[2] = "ccc";
        System.out.println("arrayOfStrings : " + Arrays.toString(arrayOfStrings));
        // arrayOfStrings : [aaa, null, ccc]

        String[] arrayOfStrings2 = {"AlgoCademy", "coding", "static arrays"};
        System.out.println("arrayOfStrings2 : " + Arrays.toString(arrayOfStrings2));
        // arrayOfStrings2 : [AlgoCademy, coding, static arrays]
    }

    static void test2() {
        double[] temperatures = {72.5, 75.0, 68.4, 70.2, 69.8};
        double sum = 0;
        for (double temp : temperatures) {
            sum += temp;
        }
        double average = sum / temperatures.length;
        System.out.println("Average Temperature: " + average);
        // Average Temperature: 71.18

        // Create and initialize an array of integers
        int[] numbers = {10, 20, 30, 40, 50};
        // To convert the elements as List
        System.out.println("Integer Array as List: " + List.of(numbers)); // Integer Array as List: [[I@5674cd4d]

        // Print the elements of the array
        System.out.println("Array elements:");
        for (int number : numbers) {
            System.out.println(number);
        }
        //        Array elements:
        //        10
        //        20
        //        30
        //        40
        //        50

        // Calculate the sum of the array elements
        int sum2 = 0;
        for (int number : numbers) {
            sum2 += number;
        }
        System.out.println("Sum of array elements: " + sum2);
        // Sum of array elements: 150

        // Find the maximum element in the array
        int max = numbers[0];
        for (int number : numbers) {
            if (number > max) {
                max = number;
            }
        }
        System.out.println("Maximum element: " + max);
        // Maximum element: 50

        int intArr[] = { 10, 20, 15, 22, 35 };

        int intKey = 22;
        System.out.println(
                intKey + " found at index = "
                + Arrays.binarySearch(intArr, 1, 3, intKey));
        // 22 found at index = -4

        Arrays.sort(intArr);
        System.out.println("Sorted intArr : " + Arrays.toString(intArr));
        // Sorted intArr : [10, 15, 20, 22, 35]

    }
}

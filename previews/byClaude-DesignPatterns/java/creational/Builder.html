package creational;

import java.util.ArrayList;
import java.util.List;

// Run: javac -d java/out java/creational/Builder.java && java -cp java/out creational.Builder
public class Builder {
    public static void main(String[] args) {
        Burger classic = new BurgerBuilder()
            .setBun("sesame")
            .setPatty("beef")
            .addTopping("lettuce")
            .addTopping("tomato")
            .addTopping("onion")
            .setSauce("ketchup")
            .build();
        System.out.println("Classic Burger: " + classic);
        //Classic Burger: Burger { bun='sesame', patty='beef', toppings=[lettuce, tomato, onion], sauce='ketchup' }

        Burger veggie = new BurgerBuilder()
            .setBun("whole wheat")
            .setPatty("veggie")
            .addTopping("avocado")
            .addTopping("spinach")
            .setSauce("mustard")
            .build();
        System.out.println("Veggie Burger:  " + veggie);
        //Veggie Burger:  Burger { bun='whole wheat', patty='veggie', toppings=[avocado, spinach], sauce='mustard' }
    }
}

class Burger {
    private final String bun;
    private final String patty;
    private final List<String> toppings;
    private final String sauce;

    Burger(String bun, String patty, List<String> toppings, String sauce) {
        this.bun = bun; this.patty = patty; this.toppings = toppings; this.sauce = sauce;
    }

    public String toString() {
        return "Burger { bun='" + bun + "', patty='" + patty +
               "', toppings=" + toppings + ", sauce='" + sauce + "' }";
    }
}

class BurgerBuilder {
    private String bun   = "plain";
    private String patty = "beef";
    private final List<String> toppings = new ArrayList<>();
    private String sauce = "none";

    public BurgerBuilder setBun(String bun)         { this.bun = bun;           return this; }
    public BurgerBuilder setPatty(String patty)     { this.patty = patty;       return this; }
    public BurgerBuilder addTopping(String topping) { toppings.add(topping);    return this; }
    public BurgerBuilder setSauce(String sauce)     { this.sauce = sauce;       return this; }
    public Burger build() { return new Burger(bun, patty, new ArrayList<>(toppings), sauce); }
}

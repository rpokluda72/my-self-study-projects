package tutorial;

import io.reactivex.rxjava3.core.Flowable;
import io.reactivex.rxjava3.core.Observable;
import io.reactivex.rxjava3.schedulers.Schedulers;

/**
 * TUTORIAL 09 — Flowable and Backpressure
 *
 * <p>When a producer emits items faster than a consumer can process them, you have a
 * *backpressure* problem. {@link Observable} has no built-in backpressure handling —
 * if the consumer is too slow, items pile up in memory until an {@link io.reactivex.rxjava3.exceptions.MissingBackpressureException}
 * or {@link OutOfMemoryError} occurs.
 *
 * <p>{@link Flowable} is the backpressure-aware version of Observable. It implements the
 * <a href="https://www.reactive-streams.org/">Reactive Streams</a> protocol, where the
 * subscriber signals how many items it is ready to receive ({@code request(n)}).
 *
 * <p>When to use Flowable vs Observable:
 * <ul>
 *   <li>Use {@code Flowable} when the source can produce faster than the consumer can handle
 *       (large file reads, database cursors, sensor data, etc.).</li>
 *   <li>Use {@code Observable} for UI events, short sequences, or when you control the rate.</li>
 * </ul>
 *
 * <p>Backpressure strategies (used with {@code toFlowable(BackpressureStrategy)}):
 * <ul>
 *   <li>{@code BUFFER}  — buffer all items (can grow unbounded → OOM risk)</li>
 *   <li>{@code DROP}    — drop new items when the buffer is full</li>
 *   <li>{@code LATEST}  — keep only the latest item, overwriting the buffer</li>
 *   <li>{@code ERROR}   — throw {@code MissingBackpressureException} when buffer is full</li>
 *   <li>{@code MISSING} — no strategy applied; downstream must handle it via operators</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_09_Flowable.main()'
 */
public class _09_Flowable {

    public static void main(String[] args) throws InterruptedException {
        demoFlowableBasic();
        demoBackpressureBuffer();
        demoBackpressureDrop();
        demoBackpressureLatest();
        demoFlowableFromRange();
    }

    // -------------------------------------------------------------------------
    // Flowable basics
    // -------------------------------------------------------------------------

    /**
     * A simple {@link Flowable} works just like an Observable for small, synchronous sources.
     * The difference is internal: the subscriber signals demand via {@code request(n)}.
     *
     * <p>When you use the lambda {@code subscribe(onNext)} form, RxJava automatically
     * requests {@code Long.MAX_VALUE} items — effectively disabling backpressure control
     * and behaving like an Observable.
     */
demoFlowableBasic() 
    private static void demoFlowableBasic() {
        System.out.println("=== Flowable basics ===");

        Flowable.just("A", "B", "C")
                .subscribe(item -> System.out.println("Item: " + item));
        // === Flowable basics ===
        //Item: A
        //Item: B
        //Item: C

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // BackpressureStrategy.BUFFER
    // -------------------------------------------------------------------------

    /**
     * {@code BackpressureStrategy.BUFFER} tells the Flowable to buffer all items that
     * cannot be delivered immediately. The buffer grows without bound, so this is safe
     * only when you are confident the source won't produce too many items.
     *
     * <p>Here we create a fast Observable (emits 1000 items), convert it to a Flowable
     * with BUFFER strategy, and slow down the consumer with {@code observeOn} so the
     * buffering kicks in.
     */
demoBackpressureBuffer() 
    private static void demoBackpressureBuffer() throws InterruptedException {
        System.out.println("=== BackpressureStrategy.BUFFER ===");

        Observable.range(1, 10)
                .toFlowable(io.reactivex.rxjava3.core.BackpressureStrategy.BUFFER)
                .observeOn(Schedulers.computation(), false, 2) // downstream buffer of 2
                .subscribe(
                        item -> System.out.println("BUFFER received: " + item),
                        err  -> System.out.println("Error: " + err)
                );
        // === BackpressureStrategy.BUFFER ===
        //BUFFER received: 1
        //BUFFER received: 2
        //BUFFER received: 3
        //BUFFER received: 4
        //BUFFER received: 5
        //BUFFER received: 6
        //BUFFER received: 7
        //BUFFER received: 8
        //BUFFER received: 9
        //BUFFER received: 10

        Thread.sleep(500);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // BackpressureStrategy.DROP
    // -------------------------------------------------------------------------

    /**
     * {@code BackpressureStrategy.DROP} discards items that arrive when the downstream
     * is not ready. The consumer only sees items that fit in its current request window.
     *
     * <p>Use this when missing some data is acceptable (e.g., live stock price ticks —
     * you only care about the ones you can process; old ticks can be dropped).
     */
demoBackpressureDrop() 
    private static void demoBackpressureDrop() throws InterruptedException {
        System.out.println("=== BackpressureStrategy.DROP ===");

        // Fast source: 1..20; small downstream buffer: only some items will be delivered
        Observable.range(1, 20)
                .toFlowable(io.reactivex.rxjava3.core.BackpressureStrategy.DROP)
                .observeOn(Schedulers.computation(), false, 4) // downstream buffer of 4
                .subscribe(
                        item -> System.out.println("DROP received: " + item),
                        err  -> System.out.println("Error: " + err),
                        ()   -> System.out.println("DROP completed")
                );
        // Only the first ~4 items (fitting in the buffer) are delivered; rest are dropped.
        // === BackpressureStrategy.DROP ===
        //DROP received: 1
        //DROP received: 2
        //DROP received: 3
        //DROP received: 4
        //DROP completed
        // (remaining items 5-20 were dropped)

        Thread.sleep(500);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // BackpressureStrategy.LATEST
    // -------------------------------------------------------------------------

    /**
     * {@code BackpressureStrategy.LATEST} keeps only the most recently emitted item
     * in a 1-item buffer, overwriting the previous one if the consumer hasn't
     * consumed it yet.
     *
     * <p>Use this when you always want the freshest data (e.g., GPS coordinates,
     * temperature readings — the latest reading is the most relevant).
     */
demoBackpressureLatest() 
    private static void demoBackpressureLatest() throws InterruptedException {
        System.out.println("=== BackpressureStrategy.LATEST ===");

        Observable.range(1, 20)
                .toFlowable(io.reactivex.rxjava3.core.BackpressureStrategy.LATEST)
                .observeOn(Schedulers.computation(), false, 4)
                .subscribe(
                        item -> System.out.println("LATEST received: " + item),
                        err  -> System.out.println("Error: " + err),
                        ()   -> System.out.println("LATEST completed")
                );
        // You will typically see only a few early items and then the very last item (20),
        // because intermediate items are replaced by newer ones.
        // === BackpressureStrategy.LATEST ===
        //LATEST received: 1
        //LATEST received: 2
        //LATEST received: 3
        //LATEST received: 4
        //LATEST received: 20
        //LATEST completed

        Thread.sleep(500);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Flowable.range — natively backpressure-aware
    // -------------------------------------------------------------------------

    /**
     * {@code Flowable.range(start, count)} is the Flowable equivalent of
     * {@code Observable.range}. It natively supports backpressure — the source only
     * produces as many items as the downstream has requested.
     *
     * <p>This is the preferred way to create large numeric sequences in a Flowable pipeline
     * because it never buffers: items are generated on demand.
     */
demoFlowableFromRange() 
    private static void demoFlowableFromRange() throws InterruptedException {
        System.out.println("=== Flowable.range (native backpressure) ===");

        Flowable.range(1, 1_000_000)       // generates 1 million items on demand
                .filter(n -> n % 100_000 == 0) // keep every 100,000th
                .observeOn(Schedulers.computation())
                .subscribe(
                        n   -> System.out.println("Milestone: " + n),
                        err -> System.out.println("Error: " + err),
                        ()  -> System.out.println("Done processing 1,000,000 items")
                );

        Thread.sleep(2_000);
        // === Flowable.range (native backpressure) ===
        //Milestone: 100000
        //Milestone: 200000
        //Milestone: 300000
        //Milestone: 400000
        //Milestone: 500000
        //Milestone: 600000
        //Milestone: 700000
        //Milestone: 800000
        //Milestone: 900000
        //Milestone: 1000000
        //Done processing 1,000,000 items
        System.out.println();
    }
}

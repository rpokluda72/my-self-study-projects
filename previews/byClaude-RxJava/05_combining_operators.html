package tutorial;

import io.reactivex.rxjava3.core.Observable;

import java.util.concurrent.TimeUnit;

/**
 * TUTORIAL 05 — Combining Operators
 *
 * <p>Combining operators merge multiple Observables into one. The key difference between them
 * is *when* they emit and *how* they coordinate items from different sources.
 *
 * <p>Operators covered:
 * <ul>
 *   <li>{@code merge}         — subscribe to all sources simultaneously, emit as items arrive</li>
 *   <li>{@code concat}        — subscribe sources one-by-one, strictly in order</li>
 *   <li>{@code zip}           — pair items by position across multiple sources</li>
 *   <li>{@code combineLatest} — emit whenever any source emits, using the latest from each</li>
 *   <li>{@code startWithItem} — prepend a fixed item before the source stream</li>
 *   <li>{@code amb}           — emit from whichever source emits first, ignore the rest</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_05_CombiningOperators.main()'
 */
public class _05_CombiningOperators {

    public static void main(String[] args) throws InterruptedException {
        demoMerge();
        demoConcat();
        demoZip();
        demoCombineLatest();
        demoStartWith();
        demoAmb();
    }

    // -------------------------------------------------------------------------
    // merge
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.merge(source1, source2, ...)} subscribes to all sources at once
     * and emits items as they arrive — there is no guaranteed ordering between sources.
     *
     * <p>If any source errors, the merged stream errors immediately.
     * The merged stream completes when ALL sources have completed.
     */
demoMerge() 
    private static void demoMerge() {
        System.out.println("=== merge ===");

        Observable<String> source1 = Observable.just("A", "B", "C");
        Observable<String> source2 = Observable.just("1", "2", "3");

        // Both sources are synchronous here, so source1 completes before source2 starts
        Observable.merge(source1, source2)
                .subscribe(item -> System.out.println("Merged: " + item));
        // === merge ===
        //Merged: A
        //Merged: B
        //Merged: C
        //Merged: 1
        //Merged: 2
        //Merged: 3

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // concat
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.concat(source1, source2, ...)} subscribes to sources *sequentially*:
     * it only subscribes to source2 after source1 has completed, and so on.
     *
     * <p>The output order is always: all items from source1, then all from source2, etc.
     * Use this when strict ordering is required.
     */
demoConcat() 
    private static void demoConcat() {
        System.out.println("=== concat ===");

        Observable<String> first  = Observable.just("First-1",  "First-2");
        Observable<String> second = Observable.just("Second-1", "Second-2");

        Observable.concat(first, second)
                .subscribe(item -> System.out.println("Concat: " + item));
        // === concat ===
        //Concat: First-1
        //Concat: First-2
        //Concat: Second-1
        //Concat: Second-2

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // zip
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.zip(source1, source2, zipper)} pairs up items by *position*:
     * the zipper function receives the 1st item of source1 with the 1st item of source2,
     * the 2nd with the 2nd, etc. It emits a result for each pair.
     *
     * <p>The output length equals the length of the *shortest* source.
     * Use zip to combine logically related items from parallel streams.
     */
demoZip() 
    private static void demoZip() {
        System.out.println("=== zip ===");

        Observable<String> names  = Observable.just("Alice", "Bob", "Carol");
        Observable<Integer> scores = Observable.just(95, 87, 92);

        Observable.zip(names, scores,
                (name, score) -> name + " → " + score)
                .subscribe(result -> System.out.println("Zipped: " + result));
        // === zip ===
        //Zipped: Alice → 95
        //Zipped: Bob → 87
        //Zipped: Carol → 92

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // combineLatest
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.combineLatest(source1, source2, combiner)} emits a combined item
     * *whenever any source emits*, using the most recently emitted value from each source.
     *
     * <p>It does NOT wait for all sources to emit at the same position — it just reacts
     * to any new emission using the "latest" snapshot of all sources.
     *
     * <p>Common use case: combining multiple UI input fields — update the result whenever
     * any field changes.
     */
demoCombineLatest() 
    private static void demoCombineLatest() throws InterruptedException {
        System.out.println("=== combineLatest ===");

        // Ticks every 100ms, takes 3 values; other ticks every 250ms, takes 2 values
        Observable<Long> fast = Observable.interval(100, TimeUnit.MILLISECONDS).take(3);
        Observable<Long> slow = Observable.interval(250, TimeUnit.MILLISECONDS).take(2);

        Observable.combineLatest(fast, slow,
                (f, s) -> "fast=" + f + " slow=" + s)
                .subscribe(combo -> System.out.println("CombineLatest: " + combo));
        // Output depends on timing — something like:
        // === combineLatest ===
        //CombineLatest: fast=1 slow=0
        //CombineLatest: fast=2 slow=0
        //CombineLatest: fast=2 slow=1

        Thread.sleep(800); // wait for async sources
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // startWithItem
    // -------------------------------------------------------------------------

    /**
     * {@code startWithItem(item)} prepends a single item to the beginning of the stream
     * before the source starts emitting. Useful for providing an initial/default value.
     */
demoStartWith() 
    private static void demoStartWith() {
        System.out.println("=== startWithItem ===");

        Observable.just("B", "C", "D")
                .startWithItem("A")          // prepend "A"
                .subscribe(s -> System.out.println("Item: " + s));
        // === startWithItem ===
        //Item: A
        //Item: B
        //Item: C
        //Item: D

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // amb (ambiguous)
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.amb(sources)} subscribes to all provided sources simultaneously
     * and emits from whichever one emits its *first item first*, then disposes all others.
     *
     * <p>"amb" stands for "ambiguous" — you don't know in advance which will be first.
     * Useful for racing multiple data sources (e.g., cache vs. network) and using
     * whichever responds first.
     */
demoAmb() 
    private static void demoAmb() throws InterruptedException {
        System.out.println("=== amb ===");

        // source1 emits after 100ms; source2 emits after 200ms — source1 wins
        Observable<String> source1 = Observable.just("Fast source")
                .delay(100, TimeUnit.MILLISECONDS);
        Observable<String> source2 = Observable.just("Slow source")
                .delay(200, TimeUnit.MILLISECONDS);

        Observable.amb(java.util.Arrays.asList(source1, source2))
                .subscribe(winner -> System.out.println("Winner: " + winner));
        // === amb ===
        //Winner: Fast source

        Thread.sleep(500);
        System.out.println();
    }
}

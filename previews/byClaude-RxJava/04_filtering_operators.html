package tutorial;

import io.reactivex.rxjava3.core.Observable;

/**
 * TUTORIAL 04 — Filtering Operators
 *
 * <p>Filtering operators selectively pass items downstream based on a predicate or position,
 * or they limit the number of items that flow through. No items are transformed — only
 * selected or dropped.
 *
 * <p>Operators covered:
 * <ul>
 *   <li>{@code filter}               — keep items matching a predicate</li>
 *   <li>{@code take} / {@code skip}  — take or skip the first N items</li>
 *   <li>{@code takeLast}             — keep only the last N items</li>
 *   <li>{@code distinct}             — suppress duplicates (globally)</li>
 *   <li>{@code distinctUntilChanged} — suppress consecutive duplicates only</li>
 *   <li>{@code first} / {@code last} — emit only the first or last item</li>
 *   <li>{@code elementAt}            — emit the item at a specific index</li>
 *   <li>{@code ignoreElements}       — discard all items, pass only terminal events</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_04_FilteringOperators.main()'
 */
public class _04_FilteringOperators {

    public static void main(String[] args) {
        demoFilter();
        demoTake();
        demoSkip();
        demoTakeLast();
        demoDistinct();
        demoDistinctUntilChanged();
        demoFirst();
        demoLast();
        demoElementAt();
        demoIgnoreElements();
    }

    // -------------------------------------------------------------------------
    // filter
    // -------------------------------------------------------------------------

    /**
     * {@code filter(Predicate<T>)} emits only those items for which the predicate returns
     * {@code true}. Items that do not match are silently dropped.
     */
demoFilter() 
    private static void demoFilter() {
        System.out.println("=== filter ===");

        Observable.range(1, 10)
                .filter(n -> n % 2 == 0) // keep only even numbers
                .subscribe(n -> System.out.println("Even: " + n));
        // === filter ===
        // Even: 2
        // Even: 4
        // Even: 6
        // Even: 8
        // Even: 10
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // take
    // -------------------------------------------------------------------------

    /**
     * {@code take(n)} emits at most the first {@code n} items, then completes.
     * Items after position {@code n} are never emitted (and the upstream may be disposed).
     */
demoTake() 
    private static void demoTake() {
        System.out.println("=== take(3) ===");

        Observable.range(1, 100)
                .take(3)
                .subscribe(n -> System.out.println("Taken: " + n));
        // === take(3) ===
        // Taken: 1
        // Taken: 2
        // Taken: 3
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // skip
    // -------------------------------------------------------------------------

    /**
     * {@code skip(n)} drops the first {@code n} items and emits everything after them.
     * It is the complement of {@code take}.
     */
demoSkip() 
    private static void demoSkip() {
        System.out.println("=== skip(3) from range(1,6) ===");

        Observable.range(1, 6)
                .skip(3) // drop 1, 2, 3
                .subscribe(n -> System.out.println("After skip: " + n));
        // === skip(3) from range(1,6) ===
        //After skip: 4
        //After skip: 5
        //After skip: 6

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // takeLast
    // -------------------------------------------------------------------------

    /**
     * {@code takeLast(n)} waits for the source to complete, then emits only the
     * last {@code n} items. It must buffer the last N items internally.
     */
demoTakeLast() 
    private static void demoTakeLast() {
        System.out.println("=== takeLast(3) ===");

        Observable.just("a", "b", "c", "d", "e")
                .takeLast(3)
                .subscribe(s -> System.out.println("Last: " + s));
        // === takeLast(3) ===
        //Last: c
        //Last: d
        //Last: e

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // distinct
    // -------------------------------------------------------------------------

    /**
     * {@code distinct()} filters out items that have already been seen anywhere
     * in the stream. It keeps an internal {@code Set} of all emitted items.
     *
     * <p>Use only on finite streams — an infinite stream would grow the set forever.
     */
demoDistinct() 
    private static void demoDistinct() {
        System.out.println("=== distinct ===");

        Observable.just(1, 2, 2, 3, 1, 4, 3)
                .distinct()
                .subscribe(n -> System.out.println("Distinct: " + n));
        // === distinct ===
        //Distinct: 1
        //Distinct: 2
        //Distinct: 3
        //Distinct: 4

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // distinctUntilChanged
    // -------------------------------------------------------------------------

    /**
     * {@code distinctUntilChanged()} suppresses an item only if it is equal to the
     * *immediately preceding* item. Non-consecutive duplicates are allowed through.
     *
     * <p>This is cheaper than {@code distinct()} because it only needs to remember
     * the last emitted value, not the entire history.
     */
demoDistinctUntilChanged() 
    private static void demoDistinctUntilChanged() {
        System.out.println("=== distinctUntilChanged ===");

        Observable.just(1, 1, 2, 3, 3, 3, 2)
                .distinctUntilChanged()
                .subscribe(n -> System.out.println("DUC: " + n));
        // === distinctUntilChanged ===
        //DUC: 1
        //DUC: 2
        //DUC: 3
        //DUC: 2

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // first
    // -------------------------------------------------------------------------

    /**
     * {@code first(defaultValue)} emits the very first item from the stream as a
     * {@code Single<T>}, or the {@code defaultValue} if the source is empty.
     *
     * <p>Use {@code firstOrError()} if you want an error when the source is empty.
     */
demoFirst() 
    private static void demoFirst() {
        System.out.println("=== first ===");

        Observable.just(10, 20, 30)
                .first(-1)  // -1 is the default if source is empty
                .subscribe(
                        val -> System.out.println("First: " + val), // → First: 10
                        err -> System.out.println("Error: " + err)
                );

        // Empty source uses the default
        Observable.<Integer>empty()
                .first(-1)
                .subscribe(val -> System.out.println("First (empty): " + val)); // → First (empty): -1
        // === first ===
        //First: 10
        //First (empty): -1
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // last
    // -------------------------------------------------------------------------

    /**
     * {@code last(defaultValue)} waits for the source to complete, then emits
     * only the last item as a {@code Single<T>}, or {@code defaultValue} if empty.
     */
demoLast() 
    private static void demoLast() {
        System.out.println("=== last ===");

        Observable.just("x", "y", "z")
                .last("none")
                .subscribe(
                        val -> System.out.println("Last: " + val), // → Last: z
                        err -> System.out.println("Error: " + err)
                );

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // elementAt
    // -------------------------------------------------------------------------

    /**
     * {@code elementAt(index)} emits the item at the specified zero-based index
     * as a {@code Maybe<T>}. If the source has fewer items, the Maybe completes empty.
     */
demoElementAt() 
    private static void demoElementAt() {
        System.out.println("=== elementAt ===");

        Observable.just("zero", "one", "two", "three")
                .elementAt(2) // zero-based index → "two"
                .subscribe(
                        val -> System.out.println("Element at 2: " + val), // → Element at 2: two
                        err -> System.out.println("Error: " + err)
                );

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // ignoreElements
    // -------------------------------------------------------------------------

    /**
     * {@code ignoreElements()} discards all {@code onNext} items and converts the
     * Observable to a {@code Completable} that only signals {@code onComplete} or
     * {@code onError}. Useful when you only care whether a stream finished, not its values.
     */
demoIgnoreElements() 
    private static void demoIgnoreElements() {
        System.out.println("=== ignoreElements ===");

        Observable.just(1, 2, 3)
                .ignoreElements()           // returns Completable
                .subscribe(
                        () -> System.out.println("Stream completed (items ignored)"), // → Stream completed (items ignored)
                        err -> System.out.println("Error: " + err)
                );

        System.out.println();
    }
}

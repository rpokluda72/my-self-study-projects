package tutorial;

import io.reactivex.rxjava3.core.Observable;
import io.reactivex.rxjava3.schedulers.Schedulers;

/**
 * TUTORIAL 08 — Schedulers (Threading)
 *
 * <p>By default, RxJava is single-threaded: everything runs on the thread that calls
 * {@code subscribe()}. Schedulers let you move work to other threads.
 *
 * <p>Two key operators control threading:
 * <ul>
 *   <li>{@code subscribeOn(Scheduler)} — controls which thread the *source* (data production)
 *       runs on. Only the first {@code subscribeOn} in a chain has effect.</li>
 *   <li>{@code observeOn(Scheduler)} — controls which thread *downstream operators and
 *       the subscriber* run on. Can appear multiple times to switch threads mid-chain.</li>
 * </ul>
 *
 * <p>Built-in Schedulers:
 * <ul>
 *   <li>{@link Schedulers#io()}            — elastic thread pool for I/O (DB, network, file)</li>
 *   <li>{@link Schedulers#computation()}   — fixed thread pool (CPU count) for CPU work</li>
 *   <li>{@link Schedulers#newThread()}     — creates a new thread for each subscription</li>
 *   <li>{@link Schedulers#single()}        — single background thread, FIFO queue</li>
 *   <li>{@link Schedulers#trampoline()}    — queues work on the *current* thread (no new thread)</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_08_Schedulers.main()'
 */
public class _08_Schedulers {

    public static void main(String[] args) throws InterruptedException {
        demoDefaultThread();
        demoSubscribeOn();
        demoObserveOn();
        demoSubscribeOnAndObserveOn();
        demoIoScheduler();
        demoComputationScheduler();
        demoNewThreadScheduler();
        demoSingleScheduler();
    }

    /** Prints the current thread name with a label. */
    private static void printThread(String label) {
        System.out.println(label + " → thread: [" + Thread.currentThread().getName() + "]");
    }

    // -------------------------------------------------------------------------
    // Default threading
    // -------------------------------------------------------------------------

    /**
     * Without any Scheduler, everything runs on the calling thread ({@code main}).
     */
demoDefaultThread() 
    private static void demoDefaultThread() {
        System.out.println("=== Default thread (no Scheduler) ===");

        Observable.just("Hello")
                .map(s -> { printThread("map"); return s.toUpperCase(); }) // → map → thread: [main]
                .subscribe(s -> printThread("subscriber"));                // → subscriber → thread: [main]
        // === Default thread (no Scheduler) ===
        //map → thread: [main]
        //subscriber → thread: [main]
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // subscribeOn
    // -------------------------------------------------------------------------

    /**
     * {@code subscribeOn(Scheduler)} moves the *subscription* (source data production)
     * to the given Scheduler. All operators above it run on that Scheduler's thread.
     *
     * <p>Rule: only the *first* {@code subscribeOn} in a chain takes effect — subsequent
     * ones are ignored because the thread is already selected by the first one.
     */
demoSubscribeOn() 
    private static void demoSubscribeOn() throws InterruptedException {
        System.out.println("=== subscribeOn ===");

        Observable.just("data")
                .doOnSubscribe(d -> printThread("doOnSubscribe"))
                .map(s -> { printThread("map"); return s; })
                .subscribeOn(Schedulers.io())    // move source to an IO thread
                .subscribe(s -> printThread("subscriber"));

        Thread.sleep(300); // wait for async work

        // === subscribeOn ===
        //doOnSubscribe → thread: [RxCachedThreadScheduler-1]
        //map → thread: [RxCachedThreadScheduler-1]
        //subscriber → thread: [RxCachedThreadScheduler-1]
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // observeOn
    // -------------------------------------------------------------------------

    /**
     * {@code observeOn(Scheduler)} switches the thread for *all downstream* operators
     * (from that point in the chain). It can appear multiple times to hop between threads.
     */
demoObserveOn() 
    private static void demoObserveOn() throws InterruptedException {
        System.out.println("=== observeOn ===");

        Observable.just("data")
                .map(s -> { printThread("map (before observeOn)"); return s; })
                .observeOn(Schedulers.newThread())
                .map(s -> { printThread("map (after observeOn)"); return s; })
                .subscribe(s -> printThread("subscriber"));

        Thread.sleep(300);
        // === observeOn ===
        //map (before observeOn) → thread: [main]
        //map (after observeOn) → thread: [RxNewThreadScheduler-1]
        //subscriber → thread: [RxNewThreadScheduler-1]
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // subscribeOn + observeOn together
    // -------------------------------------------------------------------------

    /**
     * The classic pattern:
     * <ul>
     *   <li>{@code subscribeOn(Schedulers.io())} — produce data on a background IO thread</li>
     *   <li>{@code observeOn(Schedulers.single())} — deliver results on a specific thread
     *       (in Android this would be {@code AndroidSchedulers.mainThread()})</li>
     * </ul>
     */
demoSubscribeOnAndObserveOn() 
    private static void demoSubscribeOnAndObserveOn() throws InterruptedException {
        System.out.println("=== subscribeOn + observeOn ===");

        Observable.fromCallable(() -> {
                    printThread("source (producing data)");
                    // → source (producing data) → thread: [RxCachedThreadScheduler-2]
                    Thread.sleep(100); // simulate I/O
                    return "DB result";
                })
                .subscribeOn(Schedulers.io())         // production on IO thread
                .observeOn(Schedulers.single())       // delivery on single-thread
                .subscribe(result -> {
                    printThread("subscriber (consuming result)");
                    System.out.println("Result: " + result); // → Result: DB result
                });

        Thread.sleep(500);
        // === subscribeOn + observeOn ===
        //source (producing data) → thread: [RxCachedThreadScheduler-1]
        //subscriber (consuming result) → thread: [RxSingleScheduler-1]
        //Result: DB result
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Schedulers.io()
    // -------------------------------------------------------------------------

    /**
     * {@link Schedulers#io()} maintains an elastic thread pool that grows as needed
     * and reuses idle threads. Perfect for I/O-bound work (network calls, file access,
     * database queries) where threads spend most of their time waiting.
     */
demoIoScheduler() 
    private static void demoIoScheduler() throws InterruptedException {
        System.out.println("=== Schedulers.io() — multiple parallel I/O tasks ===");

        // Three "I/O" tasks run in parallel on the IO pool
        for (int i = 1; i <= 3; i++) {
            final int taskId = i;
            Observable.fromCallable(() -> {
                        Thread.sleep(50);
                        return "IO Task " + taskId + " done";
                    })
                    .subscribeOn(Schedulers.io())
                    .subscribe(msg -> System.out.println(msg + " on [" + Thread.currentThread().getName() + "]"));
        }
        // Output (order may vary, all on different RxCachedThreadScheduler threads):
        // === Schedulers.io() — multiple parallel I/O tasks ===
        //IO Task 1 done on [RxCachedThreadScheduler-1]
        //IO Task 2 done on [RxCachedThreadScheduler-2]
        //IO Task 3 done on [RxCachedThreadScheduler-3]

        Thread.sleep(400);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Schedulers.computation()
    // -------------------------------------------------------------------------

    /**
     * {@link Schedulers#computation()} uses a fixed-size thread pool equal to the number
     * of available CPU cores. Use it for CPU-intensive work (parsing, calculations, sorting)
     * where you want to saturate cores without overloading the system.
     *
     * <p>Do NOT use it for blocking I/O — a blocked computation thread wastes a CPU slot.
     */
demoComputationScheduler() 
    private static void demoComputationScheduler() throws InterruptedException {
        System.out.println("=== Schedulers.computation() ===");

        Observable.range(1, 4)
                .flatMap(n -> Observable.fromCallable(() -> {
                            // Simulate CPU work
                            long sum = 0;
                            for (int i = 0; i < 1_000_000; i++) sum += i;
                            return "Task " + n + " result=" + sum;
                        })
                        .subscribeOn(Schedulers.computation()))
                .subscribe(result ->
                        System.out.println(result + " on [" + Thread.currentThread().getName() + "]"));
        // ... (order may vary)
        // === Schedulers.computation() ===
        //Task 1 result=499999500000 on [RxComputationThreadPool-1]
        //Task 2 result=499999500000 on [RxComputationThreadPool-1]
        //Task 3 result=499999500000 on [RxComputationThreadPool-1]
        //Task 4 result=499999500000 on [RxComputationThreadPool-1]

        Thread.sleep(1_000);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Schedulers.newThread()
    // -------------------------------------------------------------------------

    /**
     * {@link Schedulers#newThread()} creates a brand-new thread for every subscription.
     * Unlike {@code io()}, threads are NOT pooled or reused — each subscribe spawns one
     * fresh thread. Use sparingly; prefer {@code io()} for pooling efficiency.
     */
demoNewThreadScheduler() 
    private static void demoNewThreadScheduler() throws InterruptedException {
        System.out.println("=== Schedulers.newThread() ===");

        Observable.just("Task")
                .subscribeOn(Schedulers.newThread())
                .subscribe(t ->
                        System.out.println(t + " on [" + Thread.currentThread().getName() + "]"));
        // === Schedulers.newThread() ===
        //Task on [RxNewThreadScheduler-2]

        Thread.sleep(200);
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Schedulers.single()
    // -------------------------------------------------------------------------

    /**
     * {@link Schedulers#single()} uses a single background thread shared across
     * all subscribers. Tasks are queued and executed one at a time in FIFO order.
     *
     * <p>Useful when you need to serialize operations (e.g., write to a resource
     * that is not thread-safe) without using synchronization primitives.
     */
demoSingleScheduler() 
    private static void demoSingleScheduler() throws InterruptedException {
        System.out.println("=== Schedulers.single() — tasks run one at a time ===");

        for (int i = 1; i <= 3; i++) {
            final int n = i;
            Observable.fromCallable(() -> {
                        Thread.sleep(50);
                        return "Single task " + n;
                    })
                    .subscribeOn(Schedulers.single())
                    .subscribe(msg ->
                            System.out.println(msg + " on [" + Thread.currentThread().getName() + "]"));
        }
        // All tasks run on the SAME thread, sequentially:
        // === Schedulers.single() — tasks run one at a time ===
        //Single task 1 on [RxSingleScheduler-1]
        //Single task 2 on [RxSingleScheduler-1]
        //Single task 3 on [RxSingleScheduler-1]

        Thread.sleep(500);
        System.out.println();
    }
}

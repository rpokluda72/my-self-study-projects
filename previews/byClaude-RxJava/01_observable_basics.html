package tutorial;

import io.reactivex.rxjava3.core.Observable;
import io.reactivex.rxjava3.core.Observer;
import io.reactivex.rxjava3.disposables.Disposable;

/**
 * TUTORIAL 01 — Observable Basics
 *
 * <p>An {@link Observable} is the core type in RxJava. It represents a stream of data or events
 * that can be observed over time. An {@link Observer} subscribes to an Observable and reacts to
 * each emitted item, an error, or the completion signal.
 *
 * <p>The three signals an Observable can emit:
 * <ul>
 *   <li>{@code onNext(T item)}  — a new item in the stream</li>
 *   <li>{@code onError(Throwable e)} — a terminal error; no more items will follow</li>
 *   <li>{@code onComplete()} — the stream ended successfully; no more items will follow</li>
 * </ul>
 *
 * <p>Run this class with IntelliJ: right-click → Run '_01_ObservableBasics.main()'
 */
public class _01_ObservableBasics {

    public static void main(String[] args) {
        demoJust();
        demoFromArray();
        demoFromIterable();
        demoRange();
        demoEmpty();
        demoNever();
        demoError();
        demoFullObserver();
    }

    // -------------------------------------------------------------------------
    // Observable.just
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.just(...)} creates an Observable that emits each supplied value in order,
     * then completes. It accepts 1–10 arguments (varargs overloads).
     *
     * <p>Use it when you already have a fixed set of known values.
     */
demoJust() 
    private static void demoJust() {
        System.out.println("=== Observable.just ===");

        Observable<String> observable = Observable.just("Alpha", "Beta", "Gamma");

        // subscribe(onNext) — the simplest form; errors are thrown if they occur
        observable.subscribe(item -> System.out.println("Received: " + item));
        // → Received: Alpha
        // → Received: Beta
        // → Received: Gamma

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.fromArray
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.fromArray(T[])} emits each element of an existing array.
     * Equivalent to {@code Observable.just(...)} but works with a pre-existing array variable.
     */
demoFromArray() 
    private static void demoFromArray() {
        System.out.println("=== Observable.fromArray ===");

        Integer[] numbers = {10, 20, 30, 40, 50};
        Integer[] numbers2 = new Integer[5];
//        numbers2[0] = 10;
        for (int i = 0; i < numbers2.length; i++) {
            numbers2[i] = (i + 1) * 10;
        }

        Observable.fromArray(numbers)
                .subscribe(n -> System.out.println("Number: " + n));
        // → Number: 10
        // → Number: 20
        // → Number: 30
        // → Number: 40
        // → Number: 50

        Observable.fromArray(numbers2)
                .subscribe(n -> System.out.println("Number 2 : " + n));
        // → Number 2: 10
        // → Number 2: 20
        // → Number 2: 30
        // → Number 2: 40
        // → Number 2: 50

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.fromIterable
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.fromIterable(Iterable<T>)} emits each element of any {@link Iterable}
     * (List, Set, etc.). The elements are emitted in the iteration order of the collection.
     */
demoFromIterable() 
    private static void demoFromIterable() {
        System.out.println("=== Observable.fromIterable ===");

        java.util.List<String> colors = java.util.Arrays.asList("Red", "Green", "Blue");

        Observable.fromIterable(colors)
                .subscribe(color -> System.out.println("Color: " + color));
        // → Color: Red
        // → Color: Green
        // → Color: Blue

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.range
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.range(start, count)} emits {@code count} consecutive integers starting
     * from {@code start}. Useful for generating sequences without creating a collection.
     */
demoRange() 
    private static void demoRange() {
        System.out.println("=== Observable.range ===");

        // Emits 1, 2, 3, 4, 5
        Observable.range(1, 5)
                .subscribe(i -> System.out.println("Value: " + i));
        // → Value: 1
        // → Value: 2
        // → Value: 3
        // → Value: 4
        // → Value: 5

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.empty
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.empty()} emits nothing and immediately calls {@code onComplete()}.
     * Useful as a no-op placeholder or base case in composition.
     */
demoEmpty() 
    private static void demoEmpty() {
        System.out.println("=== Observable.empty ===");

        Observable.empty()
                .subscribe(
                        item    -> System.out.println("Item: " + item),   // never called
                        error   -> System.out.println("Error: " + error), // never called
                        ()      -> System.out.println("Completed!")        // called immediately
                );
        // → Completed!

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.never
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.never()} emits nothing and never completes or errors.
     * It represents an infinite, empty stream. Useful in testing or as a sentinel.
     *
     * <p>Because it never terminates, {@code onComplete} and {@code onError} are never called.
     * Here we just confirm subscription happened.
     */
demoNever() 
    private static void demoNever() {
        System.out.println("=== Observable.never ===");

        Disposable d = Observable.never()
                .subscribe(
                        item -> System.out.println("Item: " + item),
                        err  -> System.out.println("Error: " + err),
                        ()   -> System.out.println("Completed!")  // never reached
                );
        // Disposed / Zlikvidováno
        System.out.println("Subscribed, nothing emitted: isDisposed=" + d.isDisposed()); // → Subscribed, nothing emitted: isDisposed=false
        d.dispose(); // clean up

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.error
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.error(Throwable)} immediately signals an error to the subscriber
     * without emitting any items. Useful for propagating known errors in a pipeline.
     */
demoError() 
    private static void demoError() {
        System.out.println("=== Observable.error ===");

        Observable.error(new RuntimeException("Something went wrong"))
                .subscribe(
                        item  -> System.out.println("Item: " + item),        // never called
                        error -> System.out.println("Caught error: " + error.getMessage()), // called
                        ()    -> System.out.println("Completed!")             // never called
                );
        // → Caught error: Something went wrong

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Full Observer interface
    // -------------------------------------------------------------------------

    /**
     * Instead of lambda shortcuts, you can implement the full {@link Observer} interface.
     * This also gives you access to the {@link Disposable}, allowing you to cancel
     * the subscription from within {@code onSubscribe}.
     */
demoFullObserver() 
    private static void demoFullObserver() {
        System.out.println("=== Full Observer interface ===");

        Observable<Integer> source = Observable.range(1, 3);

        source.subscribe(new Observer<Integer>() {

            private Disposable upstream;

            @Override
            public void onSubscribe(Disposable d) {
                this.upstream = d;
                System.out.println("onSubscribe called"); // → onSubscribe called
            }

            @Override
            public void onNext(Integer value) {
                System.out.println("onNext: " + value);
                // → onNext: 1
                // → onNext: 2
                // → onNext: 3
            }

            @Override
            public void onError(Throwable e) {
                System.out.println("onError: " + e.getMessage()); // not called here
            }

            @Override
            public void onComplete() {
                System.out.println("onComplete"); // → onComplete
            }
        });

        System.out.println();
    }
}

package tutorial;

import io.reactivex.rxjava3.core.Observable;

import java.util.List;

/**
 * TUTORIAL 03 — Transforming Operators
 *
 * <p>Transforming operators take an upstream Observable and return a new Observable
 * whose items have been reshaped in some way. They are the "map", "flatMap", and
 * "reduce" equivalents of the reactive world.
 *
 * <p>Operators covered:
 * <ul>
 *   <li>{@code map}        — transform each item one-to-one</li>
 *   <li>{@code flatMap}    — transform each item into an Observable, merge results</li>
 *   <li>{@code concatMap}  — like flatMap but preserves (zachová) order</li>
 *   <li>{@code switchMap}  — like flatMap but cancels the previous inner Observable</li>
 *   <li>{@code scan}       — running accumulation (like reduce but emits intermediates)</li>
 *   <li>{@code buffer}     — collect items into lists of a fixed size</li>
 *   <li>{@code toList}     — collect all items into a single list, emit once</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_03_TransformingOperators.main()'
 */
public class _03_TransformingOperators {

    public static void main(String[] args) {
        demoMap();
        demoFlatMap();
        demoConcatMap();
        demoSwitchMap();
        demoScan();
        demoBuffer();
        demoToList();
    }

    // -------------------------------------------------------------------------
    // map
    // -------------------------------------------------------------------------

    /**
     * {@code map(Function<T, R>)} applies a function to every emitted item and emits
     * the transformed result. The mapping is synchronous and one-to-one (T → R).
     *
     * <p>Example: convert a stream of strings to their lengths.
     */
demoMap() 
    private static void demoMap() {
        System.out.println("=== map ===");

        Observable.just("Hello", "RxJava", "World")
                .map(s -> s.toUpperCase() + " (" + s.length() + ")")
                .subscribe(item -> System.out.println(item));
        // → HELLO (5)
        // → RXJAVA (6)
        // → WORLD (5)

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // flatMap
    // -------------------------------------------------------------------------

    /**
     * {@code flatMap(Function<T, Observable<R>>)} maps each item to an inner Observable,
     * then *merges* all inner Observables into a single output stream.
     *
     * <p>Because it merges (not concatenates), the order of items from different inner
     * Observables is NOT guaranteed if they are asynchronous. For synchronous sources
     * the output order matches input order.
     *
     * <p>Think of it as: "for each item, produce zero or more new items".
     */
demoFlatMap() 
    private static void demoFlatMap() {
        System.out.println("=== flatMap ===");

        // For each number, emit the number and its square
        Observable.just(1, 2, 3)
                .flatMap(n -> Observable.just(n, n * n))
                .subscribe(v -> System.out.println("flatMap item: " + v));
        // → flatMap item: 1
        // → flatMap item: 1
        // → flatMap item: 2
        // → flatMap item: 4
        // → flatMap item: 3
        // → flatMap item: 9

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // concatMap
    // -------------------------------------------------------------------------

    /**
     * {@code concatMap(Function<T, Observable<R>>)} is like {@code flatMap} but it
     * subscribes to each inner Observable *sequentially* — it waits for the current
     * inner Observable to complete before subscribing to the next one.
     *
     * <p>Use {@code concatMap} when order matters and inner sources may be async.
     */
demoConcatMap() 
    private static void demoConcatMap() {
        System.out.println("=== concatMap ===");

        // Emit a word, then each of its characters — in strict order
        Observable.just("Hi", "Rx")
                .concatMap(word -> Observable.fromArray(word.split("")))
                .subscribe(ch -> System.out.println("char: " + ch));
        // → char: H
        // → char: i
        // → char: R
        // → char: x

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // switchMap
    // -------------------------------------------------------------------------

    /**
     * {@code switchMap(Function<T, Observable<R>>)} maps each item to an inner Observable
     * but *disposes the previous inner Observable* as soon as a new item arrives.
     *
     * <p>This is ideal for "cancel the current search, start a new one" patterns
     * (e.g., autocomplete). With synchronous sources the switch happens so fast that
     * only the last inner Observable's items appear in practice.
     */
demoSwitchMap() 
    private static void demoSwitchMap() {
        System.out.println("=== switchMap ===");

        // Each outer item cancels the previous inner observable.
        // With synchronous sources, only the LAST inner observable's items are seen.
        Observable.just("a", "b", "c")
                .switchMap(letter -> Observable.just(letter + "1", letter + "2"))
                .subscribe(v -> System.out.println("switchMap: " + v));
        // → switchMap: c1
        // → switchMap: c2

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // scan
    // -------------------------------------------------------------------------

    /**
     * {@code scan(BiFunction<T, T, T>)} applies an accumulator function to each item
     * and emits *every intermediate accumulated value* (not just the final one).
     *
     * <p>Contrast with {@code reduce}, which only emits the final value.
     * {@code scan} is useful for running totals, running max/min, state machines, etc.
     */
demoScan() 
    private static void demoScan() {
        System.out.println("=== scan (running sum) ===");

        Observable.just(1, 2, 3, 4, 5)
                .scan((accumulator, nextItem) -> accumulator + nextItem)
                .subscribe(runningSum -> System.out.println("Running sum: " + runningSum));
        // → Running sum: 1
        // → Running sum: 3
        // → Running sum: 6
        // → Running sum: 10
        // → Running sum: 15

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // buffer
    // -------------------------------------------------------------------------

    /**
     * {@code buffer(count)} collects emitted items into {@link List}s of a fixed size
     * and emits each list as a single item downstream.
     *
     * <p>The last list may be smaller than {@code count} if the source completes with
     * fewer remaining items.
     */
demoBuffer() 
    private static void demoBuffer() {
        System.out.println("=== buffer(3) ===");

        Observable.range(1, 9)
                .buffer(3) // collect into groups of 3
                .subscribe(batch -> System.out.println("Batch: " + batch));
        // → Batch: [1, 2, 3]
        // → Batch: [4, 5, 6]
        // → Batch: [7, 8, 9]

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // toList
    // -------------------------------------------------------------------------

    /**
     * {@code toList()} waits for the source Observable to complete, collects all items
     * into a {@link List}, and emits that list as a single item via a {@code Single<List<T>>}.
     *
     * <p>Only use this on finite Observables — an infinite source will never complete
     * so {@code toList()} will never emit.
     */
demoToList() 
    private static void demoToList() {
        System.out.println("=== toList ===");

        Observable.just("apple", "banana", "cherry")
                .map(String::toUpperCase)
                .toList()  // returns Single<List<String>>
                .subscribe(
                        list  -> System.out.println("All items: " + list), // → All items: [APPLE, BANANA, CHERRY]
                        error -> System.out.println("Error: " + error)
                );

        System.out.println();
    }
}

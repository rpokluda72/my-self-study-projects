package tutorial;

import io.reactivex.rxjava3.core.Observable;
import io.reactivex.rxjava3.core.ObservableEmitter;

import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * TUTORIAL 02 — Observable Creation (Advanced Factories)
 *
 * <p>Beyond the static factories in Tutorial 01, RxJava provides factories for building
 * Observables from custom logic, lazy evaluation, or time-based sequences.
 *
 * <p>Topics covered:
 * <ul>
 *   <li>{@code Observable.create}    — full control via an emitter</li>
 *   <li>{@code Observable.defer}     — lazy factory, evaluated fresh per subscriber</li>
 *   <li>{@code Observable.fromCallable} — wraps a single value-producing call</li>
 *   <li>{@code Observable.interval}  — ticks at a fixed time interval</li>
 *   <li>{@code Observable.timer}     — emits a single {@code 0L} after a delay</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_02_ObservableCreation.main()'
 */
public class _02_ObservableCreation {

    public static void main(String[] args) throws InterruptedException {
        demoCreate();
        demoDefer();
        demoFromCallable();
        demoInterval();
        demoTimer();
    }

    // -------------------------------------------------------------------------
    // Observable.create
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.create(ObservableOnSubscribe)} gives you full manual control:
     * you receive an {@link ObservableEmitter} and decide when to call
     * {@code onNext}, {@code onError}, and {@code onComplete}.
     *
     * <p>Always check {@code emitter.isDisposed()} before emitting to avoid work
     * after the downstream has unsubscribed.
     */
demoCreate() 
    private static void demoCreate() {
        System.out.println("=== Observable.create ===");

        Observable<String> observable = Observable.create((ObservableEmitter<String> emitter) -> {
            try {
                emitter.onNext("First");   // emit item 1
                emitter.onNext("Second");  // emit item 2
                emitter.onNext("Third");   // emit item 3
                emitter.onComplete();      // signal completion
            } catch (Exception e) {
                emitter.onError(e);        // forward any exception as an error signal
            }
        });

        observable.subscribe(
                item -> System.out.println("Received: " + item),
                // → Received: First
                // → Received: Second
                // → Received: Third
                err  -> System.out.println("Error: " + err),
                ()   -> System.out.println("Done!")             // → Done!
        );

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.defer
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.defer(Callable<Observable<T>>)} delays the creation of the Observable
     * until a subscriber subscribes. Each subscriber gets a *fresh* Observable instance,
     * so any state captured at creation time is re-evaluated per subscription.
     *
     * <p>This solves a common bug where a value is captured once at definition time
     * instead of being read lazily at subscription time.
     */
demoDefer() 
    private static void demoDefer() {
        System.out.println("=== Observable.defer ===");

        AtomicInteger counter = new AtomicInteger(0);

        // Without defer — captures counter.get() ONCE when the observable is defined
        Observable<Integer> eager = Observable.just(counter.get());

        // With defer — evaluates counter.get() fresh for EACH subscriber
        Observable<Integer> lazy = Observable.defer(() -> Observable.just(counter.get()));

        counter.set(42); // change the value AFTER defining both observables

        System.out.print("Eager (no defer): ");
        eager.subscribe(v -> System.out.println(v));  // → 0  (captured before counter changed)

        System.out.print("Lazy  (defer):    ");
        lazy.subscribe(v -> System.out.println(v));   // → 42 (reads current value at subscribe time)

        // === Observable.defer ===
        // Eager (no defer): 0
        // Lazy  (defer):    42
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.fromCallable
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.fromCallable(Callable<T>)} wraps a single value-producing call.
     * The {@link java.util.concurrent.Callable} is executed lazily when someone subscribes,
     * and any checked exception thrown by it is forwarded to {@code onError}.
     *
     * <p>Think of it as a lazy version of {@code Observable.just(value)} that also handles
     * exceptions automatically.
     */
demoFromCallable() 
    private static void demoFromCallable() {
        System.out.println("=== Observable.fromCallable ===");

        // Simulates a method that might throw a checked exception (e.g., a DB call)
        Observable<String> observable = Observable.fromCallable(() -> {
            System.out.println("  [Callable executing...]"); // → [Callable executing...]
            return "Result from callable";
        });

        System.out.println("Observable defined, callable not yet executed.");
        // → Observable defined, callable not yet executed.

        observable.subscribe(
                result -> System.out.println("Got: " + result),  // → Got: Result from callable
                err    -> System.out.println("Error: " + err)
        );
        // === Observable.fromCallable ===
        // Observable defined, callable not yet executed.
        //   [Callable executing...]
        // Got: Result from callable
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.interval
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.interval(period, TimeUnit)} emits {@code 0L, 1L, 2L, ...} at a fixed
     * time interval on the {@code Schedulers.computation()} thread.
     *
     * <p>It is infinite — use {@code take(n)} to limit items, or {@code dispose()} to cancel.
     * Because it runs on a background thread, we {@code Thread.sleep} on the main thread
     * to wait for the output before the JVM exits.
     */
demoInterval() 
    private static void demoInterval() throws InterruptedException {
        System.out.println("=== Observable.interval ===");

        // Emit every 300 ms, take the first 4 items
        Observable.interval(300, TimeUnit.MILLISECONDS)
                .take(4)
                .subscribe(tick -> System.out.println("Tick: " + tick));
        // → Tick: 0
        // → Tick: 1
        // → Tick: 2
        // → Tick: 3

        Thread.sleep(1_500); // wait for all 4 ticks to be emitted
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable.timer
    // -------------------------------------------------------------------------

    /**
     * {@code Observable.timer(delay, TimeUnit)} emits a single {@code 0L} after the given
     * delay, then completes. Useful for scheduling a one-shot delayed action.
     */
demoTimer() 
    private static void demoTimer() throws InterruptedException {
        System.out.println("=== Observable.timer ===");

        System.out.println("Waiting 500 ms...");
        Observable.timer(500, TimeUnit.MILLISECONDS)
                .subscribe(
                        v  -> System.out.println("Timer fired! value=" + v), // → Timer fired! value=0
                        e  -> System.out.println("Error: " + e),
                        () -> System.out.println("Timer completed.")          // → Timer completed.
                );

        Thread.sleep(1_000); // wait for the timer to fire
        // === Observable.timer ===
        // Waiting 500 ms...
        // Timer fired! value=0
        // Timer completed.
        System.out.println();
    }
}

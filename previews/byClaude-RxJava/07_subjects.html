package tutorial;

import io.reactivex.rxjava3.subjects.AsyncSubject;
import io.reactivex.rxjava3.subjects.BehaviorSubject;
import io.reactivex.rxjava3.subjects.PublishSubject;
import io.reactivex.rxjava3.subjects.ReplaySubject;

/**
 * TUTORIAL 07 — Subjects
 *
 * <p>A {@code Subject} is both an {@code Observable} (you can subscribe to it) and an
 * {@code Observer} (you can push items into it manually). This makes it a bridge between
 * imperative code and reactive streams.
 *
 * <p>Use Subjects when you need to push values from non-reactive code into a reactive pipeline,
 * or when multiple subscribers need to share the same stream.
 *
 * <p>Subject types covered:
 * <ul>
 *   <li>{@link PublishSubject}   — emits only items that arrive AFTER subscription</li>
 *   <li>{@link BehaviorSubject}  — replays the last item to new subscribers</li>
 *   <li>{@link ReplaySubject}    — replays all (or N) past items to new subscribers</li>
 *   <li>{@link AsyncSubject}     — emits only the LAST item, and only on completion</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_07_Subjects.main()'
 */
public class _07_Subjects {

    public static void main(String[] args) {
        demoPublishSubject();
        demoBehaviorSubject();
        demoReplaySubject();
        demoAsyncSubject();
    }

    // -------------------------------------------------------------------------
    // PublishSubject
    // -------------------------------------------------------------------------

    /**
     * {@link PublishSubject} is the simplest Subject. Subscribers only receive items
     * emitted *after* they subscribe — they miss anything that was emitted before.
     *
     * <p>Good analogy: a live radio broadcast — you only hear what's playing right now,
     * not what played before you tuned in.
     */
demoPublishSubject() 
    private static void demoPublishSubject() {
        System.out.println("=== PublishSubject ===");

        PublishSubject<String> subject = PublishSubject.create();

        // Subscriber 1 subscribes before any items are emitted
        subject.subscribe(item -> System.out.println("Subscriber-1: " + item));

        subject.onNext("Item A"); // Subscriber-1 receives this
        subject.onNext("Item B"); // Subscriber-1 receives this

        // Subscriber 2 subscribes AFTER "A" and "B" were emitted — misses them!
        subject.subscribe(item -> System.out.println("Subscriber-2: " + item));

        subject.onNext("Item C"); // Both subscribers receive this
        subject.onComplete();

        // === PublishSubject ===
        //Subscriber-1: Item A
        //Subscriber-1: Item B
        //Subscriber-1: Item C
        //Subscriber-2: Item C

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // BehaviorSubject
    // -------------------------------------------------------------------------

    /**
     * {@link BehaviorSubject} remembers the *last emitted item* (or the initial seed value
     * if nothing has been emitted yet) and replays it to each new subscriber immediately
     * upon subscription.
     *
     * <p>Good analogy: a thermostat display — a new viewer always sees the current temperature,
     * not a blank screen.
     */
demoBehaviorSubject() 
    private static void demoBehaviorSubject() {
        System.out.println("=== BehaviorSubject ===");

        // Create with initial value "Initial"
        BehaviorSubject<String> subject = BehaviorSubject.createDefault("Initial");

        // Subscriber 1 immediately receives the current value "Initial"
        subject.subscribe(item -> System.out.println("Subscriber-1: " + item));
        // → Subscriber-1: Initial

        subject.onNext("Update-1");
        // → Subscriber-1: Update-1

        subject.onNext("Update-2");
        // → Subscriber-1: Update-2

        // Subscriber 2 subscribes now — immediately receives the latest value "Update-2"
        subject.subscribe(item -> System.out.println("Subscriber-2: " + item));
        // → Subscriber-2: Update-2

        subject.onNext("Update-3");
        // → Subscriber-1: Update-3
        // → Subscriber-2: Update-3

        subject.onComplete();

        // You can also query the current value directly:
        System.out.println("Current value (after complete): " + subject.getValue());
        // → Current value (after complete): Update-3

        // === BehaviorSubject ===
        //Subscriber-1: Initial
        //Subscriber-1: Update-1
        //Subscriber-1: Update-2
        //Subscriber-2: Update-2
        //Subscriber-1: Update-3
        //Subscriber-2: Update-3
        //Current value (after complete): null
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // ReplaySubject
    // -------------------------------------------------------------------------

    /**
     * {@link ReplaySubject} buffers all emitted items (or the last N items) and replays
     * them in full to every new subscriber, regardless of when they subscribe.
     *
     * <p>Good analogy: a recorded podcast — you can always listen from the beginning
     * no matter when you press play.
     *
     * <p>Variants:
     * <ul>
     *   <li>{@code ReplaySubject.create()}          — unbounded buffer (all items)</li>
     *   <li>{@code ReplaySubject.createWithSize(n)} — keeps only the last N items</li>
     *   <li>{@code ReplaySubject.createWithTime(t, unit, scheduler)} — time-based window</li>
     * </ul>
     */
demoReplaySubject() 
    private static void demoReplaySubject() {
        System.out.println("=== ReplaySubject ===");

        ReplaySubject<Integer> subject = ReplaySubject.create();

        subject.onNext(10);
        subject.onNext(20);
        subject.onNext(30);

        // Late subscriber still receives all past items
        subject.subscribe(n -> System.out.println("Subscriber (late): " + n));
        // → Subscriber (late): 10
        // → Subscriber (late): 20
        // → Subscriber (late): 30

        subject.onNext(40); // also receives future items
        // → Subscriber (late): 40

        subject.onComplete();

        System.out.println("--- ReplaySubject with size limit (last 2) ---");
        ReplaySubject<String> limited = ReplaySubject.createWithSize(2);
        limited.onNext("A");
        limited.onNext("B");
        limited.onNext("C"); // "A" is evicted; buffer holds ["B", "C"]

        // Late subscriber only gets the last 2
        limited.subscribe(s -> System.out.println("Limited replay: " + s));
        // → Limited replay: B
        // → Limited replay: C

        limited.onComplete();

        // === ReplaySubject ===
        //Subscriber (late): 10
        //Subscriber (late): 20
        //Subscriber (late): 30
        //Subscriber (late): 40
        //--- ReplaySubject with size limit (last 2) ---
        //Limited replay: B
        //Limited replay: C
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // AsyncSubject
    // -------------------------------------------------------------------------

    /**
     * {@link AsyncSubject} only emits the *very last item* emitted by the source,
     * and it does so only after {@code onComplete()} is called.
     *
     * <p>Good analogy: a computation that you submit — you only get the final result
     * once it finishes, not intermediate values.
     *
     * <p>If the subject errors before completing, subscribers only receive the error
     * (no items at all).
     */
demoAsyncSubject() 
    private static void demoAsyncSubject() {
        System.out.println("=== AsyncSubject ===");

        AsyncSubject<String> subject = AsyncSubject.create();

        subject.subscribe(item -> System.out.println("Subscriber-1: " + item));

        subject.onNext("Value 1"); // intermediate — not yet delivered
        subject.onNext("Value 2"); // intermediate — not yet delivered
        subject.onNext("Value 3"); // this will be the final value

        // Subscriber 2 subscribes before completion — also gets only the last value
        subject.subscribe(item -> System.out.println("Subscriber-2: " + item));

        subject.onComplete(); // triggers delivery of the last item ("Value 3") to all subscribers

        // === AsyncSubject ===
        //Subscriber-1: Value 3
        //Subscriber-2: Value 3

        System.out.println();
    }
}

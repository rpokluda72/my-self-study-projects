package tutorial;

import io.reactivex.rxjava3.core.Observable;

import java.util.concurrent.atomic.AtomicInteger;

/**
 * TUTORIAL 06 — Error Handling
 *
 * <p>In RxJava, errors are first-class citizens — they travel down the stream as an
 * {@code onError} signal. Proper error handling keeps your pipeline alive and predictable
 * instead of crashing with an unhandled exception.
 *
 * <p>Operators covered:
 * <ul>
 *   <li>{@code onErrorReturn}        — replace an error with a fallback item</li>
 *   <li>{@code onErrorReturnItem}    — replace an error with a fixed fallback item</li>
 *   <li>{@code onErrorResumeNext}    — replace an error with a fallback Observable</li>
 *   <li>{@code doOnError}            — side-effect on error without intercepting it</li>
 *   <li>{@code retry}                — re-subscribe on error, up to N times</li>
 *   <li>{@code retryWhen}            — conditional/delayed retry with custom logic</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_06_ErrorHandling.main()'
 */
public class _06_ErrorHandling {

    public static void main(String[] args) throws InterruptedException {
        demoOnErrorReturn();
        demoOnErrorReturnItem();
        demoOnErrorResumeNext();
        demoDoOnError();
        demoRetry();
        demoRetryWhen();
    }

    // -------------------------------------------------------------------------
    // onErrorReturn
    // -------------------------------------------------------------------------

    /**
     * {@code onErrorReturn(Function<Throwable, T>)} intercepts an error signal, maps it
     * to a fallback item using the provided function, emits that item, then completes normally.
     *
     * <p>The stream terminates after the fallback item — you get the fallback and {@code onComplete},
     * NOT further items from the original source.
     */
demoOnErrorReturn() 
    private static void demoOnErrorReturn() {
        System.out.println("=== onErrorReturn ===");

        Observable.just(1, 2, 3)
                .map(n -> {
                    if (n == 2) throw new RuntimeException("Bad value: " + n);
                    return n;
                })
                .onErrorReturn(err -> {
                    System.out.println("  Handling error: " + err.getMessage()); // → Handling error: Bad value: 2
                    return -1; // fallback value
                })
                .subscribe(
                        item -> System.out.println("Item: " + item),
                        err  -> System.out.println("Error reached subscriber: " + err),
                        ()   -> System.out.println("Completed normally") // → Completed normally
                );
        // === onErrorReturn ===
        //Item: 1
        //  Handling error: Bad value: 2
        //Item: -1
        //Completed normally
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // onErrorReturnItem
    // -------------------------------------------------------------------------

    /**
     * {@code onErrorReturnItem(T item)} is a simpler variant of {@code onErrorReturn}
     * that always returns the same fixed value, regardless of which error occurred.
     */
demoOnErrorReturnItem() 
    private static void demoOnErrorReturnItem() {
        System.out.println("=== onErrorReturnItem ===");

        Observable.<String>error(new RuntimeException("Connection failed"))
                .onErrorReturnItem("DEFAULT")
                .subscribe(
                        item -> System.out.println("Received: " + item), // → Received: DEFAULT
                        err  -> System.out.println("Error: " + err)
                );
        // === onErrorReturnItem ===
        //Received: DEFAULT
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // onErrorResumeNext
    // -------------------------------------------------------------------------

    /**
     * {@code onErrorResumeNext(Function<Throwable, Observable<T>>)} intercepts an error
     * and switches to a fallback Observable. The fallback can emit multiple items,
     * giving you a whole backup stream rather than just one item.
     */
demoOnErrorResumeNext() 
    private static void demoOnErrorResumeNext() {
        System.out.println("=== onErrorResumeNext ===");

        Observable<String> primary  = Observable.error(new RuntimeException("Primary failed"));
        Observable<String> fallback = Observable.just("Fallback-1", "Fallback-2");

        primary.onErrorResumeNext(err -> {
                    System.out.println("  Switching to fallback because: " + err.getMessage());
                    // → Switching to fallback because: Primary failed
                    return fallback;
                })
                .subscribe(
                        item -> System.out.println("Received: " + item),
                        err  -> System.out.println("Error: " + err),
                        ()   -> System.out.println("Done") // → Done
                );
        // === onErrorResumeNext ===
        //  Switching to fallback because: Primary failed
        //Received: Fallback-1
        //Received: Fallback-2
        //Done
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // doOnError
    // -------------------------------------------------------------------------

    /**
     * {@code doOnError(Consumer<Throwable>)} is a *side-effect* operator — it lets you
     * observe (log, alert, etc.) an error without actually handling or intercepting it.
     * The error continues to propagate downstream after the side-effect runs.
     */
demoDoOnError() 
    private static void demoDoOnError() {
        System.out.println("=== doOnError ===");

        Observable.<Integer>error(new IllegalStateException("Bad state"))
                .doOnError(err -> System.out.println("  [LOG] Error intercepted: " + err.getMessage()))
                .subscribe(
                        item -> System.out.println("Item: " + item),
                        err  -> System.out.println("Subscriber caught: " + err.getMessage())
                );
        // === doOnError ===
        //  [LOG] Error intercepted: Bad state
        //Subscriber caught: Bad state
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // retry
    // -------------------------------------------------------------------------

    /**
     * {@code retry(times)} re-subscribes to the source Observable when it errors,
     * up to {@code times} attempts. If it still errors after all retries, the error
     * is forwarded to the subscriber.
     *
     * <p>Each retry re-subscribes from scratch — for cold Observables this means
     * restarting the whole sequence.
     */
demoRetry() 
    private static void demoRetry() {
        System.out.println("=== retry(2) ===");

        AtomicInteger attemptCount = new AtomicInteger(0);

        Observable.create(emitter -> {
                    int attempt = attemptCount.incrementAndGet();
                    System.out.println("  Attempt #" + attempt);
                    if (attempt < 3) {
                        emitter.onError(new RuntimeException("Fail on attempt " + attempt));
                    } else {
                        emitter.onNext("Success on attempt " + attempt);
                        emitter.onComplete();
                    }
                })
                .retry(3)  // retry up to 3 times
                .subscribe(
                        item -> System.out.println("Result: " + item),
                        err  -> System.out.println("Final error: " + err.getMessage())
                );
        // === retry(2) ===
        //  Attempt #1
        //  Attempt #2
        //  Attempt #3
        //Result: Success on attempt 3

        System.out.println();
    }

    // -------------------------------------------------------------------------
    // retryWhen
    // -------------------------------------------------------------------------

    /**
     * {@code retryWhen(Function<Observable<Throwable>, Observable<?>>)} gives you full
     * control over the retry logic. The function receives an Observable of errors and
     * must return an Observable — when that Observable emits, a retry happens;
     * when it completes or errors, the retry chain ends.
     *
     * <p>This lets you implement strategies like exponential back-off or retry-only-on-specific-errors.
     * Below we do a simple "retry up to 2 times, but only for RuntimeException".
     */
demoRetryWhen() 
    private static void demoRetryWhen() throws InterruptedException {
        System.out.println("=== retryWhen ===");

        AtomicInteger counter = new AtomicInteger(0);

        Observable.create(emitter -> {
                    int n = counter.incrementAndGet();
                    System.out.println("  Try #" + n);
                    if (n <= 2) {
                        emitter.onError(new RuntimeException("Transient error #" + n));
                    } else {
                        emitter.onNext("OK after " + n + " tries");
                        emitter.onComplete();
                    }
                })
                .retryWhen(errors -> errors
                        .zipWith(Observable.range(1, 2), (err, retryNumber) -> retryNumber)
                        // zip with range(1,2) → allows exactly 2 retries; when range ends, retryWhen stops
                )
                .subscribe(
                        item -> System.out.println("Result: " + item),
                        // → Try #1
                        // → Try #2
                        // → Try #3
                        // → Result: OK after 3 tries
                        err  -> System.out.println("Final error: " + err.getMessage())
                );
        // .zipWith(Observable.range(1, 2)
        // === retryWhen ===
        //  Try #1
        //  Try #2

        // .zipWith(Observable.range(1, 4)
        // === retryWhen ===
        //  Try #1
        //  Try #2
        //  Try #3
        //Result: OK after 3 tries
        System.out.println();
    }
}

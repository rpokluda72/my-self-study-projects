package tutorial;

import io.reactivex.rxjava3.core.Completable;
import io.reactivex.rxjava3.core.Maybe;
import io.reactivex.rxjava3.core.Single;

/**
 * TUTORIAL 10 — Single, Maybe, and Completable
 *
 * <p>RxJava provides specialized reactive types for streams with a known fixed number of items.
 * These are more expressive than Observable when you know the cardinality up front:
 *
 * <ul>
 *   <li>{@link Single}&lt;T&gt;      — emits exactly ONE item, or an error. Never completes empty.
 *                                     Use for HTTP requests, database queries, computations.</li>
 *   <li>{@link Maybe}&lt;T&gt;       — emits ZERO or ONE item, then completes (or errors).
 *                                     Use for optional results (nullable cache lookups).</li>
 *   <li>{@link Completable}          — emits NO items, only a completion or error signal.
 *                                     Use for fire-and-forget actions (write to DB, send email).</li>
 * </ul>
 *
 * <p>Run: right-click → Run '_10_SingleMaybeCompletable.main()'
 */
public class _10_SingleMaybeCompletable {

    public static void main(String[] args) {
        // Single
        demoSingleJust();
        demoSingleError();
        demoSingleFromCallable();
        demoSingleFlatMap();

        // Maybe
        demoMaybeJust();
        demoMaybeEmpty();
        demoMaybeError();
        demoMaybeFilter();

        // Completable
        demoCompletableFromAction();
        demoCompletableAndThen();
        demoCompletableError();
        demoChainAll();
    }

    // =========================================================================
    //  SINGLE
    // =========================================================================

    // -------------------------------------------------------------------------
    // Single.just
    // -------------------------------------------------------------------------

    /**
     * {@code Single.just(value)} creates a Single that immediately emits the given value
     * and then signals success. The subscriber uses {@code onSuccess} (not {@code onNext}).
     */
demoSingleJust() 
    private static void demoSingleJust() {
        System.out.println("=== Single.just ===");

        Single.just(42)
                .subscribe(
                        value -> System.out.println("Success: " + value), // → Success: 42
                        error -> System.out.println("Error: " + error)
                );
        // === Single.just ===
        //Success: 42
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Single.error
    // -------------------------------------------------------------------------

    /**
     * {@code Single.error(Throwable)} creates a Single that immediately signals an error.
     * No success value is ever emitted.
     */
demoSingleError() 
    private static void demoSingleError() {
        System.out.println("=== Single.error ===");

        Single.error(new RuntimeException("Query failed"))
                .subscribe(
                        value -> System.out.println("Success: " + value), // not called
                        error -> System.out.println("Error: " + error.getMessage()) // → Error: Query failed
                );
        // === Single.error ===
        //Error: Query failed
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Single.fromCallable
    // -------------------------------------------------------------------------

    /**
     * {@code Single.fromCallable(Callable<T>)} lazily executes a {@link java.util.concurrent.Callable}
     * when subscribed. Any checked exception thrown is forwarded to {@code onError}.
     */
demoSingleFromCallable() 
    private static void demoSingleFromCallable() {
        System.out.println("=== Single.fromCallable ===");

        Single<String> single = Single.fromCallable(() -> {
            // Simulate a database lookup
            return "User{id=1, name='Alice'}";
        });

        single.subscribe(
                result -> System.out.println("Fetched: " + result), // → Fetched: User{id=1, name='Alice'}
                error  -> System.out.println("Error: " + error)
        );
        // === Single.fromCallable ===
        //Fetched: User{id=1, name='Alice'}
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Single.flatMap
    // -------------------------------------------------------------------------

    /**
     * {@code Single.flatMap(Function<T, Single<R>>)} chains two Singles together:
     * when the first Single succeeds, its value is fed into a function that returns
     * a second Single. The final result is the second Single's value.
     *
     * <p>Use this to sequence asynchronous operations that depend on each other
     * (e.g., fetch a user, then fetch that user's orders).
     */
demoSingleFlatMap() 
    private static void demoSingleFlatMap() {
        System.out.println("=== Single.flatMap (chain two Singles) ===");

        Single<Integer> userId = Single.just(1);

        userId
                .flatMap(id -> Single.fromCallable(() -> "User #" + id + " → Alice"))
                .flatMap(user -> Single.fromCallable(() -> user + " (verified)"))
                .subscribe(
                        result -> System.out.println("Result: " + result),
                        // → Result: User #1 → Alice (verified)
                        error  -> System.out.println("Error: " + error)
                );
        // === Single.flatMap (chain two Singles) ===
        //Result: User #1 → Alice (verified)
        System.out.println();
    }

    // =========================================================================
    //  MAYBE
    // =========================================================================

    // -------------------------------------------------------------------------
    // Maybe.just
    // -------------------------------------------------------------------------

    /**
     * {@code Maybe.just(value)} creates a Maybe that emits exactly one item via
     * {@code onSuccess}, then completes. Behaves like a Single when a value is present.
     */
demoMaybeJust() 
    private static void demoMaybeJust() {
        System.out.println("=== Maybe.just ===");

        Maybe.just("Found it!")
                .subscribe(
                        value    -> System.out.println("onSuccess: " + value), // → onSuccess: Found it!
                        error    -> System.out.println("onError: " + error),
                        ()       -> System.out.println("onComplete (no value)")  // not called when value present
                );
        // === Maybe.just ===
        //onSuccess: Found it!
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Maybe.empty
    // -------------------------------------------------------------------------

    /**
     * {@code Maybe.empty()} creates a Maybe that emits NO item and immediately completes.
     * Use this to represent a "nothing found" result (like an empty Optional).
     */
demoMaybeEmpty() 
    private static void demoMaybeEmpty() {
        System.out.println("=== Maybe.empty ===");

        Maybe.empty()
                .subscribe(
                        value -> System.out.println("onSuccess: " + value), // not called
                        error -> System.out.println("onError: " + error),   // not called
                        ()    -> System.out.println("onComplete (empty)")   // → onComplete (empty)
                );
        // === Maybe.empty ===
        //onComplete (empty)
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Maybe.error
    // -------------------------------------------------------------------------

    /**
     * {@code Maybe.error(Throwable)} signals an error immediately, with no item.
     */
demoMaybeError() 
    private static void demoMaybeError() {
        System.out.println("=== Maybe.error ===");

        Maybe.error(new IllegalArgumentException("Invalid ID"))
                .subscribe(
                        value -> System.out.println("onSuccess: " + value),           // not called
                        error -> System.out.println("onError: " + error.getMessage()), // → onError: Invalid ID
                        ()    -> System.out.println("onComplete")                      // not called
                );
        // === Maybe.error ===
        //onError: Invalid ID
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Observable → Maybe via filter + firstElement
    // -------------------------------------------------------------------------

    /**
     * A common way to obtain a {@code Maybe<T>} is by calling {@code firstElement()} on
     * a filtered Observable — it emits the first matching item, or completes empty if none match.
     */
demoMaybeFilter() 
    private static void demoMaybeFilter() {
        System.out.println("=== Maybe from Observable.firstElement ===");

        // Find the first even number — exists
        io.reactivex.rxjava3.core.Observable.just(1, 3, 4, 6, 7)
                .filter(n -> n % 2 == 0)
                .firstElement()              // returns Maybe<Integer>
                .subscribe(
                        n  -> System.out.println("First even: " + n), // → First even: 4
                        e  -> System.out.println("Error: " + e),
                        () -> System.out.println("No even found")
                );

        // Find the first even number — does NOT exist
        io.reactivex.rxjava3.core.Observable.just(1, 3, 5)
                .filter(n -> n % 2 == 0)
                .firstElement()
                .subscribe(
                        n  -> System.out.println("First even: " + n),
                        e  -> System.out.println("Error: " + e),
                        () -> System.out.println("No even found") // → No even found
                );
        // === Maybe from Observable.firstElement ===
        //First even: 4
        //No even found
        System.out.println();
    }

    // =========================================================================
    //  COMPLETABLE
    // =========================================================================

    // -------------------------------------------------------------------------
    // Completable.fromAction
    // -------------------------------------------------------------------------

    /**
     * {@code Completable.fromAction(Action)} wraps a void operation in a Completable.
     * When subscribed, it runs the action and signals {@code onComplete} on success,
     * or {@code onError} if an exception is thrown.
     */
demoCompletableFromAction() 
    private static void demoCompletableFromAction() {
        System.out.println("=== Completable.fromAction ===");

        Completable saveAction = Completable.fromAction(() -> {
            System.out.println("  [Action] Saving to database..."); // → [Action] Saving to database...
            // Simulate a write operation
        });

        saveAction.subscribe(
                ()  -> System.out.println("Save completed successfully"), // → Save completed successfully
                err -> System.out.println("Save failed: " + err)
        );
        // === Completable.fromAction ===
        //  [Action] Saving to database...
        //Save completed successfully
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Completable.andThen
    // -------------------------------------------------------------------------

    /**
     * {@code completable.andThen(other)} sequences two reactive sources:
     * {@code other} only starts after {@code completable} completes successfully.
     * If {@code completable} errors, {@code other} is never subscribed to.
     *
     * <p>You can chain:
     * <ul>
     *   <li>{@code Completable.andThen(Completable)} — sequence two void operations</li>
     *   <li>{@code Completable.andThen(Single)}      — run action, then get a value</li>
     *   <li>{@code Completable.andThen(Observable)}  — run action, then stream items</li>
     * </ul>
     */
demoCompletableAndThen() 
    private static void demoCompletableAndThen() {
        System.out.println("=== Completable.andThen ===");

        Completable step1 = Completable.fromAction(() ->
                System.out.println("  Step 1: validate input")); // → Step 1: validate input

        Completable step2 = Completable.fromAction(() ->
                System.out.println("  Step 2: persist data"));   // → Step 2: persist data

        Single<String> step3 = Single.fromCallable(() -> {
            System.out.println("  Step 3: fetch confirmation");  // → Step 3: fetch confirmation
            return "Confirmation #12345";
        });

        step1.andThen(step2)
                .andThen(step3)
                .subscribe(
                        result -> System.out.println("Final result: " + result),
                        // → Final result: Confirmation #12345
                        error  -> System.out.println("Error: " + error)
                );
        // === Completable.andThen ===
        //  Step 1: validate input
        //  Step 2: persist data
        //  Step 3: fetch confirmation
        //Final result: Confirmation #12345
        System.out.println();
    }

    // -------------------------------------------------------------------------
    // Completable.error
    // -------------------------------------------------------------------------

    /**
     * {@code Completable.error(Throwable)} creates a Completable that immediately errors.
     * The {@code andThen} chain is aborted — subsequent steps are skipped.
     */
demoCompletableError() 
    private static void demoCompletableError() {
        System.out.println("=== Completable chain with error ===");

        Completable failing = Completable.error(new RuntimeException("Permission denied"));
        Completable next    = Completable.fromAction(() ->
                System.out.println("  This step is skipped")); // never reached

        failing.andThen(next)
                .subscribe(
                        ()  -> System.out.println("Completed"),
                        err -> System.out.println("Chain failed: " + err.getMessage())
                        // → Chain failed: Permission denied
                );
        // === Completable chain with error ===
        //Chain failed: Permission denied
        System.out.println();
    }

    // =========================================================================
    //  Combined example
    // =========================================================================

    /**
     * A realistic chain mixing all three types:
     * <ol>
     *   <li>Authenticate (Completable) — no value produced</li>
     *   <li>Fetch user ID (Single)     — produces an Integer</li>
     *   <li>Look up profile (Maybe)    — profile may or may not exist</li>
     * </ol>
     */
demoChainAll() 
    private static void demoChainAll() {
        System.out.println("=== Real-world chain: Completable → Single → Maybe ===");

        Completable authenticate = Completable.fromAction(() ->
                System.out.println("  [Auth] Token validated")); // → [Auth] Token validated

        Single<Integer> fetchUserId = Single.fromCallable(() -> {
            System.out.println("  [DB] Fetching user ID"); // → [DB] Fetching user ID
            return 42;
        });

        authenticate
                .andThen(fetchUserId)                                  // after auth, get ID
                .flatMapMaybe(id -> Maybe.fromCallable(() -> {         // look up profile (may be absent)
                    System.out.println("  [DB] Loading profile for ID=" + id);
                    // → [DB] Loading profile for ID=42
                    return "Profile{id=" + id + ", name='Bob'}";
                }))
                .subscribe(
                        profile  -> System.out.println("Loaded: " + profile),
                        // → Loaded: Profile{id=42, name='Bob'}
                        error    -> System.out.println("Error: " + error),
                        ()       -> System.out.println("Profile not found")
                );
        // === Real-world chain: Completable → Single → Maybe ===
        //  [Auth] Token validated
        //  [DB] Fetching user ID
        //  [DB] Loading profile for ID=42
        //Loaded: Profile{id=42, name='Bob'}
        System.out.println();
    }
}

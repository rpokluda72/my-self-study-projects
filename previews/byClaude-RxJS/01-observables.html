01-observables.ts 
/**
 * Observables
 *
 * An Observable is a lazy push-based collection that can emit zero or more values
 * over time, then optionally complete or error. Unlike Promises, Observables are
 * lazy (nothing happens until you subscribe) and can emit multiple values.
 *
 * @see https://rxjs.dev/guide/observable
 */

import { Observable } from 'rxjs';

/**
 * The most basic Observable: constructed manually with new Observable().
 * The subscriber function runs each time someone subscribes.
 */
function createObservable(): void {
  const obs$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.next(2);
    subscriber.next(3);
    subscriber.complete();
  });

  obs$.subscribe({
    next: val => console.log('next:', val),
    complete: () => console.log('complete'),
  });

  // Expected output:
  // next: 1
  // next: 2
  // next: 3
  // complete
}

/**
 * Lazy execution: the Observable body does NOT run until subscribe() is called.
 * Each call to subscribe() triggers a fresh independent execution.
 */
function lazyExecution(): void {
  console.log('Creating observable...');

  const obs$ = new Observable<string>(subscriber => {
    console.log('  Observable executing!');
    subscriber.next('hello');
    subscriber.complete();
  });

  console.log('Before subscribe');
  obs$.subscribe(val => console.log('  value:', val));
  console.log('After subscribe');

  // Expected output:
  // Creating observable...
  // Before subscribe
  //   Observable executing!
  //   value: hello
  // After subscribe
}

/**
 * Synchronous delivery: values are delivered synchronously by default —
 * the subscribe() call blocks until the Observable completes.
 */
function synchronousObservable(): void {
  let result = 0;

  new Observable<number>(subscriber => {
    subscriber.next(10);
    subscriber.next(20);
    subscriber.complete();
  }).subscribe(val => { result += val; });

  console.log('result after subscribe:', result); // 30 — already computed synchronously

  // Expected output:
  // result after subscribe: 30
}

/**
 * Asynchronous delivery: values can be emitted later using setTimeout or other async APIs.
 * The subscriber function can schedule future emissions.
 */
function asyncObservable(): void {
  const obs$ = new Observable<string>(subscriber => {
    console.log('  subscribed');
    setTimeout(() => { subscriber.next('async value 1'); }, 100);
    setTimeout(() => { subscriber.next('async value 2'); subscriber.complete(); }, 200);
  });

  obs$.subscribe({
    next: val => console.log('  next:', val),
    complete: () => console.log('  complete'),
  });

  console.log('  (subscribe returned — values arrive later)');

  // Expected output:
  //   subscribed
  //   (subscribe returned — values arrive later)
  //   next: async value 1       ← after ~100 ms
  //   next: async value 2       ← after ~200 ms
  //   complete
}

/**
 * complete() and error(): an Observable terminal states.
 * After complete() or error(), no more values are delivered.
 */
function completeAndError(): void {
  // --- complete ---
  const good$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.complete();
    subscriber.next(99); // ignored — already completed
  });

  good$.subscribe({
    next: val => console.log('good next:', val),
    complete: () => console.log('good complete'),
  });

  // --- error ---
  const bad$ = new Observable<number>(subscriber => {
    subscriber.next(42);
    subscriber.error(new Error('something went wrong'));
    subscriber.next(99); // ignored — already errored
  });

  bad$.subscribe({
    next: val => console.log('bad next:', val),
    error: err => console.log('bad error:', (err as Error).message),
  });

  // Expected output:
  // good next: 1
  // good complete
  // bad next: 42
  // bad error: something went wrong
}

/**
 * Cold Observable: each subscriber gets its own independent execution.
 * The sequence restarts from the beginning for every new subscriber.
 */
function coldObservable(): void {
  const cold$ = new Observable<number>(subscriber => {
    const value = Math.floor(Math.random() * 100);
    subscriber.next(value);
    subscriber.complete();
  });

  cold$.subscribe(val => console.log('subscriber A:', val));
  cold$.subscribe(val => console.log('subscriber B:', val));
  cold$.subscribe(val => console.log('subscriber C:', val));

  // Expected output (values differ per run — each subscriber gets its own random value):
  // subscriber A: 42
  // subscriber B: 17
  // subscriber C: 83
}

function main(): void {
  console.log('\n=== createObservable ===');
  createObservable();

  console.log('\n=== lazyExecution ===');
  lazyExecution();

  console.log('\n=== synchronousObservable ===');
  synchronousObservable();

  console.log('\n=== asyncObservable ===');
  asyncObservable();

  console.log('\n=== completeAndError ===');
  completeAndError();

  console.log('\n=== coldObservable ===');
  coldObservable();
}

main();

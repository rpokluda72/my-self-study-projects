Operators: Filtering

08-operators-filtering.ts 
/**
 * Operators: Filtering
 *
 * Filtering operators let values through (or block them) based on a predicate,
 * position, or timing. They never transform values — a value either passes
 * through unchanged or is silently discarded.
 *
 * @see https://rxjs.dev/guide/operators#filtering-operators
 */

import { of, interval, Subject, from } from 'rxjs';
import {
  filter,
  take,
  takeUntil,
  skip,
  debounceTime,
  throttleTime,
  distinctUntilChanged,
  first,
  last,
} from 'rxjs';

/**
 * filter(predicate): passes only values that satisfy the predicate function.
 * Values that return false are silently discarded.
 */
function operatorFilter(): void {
  of(1, 2, 3, 4, 5, 6)
    .pipe(filter(x => x % 2 === 0))
    .subscribe(val => console.log('filter (even):', val));

  from(['apple', 'banana', 'avocado', 'cherry'])
    .pipe(filter(s => s.startsWith('a')))
    .subscribe(val => console.log('filter (starts with a):', val));

  // Expected output:
  // filter (even): 2
  // filter (even): 4
  // filter (even): 6
  // filter (starts with a): apple
  // filter (starts with a): avocado
}

/**
 * take(count): emits the first N values, then completes.
 * Automatically unsubscribes the source after N values.
 */
function operatorTake(): void {
  of(1, 2, 3, 4, 5)
    .pipe(take(3))
    .subscribe({
      next:     val => console.log('take:', val),
      complete: ()  => console.log('take complete'),
    });

  // Expected output:
  // take: 1
  // take: 2
  // take: 3
  // take complete
}

/**
 * takeUntil(notifier$): emits values until the notifier Observable emits.
 * When the notifier fires, the source is unsubscribed immediately.
 * The canonical way to tie the lifetime of one Observable to another.
 */
function operatorTakeUntil(): void {
  const stop$ = new Subject<void>();

  interval(80)
    .pipe(takeUntil(stop$))
    .subscribe({
      next:     val => console.log('takeUntil:', val),
      complete: ()  => console.log('takeUntil complete'),
    });

  setTimeout(() => stop$.next(), 300); // stop after ~3 ticks

  // Expected output:
  // takeUntil: 0
  // takeUntil: 1
  // takeUntil: 2
  // takeUntil complete
}

/**
 * skip(count): ignores the first N values, then passes all subsequent values through.
 * Opposite of take().
 */
function operatorSkip(): void {
  of(1, 2, 3, 4, 5)
    .pipe(skip(2))
    .subscribe(val => console.log('skip:', val));

  // Expected output:
  // skip: 3
  // skip: 4
  // skip: 5
}

/**
 * debounceTime(ms): waits for a "silence period" of ms milliseconds after the
 * last emission before passing the value through. Useful for search inputs —
 * only fires after the user has stopped typing.
 */
function operatorDebounceTime(): void {
  const input$ = new Subject<string>();

  input$
    .pipe(debounceTime(100))
    .subscribe(val => console.log('debounced:', val));

  // Rapid-fire emissions — only the last one in each burst passes through
  input$.next('r');
  input$.next('rx');
  input$.next('rxj');
  input$.next('rxjs'); // only this passes (100 ms silence after)

  setTimeout(() => {
    input$.next('rxjs '); // new burst after silence
    input$.next('rxjs o');
    input$.next('rxjs op'); // only this passes
  }, 300);

  // Expected output:
  // debounced: rxjs
  // debounced: rxjs op
}

/**
 * throttleTime(ms): passes the first value through, then ignores subsequent values
 * for ms milliseconds. Limits the emission rate.
 * Useful for scroll/resize events where you want periodic updates, not every pixel.
 */
function operatorThrottleTime(): void {
  const scroll$ = new Subject<number>();

  scroll$
    .pipe(throttleTime(150))
    .subscribe(val => console.log('throttled:', val));

  // Emit rapidly — only the first of each 150 ms window passes through
  scroll$.next(0);
  scroll$.next(10);
  scroll$.next(20); // blocked
  scroll$.next(30); // blocked

  setTimeout(() => scroll$.next(100), 200); // new window — passes through
  setTimeout(() => scroll$.next(110), 220); // blocked

  // Expected output:
  // throttled: 0
  // throttled: 100
}

/**
 * distinctUntilChanged(): suppresses consecutive duplicate values.
 * A value is only emitted if it is different from the immediately preceding value.
 */
function operatorDistinctUntilChanged(): void {
  of(1, 1, 2, 2, 2, 3, 1, 1)
    .pipe(distinctUntilChanged())
    .subscribe(val => console.log('distinct:', val));

  // With a custom comparator
  of(
    { id: 1, name: 'Alice' },
    { id: 1, name: 'Alice' },
    { id: 2, name: 'Bob' },
  )
    .pipe(distinctUntilChanged((a, b) => a.id === b.id))
    .subscribe(val => console.log('distinct by id:', val.name));

  // Expected output:
  // distinct: 1
  // distinct: 2
  // distinct: 3
  // distinct: 1
  // distinct by id: Alice
  // distinct by id: Bob
}

/**
 * first(predicate?): emits only the first value (or first matching value) then completes.
 * Errors if no matching value is emitted before the source completes.
 */
function operatorFirst(): void {
  of(3, 1, 4, 1, 5, 9)
    .pipe(first())
    .subscribe({
      next:     val => console.log('first:', val),
      complete: ()  => console.log('first complete'),
    });

  of(3, 1, 4, 1, 5, 9)
    .pipe(first(x => x > 4))
    .subscribe({
      next:     val => console.log('first > 4:', val),
      complete: ()  => console.log('first > 4 complete'),
    });

  // Expected output:
  // first: 3
  // first complete
  // first > 4: 5
  // first > 4 complete
}

/**
 * last(): emits only the last value before the source completes, then completes.
 * The source must complete — last() on an infinite Observable never emits.
 */
function operatorLast(): void {
  of(10, 20, 30, 40, 50)
    .pipe(last())
    .subscribe({
      next:     val => console.log('last:', val),
      complete: ()  => console.log('last complete'),
    });

  // Expected output:
  // last: 50
  // last complete
}

function main(): void {
  console.log('\n=== operatorFilter ===');
  operatorFilter();

  console.log('\n=== operatorTake ===');
  operatorTake();

  console.log('\n=== operatorTakeUntil ===');
  operatorTakeUntil();

  console.log('\n=== operatorSkip ===');
  operatorSkip();

  console.log('\n=== operatorDebounceTime ===');
  operatorDebounceTime();

  console.log('\n=== operatorThrottleTime ===');
  operatorThrottleTime();

  console.log('\n=== operatorDistinctUntilChanged ===');
  operatorDistinctUntilChanged();

  console.log('\n=== operatorFirst ===');
  operatorFirst();

  console.log('\n=== operatorLast ===');
  operatorLast();
}

main();

10-operators-utility.ts 
/**
 * Operators: Utility
 *
 * Utility operators provide side effects, timing control, and multicasting
 * without altering the stream's values. They are the "hooks" and
 * "plumbing" of a pipeline.
 *
 * @see https://rxjs.dev/guide/operators#utility-operators
 */

import { of, interval, timer, Subject, NEVER } from 'rxjs';
import {
  tap,
  delay,
  timeout,
  shareReplay,
  finalize,
  toArray,
  take,
  map,
  catchError,
} from 'rxjs';

/**
 * tap(observer): runs side effects for each notification without modifying
 * the stream. Ideal for logging, debugging, or triggering external actions
 * while keeping the pipeline values unchanged.
 */
function operatorTap(): void {
  of(1, 2, 3)
    .pipe(
      tap(val => console.log('  tap before map:', val)),
      map(val => val * 10),
      tap({
        next:     val => console.log('  tap after map:', val),
        complete: ()  => console.log('  tap complete'),
      }),
    )
    .subscribe(val => console.log('subscriber:', val));

  // Expected output:
  //   tap before map: 1
  //   tap after map: 10
  // subscriber: 10
  //   tap before map: 2
  //   tap after map: 20
  // subscriber: 20
  //   tap before map: 3
  //   tap after map: 30
  // subscriber: 30
  //   tap complete
}

/**
 * delay(ms): shifts every emission forward in time by ms milliseconds.
 * The source runs immediately; notifications are buffered and delivered later.
 */
function operatorDelay(): void {
  console.log('before subscribe');

  of('a', 'b', 'c')
    .pipe(delay(100))
    .subscribe({
      next:     val => console.log('delay:', val),
      complete: ()  => console.log('delay complete'),
    });

  console.log('after subscribe — values arrive in 100 ms');

  // Expected output:
  // before subscribe
  // after subscribe — values arrive in 100 ms
  // delay: a       ← after 100 ms
  // delay: b
  // delay: c
  // delay complete
}

/**
 * timeout(config): errors with TimeoutError if no value arrives within the
 * specified window. Useful for enforcing SLA on slow Observables.
 */
function operatorTimeout(): void {
  // Fast source — completes within timeout
  of(42)
    .pipe(timeout({ each: 200 }))
    .subscribe({
      next:     val => console.log('timeout ok:', val),
      error:    err => console.log('timeout error:', err.message),
    });

  // Slow source — exceeds timeout
  timer(300)
    .pipe(
      timeout({ each: 100 }),
      catchError(err => {
        console.log('slow source timed out:', err.name);
        return of(-1);
      })
    )
    .subscribe(val => console.log('timeout fallback:', val));

  // Expected output:
  // timeout ok: 42
  // slow source timed out: TimeoutError
  // timeout fallback: -1
}

/**
 * shareReplay(bufferSize): multicasts the source Observable and replays the
 * last N values to new subscribers. Prevents multiple subscriptions from
 * re-executing expensive sources (e.g., HTTP requests).
 */
function operatorShareReplay(): void {
  let executeCount = 0;

  const expensive$ = new Subject<number>();
  const shared$ = expensive$.pipe(
    tap(() => executeCount++),
    shareReplay(1),
  );

  // First subscriber
  shared$.subscribe(val => console.log('shareReplay A:', val));

  expensive$.next(100);
  expensive$.next(200);

  // Second subscriber joins late — immediately gets the last replayed value (200)
  shared$.subscribe(val => console.log('shareReplay B:', val));

  expensive$.next(300);

  console.log('total emissions processed:', executeCount); // 3 (not 5)

  // Expected output:
  // shareReplay A: 100
  // shareReplay A: 200
  // shareReplay B: 200   ← replayed on subscribe
  // shareReplay A: 300
  // shareReplay B: 300
  // total emissions processed: 3
}

/**
 * finalize(callback): runs a cleanup callback when the Observable terminates
 * for any reason — complete, error, or unsubscribe. Like a finally block.
 */
function operatorFinalize(): void {
  of(1, 2, 3)
    .pipe(
      finalize(() => console.log('finalize: cleanup on complete')),
    )
    .subscribe({
      next:     val => console.log('finalize next:', val),
      complete: ()  => console.log('finalize complete'),
    });

  // Also runs on unsubscribe
  const sub = interval(50).pipe(
    take(10),
    finalize(() => console.log('finalize: cleanup on unsubscribe')),
  ).subscribe();

  setTimeout(() => sub.unsubscribe(), 120);

  // Expected output:
  // finalize next: 1
  // finalize next: 2
  // finalize next: 3
  // finalize complete
  // finalize: cleanup on complete
  // finalize: cleanup on unsubscribe   ← after ~120 ms
}

/**
 * toArray(): collects all values emitted by the source and emits them as a
 * single array when the source completes. The source must be finite.
 */
function operatorToArray(): void {
  of(1, 2, 3, 4, 5)
    .pipe(toArray())
    .subscribe(arr => console.log('toArray:', arr));

  interval(50)
    .pipe(take(4), toArray())
    .subscribe(arr => console.log('toArray from interval:', arr));

  // Expected output:
  // toArray: [ 1, 2, 3, 4, 5 ]
  // toArray from interval: [ 0, 1, 2, 3 ]
}

function main(): void {
  console.log('\n=== operatorTap ===');
  operatorTap();

  console.log('\n=== operatorDelay ===');
  operatorDelay();

  console.log('\n=== operatorTimeout ===');
  operatorTimeout();

  console.log('\n=== operatorShareReplay ===');
  operatorShareReplay();

  console.log('\n=== operatorFinalize ===');
  operatorFinalize();

  console.log('\n=== operatorToArray ===');
  operatorToArray();
}

main();

05-scheduler.ts 
/**
 * Scheduler
 *
 * A Scheduler controls when a subscription starts and when notifications are
 * delivered. It virtualizes time, allowing you to decide whether an Observable
 * emits synchronously, via setTimeout, via setInterval, or via requestAnimationFrame.
 *
 * Most operators accept an optional scheduler argument. You can also use
 * observeOn() and subscribeOn() to change delivery context at any point.
 *
 * @see https://rxjs.dev/guide/scheduler
 */

import {
  asyncScheduler,
  queueScheduler,
  animationFrameScheduler,
  scheduled,
  of,
  interval,
  Observable,
} from 'rxjs';
import { observeOn, subscribeOn, take } from 'rxjs';

/**
 * queueScheduler: executes tasks synchronously in a trampoline queue.
 * Tasks scheduled recursively are queued and executed one by one, preventing
 * stack overflows. Behaves like synchronous execution from the caller's perspective.
 */
function queueSchedulerExample(): void {
  console.log('start');

  queueScheduler.schedule(() => console.log('queueScheduler task 1'));
  queueScheduler.schedule(() => console.log('queueScheduler task 2'));

  console.log('end');

  // Expected output:
  // start
  // queueScheduler task 1
  // queueScheduler task 2
  // end
}

/**
 * asyncScheduler: defers execution using setInterval / setTimeout.
 * Tasks run asynchronously — after the current call stack clears.
 * Optionally accepts a delay in milliseconds.
 */
function asyncSchedulerExample(): void {
  console.log('before schedule');

  asyncScheduler.schedule(() => console.log('async task (no delay)'));
  asyncScheduler.schedule(() => console.log('async task (100 ms delay)'), 100);

  console.log('after schedule — async tasks run later');

  // Expected output:
  // before schedule
  // after schedule — async tasks run later
  // async task (no delay)        ← next event loop tick
  // async task (100 ms delay)    ← after 100 ms
}

/**
 * animationFrameScheduler: schedules work before the next browser repaint.
 * Only meaningful in a browser environment — in Node.js it falls back to
 * setTimeout behavior. Safe to import and use the API in Node.js.
 *
 * Note: animationFrameScheduler is browser-only. This example shows the API
 * shape. In a browser you would use it to drive smooth animations.
 */
function animationFrameSchedulerNote(): void {
  // Safe to import; but calling schedule() requires a browser (requestAnimationFrame).
  console.log('animationFrameScheduler available:', typeof animationFrameScheduler !== 'undefined');
  console.log('(browser-only: requires requestAnimationFrame — do not schedule in Node.js)');

  // In a browser you would write:
  //   animationFrameScheduler.schedule(() => updateCanvas());
  // Or use it with interval:
  //   interval(0, animationFrameScheduler).pipe(take(60)).subscribe(updateCanvas);

  // Expected output:
  // animationFrameScheduler available: true
  // (browser-only: requires requestAnimationFrame — do not schedule in Node.js)
}

/**
 * observeOn(scheduler): re-emits values on the given scheduler.
 * Changes the context in which next/error/complete notifications are delivered
 * to the subscriber — does not affect when the source produces values.
 */
function observeOnExample(): void {
  console.log('before subscribe');

  of(1, 2, 3)
    .pipe(observeOn(asyncScheduler))
    .subscribe({
      next:     val => console.log('next:', val),
      complete: ()  => console.log('complete'),
    });

  console.log('after subscribe — values delivered asynchronously');

  // Expected output:
  // before subscribe
  // after subscribe — values delivered asynchronously
  // next: 1
  // next: 2
  // next: 3
  // complete
}

/**
 * subscribeOn(scheduler): changes the scheduler on which the subscription itself
 * starts. Useful for moving the entire execution (including the subscriber body)
 * to a different context.
 */
function subscribeOnExample(): void {
  console.log('before subscribe');

  of('a', 'b', 'c')
    .pipe(subscribeOn(asyncScheduler))
    .subscribe({
      next:     val => console.log('next:', val),
      complete: ()  => console.log('complete'),
    });

  console.log('after subscribe — subscription deferred asynchronously');

  // Expected output:
  // before subscribe
  // after subscribe — subscription deferred asynchronously
  // next: a
  // next: b
  // next: c
  // complete
}

/**
 * Passing a scheduler explicitly to a creation operator.
 * interval() and other operators accept a scheduler as their last argument,
 * letting you swap between async (default), queue, or animation-frame delivery.
 */
function schedulerWithInterval(): void {
  // Explicitly pass asyncScheduler (same as default, shown for clarity)
  interval(100, asyncScheduler)
    .pipe(take(3))
    .subscribe({
      next:     val => console.log('tick:', val),
      complete: ()  => console.log('done'),
    });

  // scheduled() wraps any iterable/array with a chosen scheduler
  scheduled([10, 20, 30], asyncScheduler).subscribe({
    next:     val => console.log('scheduled value:', val),
    complete: ()  => console.log('scheduled complete'),
  });

  // Expected output (scheduled values appear first — both use asyncScheduler
  // but scheduled() has no per-item delay whereas interval() does):
  // scheduled value: 10
  // scheduled value: 20
  // scheduled value: 30
  // scheduled complete
  // tick: 0
  // tick: 1
  // tick: 2
  // done
}

function main(): void {
  console.log('\n=== queueSchedulerExample ===');
  queueSchedulerExample();

  console.log('\n=== asyncSchedulerExample ===');
  asyncSchedulerExample();

  console.log('\n=== animationFrameSchedulerNote ===');
  animationFrameSchedulerNote();

  console.log('\n=== observeOnExample ===');
  observeOnExample();

  console.log('\n=== subscribeOnExample ===');
  subscribeOnExample();

  console.log('\n=== schedulerWithInterval ===');
  schedulerWithInterval();
}

main();

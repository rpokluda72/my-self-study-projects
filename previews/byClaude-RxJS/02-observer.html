/**
 * Observer
 *
 * An Observer is the consumer of values delivered by an Observable. It is an
 * object with up to three callback methods: next (receives each value), error
 * (receives the error and stops execution), and complete (called when the
 * stream ends normally). All three callbacks are optional.
 *
 * @see https://rxjs.dev/guide/observer
 */

import { Observable, Observer } from 'rxjs';

/**
 * A full Observer object with all three handlers: next, error, complete.
 * This is the most explicit form of subscribing.
 */
function fullObserver(): void {
  const source$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.next(2);
    subscriber.complete();
  });

  const observer: Observer<number> = {
    next:     val  => console.log('next:', val),
    error:    err  => console.error('error:', err),
    complete: ()   => console.log('complete'),
  };

  source$.subscribe(observer);

  // Expected output:
  // next: 1
  // next: 2
  // complete
}

/**
 * Partial Observer: only the next callback is required.
 * Omitting error or complete is perfectly valid — unhandled errors will
 * propagate as uncaught exceptions.
 */
function partialObserver(): void {
  const source$ = new Observable<string>(subscriber => {
    subscriber.next('hello');
    subscriber.next('world');
    subscriber.complete();
  });

  source$.subscribe({ next: val => console.log('value:', val) });

  // Expected output:
  // value: hello
  // value: world
}

/**
 * Callback-style subscribe: pass next, error, and complete as positional arguments.
 * This is a shorthand — internally RxJS wraps them into an Observer object.
 *
 * Note: The positional overload (subscribe(nextFn, errorFn, completeFn)) is
 * deprecated in favour of the object observer form. Prefer { next, error, complete }.
 */
function callbackObserver(): void {
  const source$ = new Observable<number>(subscriber => {
    subscriber.next(10);
    subscriber.next(20);
    subscriber.complete();
  });

  // Modern preferred form (object observer):
  source$.subscribe({
    next:     val => console.log('next:', val),
    complete: ()  => console.log('done'),
  });

  // Expected output:
  // next: 10
  // next: 20
  // done
}

/**
 * Error handling in an Observer: when error() fires, no further next() or
 * complete() calls are made. The stream is terminated.
 */
function observerErrorHandling(): void {
  const source$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.next(2);
    subscriber.error(new Error('oops'));
    subscriber.next(3);    // never reached
    subscriber.complete(); // never reached
  });

  source$.subscribe({
    next:     val => console.log('next:', val),
    error:    err => console.log('error caught:', (err as Error).message),
    complete: ()  => console.log('complete'),  // never called
  });

  // Expected output:
  // next: 1
  // next: 2
  // error caught: oops
}

/**
 * Values emitted after complete() are silently ignored by the Observer.
 * complete() signals the end of the stream — the subscriber is automatically
 * unsubscribed and will receive nothing further.
 */
function observerAfterComplete(): void {
  const source$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.complete();
    subscriber.next(2);  // silently ignored
    subscriber.next(3);  // silently ignored
  });

  source$.subscribe({
    next:     val => console.log('next:', val),
    complete: ()  => console.log('complete'),
  });

  // Expected output:
  // next: 1
  // complete
}

function main(): void {
  console.log('\n=== fullObserver ===');
  fullObserver();

  console.log('\n=== partialObserver ===');
  partialObserver();

  console.log('\n=== callbackObserver ===');
  callbackObserver();

  console.log('\n=== observerErrorHandling ===');
  observerErrorHandling();

  console.log('\n=== observerAfterComplete ===');
  observerAfterComplete();
}

main();

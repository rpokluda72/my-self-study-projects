Operators: Combination

09-operators-combination.ts 
/**
 * Operators: Combination
 *
 * Combination operators join multiple Observables into one. They differ in
 * timing strategy: some interleave (merge), some sequence (concat), some
 * wait for all sources (combineLatest, zip, forkJoin), and some enrich a
 * primary stream with a secondary one (withLatestFrom).
 *
 * @see https://rxjs.dev/guide/operators#join-creation-operators
 */

import { of, interval, timer, merge, concat, combineLatest, zip, forkJoin } from 'rxjs';
import { withLatestFrom, startWith, take, map, delay } from 'rxjs';

/**
 * merge(...sources): subscribes to all sources simultaneously and interleaves
 * their emissions in arrival order. Completes when ALL sources complete.
 */
function operatorMerge(): void {
  const fast$ = interval(60).pipe(take(3), map(n => `fast:${n}`));
  const slow$ = interval(100).pipe(take(3), map(n => `slow:${n}`));

  merge(fast$, slow$).subscribe({
    next:     val => console.log('merge:', val),
    complete: ()  => console.log('merge complete'),
  });

  // Expected output (interleaved by arrival time):
  // merge: fast:0
  // merge: slow:0
  // merge: fast:1
  // merge: fast:2
  // merge: slow:1
  // merge: slow:2
  // merge complete
}

/**
 * concat(...sources): subscribes to sources one at a time, waiting for each
 * to complete before starting the next. Preserves strict ordering.
 */
function operatorConcat(): void {
  const first$  = of('a', 'b', 'c');
  const second$ = of('d', 'e', 'f');
  const third$  = of('g', 'h', 'i');

  concat(first$, second$, third$).subscribe({
    next:     val => console.log('concat:', val),
    complete: ()  => console.log('concat complete'),
  });

  // Expected output (strictly ordered, second$ waits for first$ to complete):
  // concat: a
  // concat: b
  // concat: c
  // concat: d
  // concat: e
  // concat: f
  // concat: g
  // concat: h
  // concat: i
  // concat complete
}

/**
 * combineLatest([sources]): whenever any source emits, combines the latest
 * value from every source into an array. Only starts emitting after ALL sources
 * have emitted at least one value.
 */
function operatorCombineLatest(): void {
  const name$  = of('Alice').pipe(delay(50));
  const score$ = of(100).pipe(delay(80));

  combineLatest([name$, score$]).subscribe({
    next:     ([name, score]) => console.log(`combineLatest: ${name} scored ${score}`),
    complete: ()              => console.log('combineLatest complete'),
  });

  // Expected output:
  // combineLatest: Alice scored 100
  // combineLatest complete
}

/**
 * zip([sources]): pairs the Nth value from each source together.
 * Only emits when every source has contributed its Nth value.
 * The output length is limited by the shortest source.
 */
function operatorZip(): void {
  const letters$ = of('A', 'B', 'C');
  const numbers$ = of(1, 2, 3, 4); // extra value ignored

  zip([letters$, numbers$]).subscribe({
    next:     ([letter, number]) => console.log(`zip: ${letter}${number}`),
    complete: ()                 => console.log('zip complete'),
  });

  // Expected output:
  // zip: A1
  // zip: B2
  // zip: C3
  // zip complete
}

/**
 * forkJoin([sources]): waits for ALL sources to complete, then emits one array
 * containing the last value from each. Equivalent to Promise.all().
 * Useful for parallel HTTP requests where you need all results together.
 */
function operatorForkJoin(): void {
  const user$    = of({ name: 'Alice' }).pipe(delay(80));
  const profile$ = of({ role: 'admin' }).pipe(delay(50));
  const settings$= of({ theme: 'dark' }).pipe(delay(100));

  forkJoin([user$, profile$, settings$]).subscribe({
    next: ([user, profile, settings]) =>
      console.log('forkJoin:', user.name, profile.role, settings.theme),
    complete: () => console.log('forkJoin complete'),
  });

  // Expected output:
  // forkJoin: Alice admin dark
  // forkJoin complete
}

/**
 * withLatestFrom(other$): on each emission of the primary source, also captures
 * the most recent value from other$. Does NOT trigger on other$ emissions alone.
 * Useful for sampling state (e.g., get current user on each button click).
 */
function operatorWithLatestFrom(): void {
  const clicks$  = interval(120).pipe(take(3));        // primary
  const counter$ = interval(50).pipe(take(20));        // state to sample

  clicks$
    .pipe(withLatestFrom(counter$))
    .subscribe({
      next:     ([click, count]) => console.log(`withLatestFrom: click ${click}, counter was ${count}`),
      complete: ()               => console.log('withLatestFrom complete'),
    });

  // Expected output (counter value depends on timing):
  // withLatestFrom: click 0, counter was 1
  // withLatestFrom: click 1, counter was 4
  // withLatestFrom: click 2, counter was 7
  // withLatestFrom complete
}

/**
 * startWith(...values): prepends one or more values to the Observable before
 * the source begins emitting. Useful for providing an initial/default value.
 */
function operatorStartWith(): void {
  of(1, 2, 3)
    .pipe(startWith(0))
    .subscribe(val => console.log('startWith:', val));

  // Practical example: loading state
  of('data loaded')
    .pipe(
      delay(50),
      startWith('loading...'),
    )
    .subscribe(val => console.log('status:', val));

  // Expected output:
  // startWith: 0
  // startWith: 1
  // startWith: 2
  // startWith: 3
  // status: loading...
  // status: data loaded   ← after 50 ms
}

function main(): void {
  console.log('\n=== operatorMerge ===');
  operatorMerge();

  console.log('\n=== operatorConcat ===');
  operatorConcat();

  console.log('\n=== operatorCombineLatest ===');
  operatorCombineLatest();

  console.log('\n=== operatorZip ===');
  operatorZip();

  console.log('\n=== operatorForkJoin ===');
  operatorForkJoin();

  console.log('\n=== operatorWithLatestFrom ===');
  operatorWithLatestFrom();

  console.log('\n=== operatorStartWith ===');
  operatorStartWith();
}

main();

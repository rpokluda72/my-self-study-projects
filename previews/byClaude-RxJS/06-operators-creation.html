Operators: Creation

06-operators-creation.ts 
/**
 * Operators: Creation
 *
 * Creation operators are standalone functions that create a new Observable
 * from scratch — from static values, arrays, promises, events, timers, etc.
 * They are the entry points of most Observable pipelines.
 *
 * @see https://rxjs.dev/guide/operators#creation-operators-1
 */

import { EventEmitter } from 'events'; // Node.js built-in
import {
  of,
  from,
  interval,
  timer,
  range,
  EMPTY,
  NEVER,
  defer,
  fromEvent,
} from 'rxjs';
import { take } from 'rxjs';

/**
 * of(...values): emits each argument synchronously, then completes.
 * Simple way to create an Observable from a known set of values.
 */
function operatorOf(): void {
  of(1, 2, 3).subscribe({
    next:     val => console.log('of:', val),
    complete: ()  => console.log('of complete'),
  });

  of('hello', 'world').subscribe(val => console.log('of string:', val));

  // Expected output:
  // of: 1
  // of: 2
  // of: 3
  // of complete
  // of string: hello
  // of string: world
}

/**
 * from(source): converts an array, iterable, Promise, or Observable-like into an Observable.
 * Each element is emitted individually.
 */
function operatorFrom(): void {
  // From an array
  from([10, 20, 30]).subscribe(val => console.log('from array:', val));

  // From a Promise
  from(Promise.resolve('promise value')).subscribe(val => console.log('from promise:', val));

  // From an iterable (generator)
  function* gen() { yield 'a'; yield 'b'; yield 'c'; }
  from(gen()).subscribe(val => console.log('from generator:', val));

  // Expected output:
  // from array: 10
  // from array: 20
  // from array: 30
  // from generator: a
  // from generator: b
  // from generator: c
  // from promise: promise value   ← async, arrives after sync output
}

/**
 * interval(period): emits an incrementing integer (0, 1, 2, …) every `period` ms.
 * Never completes on its own — use take() to limit it.
 */
function operatorInterval(): void {
  interval(100).pipe(take(4)).subscribe({
    next:     val => console.log('interval:', val),
    complete: ()  => console.log('interval complete'),
  });

  // Expected output:
  // interval: 0
  // interval: 1
  // interval: 2
  // interval: 3
  // interval complete
}

/**
 * timer(dueTime, period?): waits dueTime ms then emits 0 and completes (single-shot),
 * or if period is given, continues emitting like interval after the initial delay.
 */
function operatorTimer(): void {
  // Single-shot: emits once after 150 ms
  timer(150).subscribe({
    next:     val => console.log('timer one-shot:', val),
    complete: ()  => console.log('timer one-shot complete'),
  });

  // Interval with initial delay: wait 100 ms, then emit every 80 ms
  timer(100, 80).pipe(take(3)).subscribe({
    next:     val => console.log('timer interval:', val),
    complete: ()  => console.log('timer interval complete'),
  });

  // Expected output (approximate — timers interleave):
  // timer interval: 0
  // timer one-shot: 0
  // timer one-shot complete
  // timer interval: 1
  // timer interval: 2
  // timer interval complete
}

/**
 * range(start, count): emits a sequence of integers synchronously, then completes.
 */
function operatorRange(): void {
  range(1, 5).subscribe({
    next:     val => console.log('range:', val),
    complete: ()  => console.log('range complete'),
  });

  // Expected output:
  // range: 1
  // range: 2
  // range: 3
  // range: 4
  // range: 5
  // range complete
}

/**
 * EMPTY: an Observable that completes immediately without emitting any values.
 * Useful as a no-op source or return value in operator callbacks.
 */
function operatorEmpty(): void {
  console.log('before EMPTY subscribe');
  EMPTY.subscribe({
    next:     val => console.log('EMPTY next:', val), // never called
    complete: ()  => console.log('EMPTY complete'),
  });
  console.log('after EMPTY subscribe');

  // Expected output:
  // before EMPTY subscribe
  // EMPTY complete
  // after EMPTY subscribe
}

/**
 * NEVER: an Observable that never emits and never completes.
 * Useful in tests or as a placeholder that should have no effect.
 */
function operatorNever(): void {
  console.log('subscribing to NEVER...');
  const sub = NEVER.subscribe({
    next:     ()  => console.log('never next'),     // never called
    complete: ()  => console.log('never complete'), // never called
  });
  console.log('NEVER sub.closed:', sub.closed); // false — still open

  sub.unsubscribe();
  console.log('after unsubscribe, sub.closed:', sub.closed); // true

  // Expected output:
  // subscribing to NEVER...
  // NEVER sub.closed: false
  // after unsubscribe, sub.closed: true
}

/**
 * defer(observableFactory): creates the Observable lazily — the factory function
 * is called fresh for each subscriber. Useful when the Observable depends on
 * state that changes between subscriptions (e.g., current timestamp).
 */
function operatorDefer(): void {
  const lazy$ = defer(() => of(Date.now()));

  lazy$.subscribe(val => console.log('subscriber A timestamp:', val));

  setTimeout(() => {
    lazy$.subscribe(val => console.log('subscriber B timestamp:', val)); // different value
  }, 50);

  // Expected output (values differ because defer re-evaluates Date.now() each time):
  // subscriber A timestamp: 1713870000000
  // subscriber B timestamp: 1713870000051
}

/**
 * fromEvent(target, eventName): wraps a DOM or Node.js EventEmitter into an Observable.
 * In a browser you would pass a DOM element; here we use Node.js EventEmitter.
 *
 * Note: In a browser, you would write: fromEvent(document, 'click')
 */
function operatorFromEvent(): void {
  const emitter = new EventEmitter();
  const clicks$ = fromEvent<string>(emitter, 'data');

  const sub = clicks$.subscribe(val => console.log('event:', val));

  emitter.emit('data', 'first message');
  emitter.emit('data', 'second message');

  sub.unsubscribe();
  emitter.emit('data', 'after unsubscribe'); // not received

  // Expected output:
  // event: first message
  // event: second message
}

function main(): void {
  console.log('\n=== operatorOf ===');
  operatorOf();

  console.log('\n=== operatorFrom ===');
  operatorFrom();

  console.log('\n=== operatorInterval ===');
  operatorInterval();

  console.log('\n=== operatorTimer ===');
  operatorTimer();

  console.log('\n=== operatorRange ===');
  operatorRange();

  console.log('\n=== operatorEmpty ===');
  operatorEmpty();

  console.log('\n=== operatorNever ===');
  operatorNever();

  console.log('\n=== operatorDefer ===');
  operatorDefer();

  console.log('\n=== operatorFromEvent ===');
  operatorFromEvent();
}

main();

Operators: Transformation

07-operators-transformation.ts 
/**
 * Operators: Transformation
 *
 * Transformation operators take each value from the source and produce new
 * values or inner Observables. The "higher-order" operators (mergeMap,
 * switchMap, concatMap, exhaustMap) each flatten those inner Observables
 * back into a single stream — differing only in how they handle concurrency.
 *
 * @see https://rxjs.dev/guide/operators#transformation-operators
 */

import { of, interval, from, timer } from 'rxjs';
import { map, mergeMap, switchMap, concatMap, exhaustMap, scan, reduce, delay, take } from 'rxjs';

/**
 * map(project): synchronously transforms each value.
 * The most fundamental transformation operator.
 */
function operatorMap(): void {
  of(1, 2, 3, 4, 5)
    .pipe(map(x => x * x))
    .subscribe(val => console.log('map:', val));

  of('hello', 'world')
    .pipe(map(s => s.toUpperCase()))
    .subscribe(val => console.log('map string:', val));

  // Expected output:
  // map: 1
  // map: 4
  // map: 9
  // map: 16
  // map: 25
  // map string: HELLO
  // map string: WORLD
}

/**
 * mergeMap(project): maps each value to an inner Observable and merges all
 * inner Observables concurrently. Does NOT cancel previous inner Observables.
 * Use when order does not matter and you want maximum concurrency.
 */
function operatorMergeMap(): void {
  of('A', 'B', 'C')
    .pipe(
      mergeMap(letter =>
        timer(Math.random() * 100).pipe(map(() => letter))
      )
    )
    .subscribe(val => console.log('mergeMap:', val));

  // Expected output (order varies — concurrent, unordered):
  // mergeMap: B
  // mergeMap: A
  // mergeMap: C
}

/**
 * switchMap(project): maps to an inner Observable, but cancels the previous
 * inner whenever a new outer value arrives. Only the latest inner is active.
 * Classic use case: search-as-you-type (cancel stale HTTP requests).
 */
function operatorSwitchMap(): void {
  of(1, 2, 3)
    .pipe(
      switchMap(n =>
        // Each inner emits n after 50 ms, but is cancelled by the next outer value
        timer(50).pipe(map(() => `result for ${n}`))
      )
    )
    .subscribe(val => console.log('switchMap:', val));

  // Expected output (only the last inner survives because of(1,2,3) is synchronous):
  // switchMap: result for 3
}

/**
 * concatMap(project): maps to inner Observables one at a time, waiting for
 * each to complete before subscribing to the next. Preserves order.
 * Use when order matters and Observables must not overlap.
 */
function operatorConcatMap(): void {
  of('X', 'Y', 'Z')
    .pipe(
      concatMap(letter =>
        of(`${letter}1`, `${letter}2`).pipe(delay(50))
      )
    )
    .subscribe(val => console.log('concatMap:', val));

  // Expected output (in strict order, each pair waits for the previous):
  // concatMap: X1
  // concatMap: X2
  // concatMap: Y1
  // concatMap: Y2
  // concatMap: Z1
  // concatMap: Z2
}

/**
 * exhaustMap(project): maps to an inner Observable, but ignores new outer values
 * while the current inner is still active. Use when you want to prevent overlapping
 * executions (e.g., a login button that ignores double-clicks).
 */
function operatorExhaustMap(): void {
  // Emit 0,1,2,3 every 60 ms; inner takes 150 ms each
  interval(60)
    .pipe(
      take(4),
      exhaustMap(n =>
        timer(150).pipe(map(() => `inner from ${n}`))
      )
    )
    .subscribe(val => console.log('exhaustMap:', val));

  // Expected output (0 starts an inner; 1 and 2 arrive during it and are ignored;
  // 3 arrives after the first inner completes and starts a new one):
  // exhaustMap: inner from 0
  // exhaustMap: inner from 3
}

/**
 * scan(accumulator, seed): applies an accumulator function over time and emits
 * each intermediate result. Like Array.reduce() but emits on every step.
 */
function operatorScan(): void {
  of(1, 2, 3, 4, 5)
    .pipe(scan((acc, val) => acc + val, 0))
    .subscribe(val => console.log('scan (running sum):', val));

  // Building an array incrementally
  of('a', 'b', 'c')
    .pipe(scan((acc, val) => [...acc, val], [] as string[]))
    .subscribe(val => console.log('scan (array):', val));

  // Expected output:
  // scan (running sum): 1
  // scan (running sum): 3
  // scan (running sum): 6
  // scan (running sum): 10
  // scan (running sum): 15
  // scan (array): [ 'a' ]
  // scan (array): [ 'a', 'b' ]
  // scan (array): [ 'a', 'b', 'c' ]
}

/**
 * reduce(accumulator, seed): like scan() but only emits the final accumulated
 * value when the source completes. Equivalent to Array.reduce().
 */
function operatorReduce(): void {
  of(1, 2, 3, 4, 5)
    .pipe(reduce((acc, val) => acc + val, 0))
    .subscribe(val => console.log('reduce (final sum):', val));

  of('hello', ' ', 'world')
    .pipe(reduce((acc, val) => acc + val, ''))
    .subscribe(val => console.log('reduce (string):', val));

  // Expected output:
  // reduce (final sum): 15
  // reduce (string): hello  world
}

/**
 * mergeMap vs switchMap contrast: shows the key behavioural difference.
 * - mergeMap: all inners run concurrently, results interleave
 * - switchMap: only the latest inner runs, previous are cancelled
 */
function mergeMapVsSwitchMap(): void {
  const source$ = interval(80).pipe(take(3)); // emits 0, 1, 2

  console.log('--- mergeMap (all inners run) ---');
  source$.pipe(
    mergeMap(n => timer(120).pipe(map(() => `merge:${n}`)))
  ).subscribe(val => console.log(val));

  // Delay the switchMap demo so it doesn't interleave with the above
  setTimeout(() => {
    console.log('--- switchMap (only last inner runs) ---');
    source$.pipe(
      switchMap(n => timer(120).pipe(map(() => `switch:${n}`)))
    ).subscribe(val => console.log(val));
  }, 500);

  // Expected output:
  // --- mergeMap (all inners run) ---
  // merge:0
  // merge:1
  // merge:2
  // --- switchMap (only last inner runs) ---
  // switch:2       ← 0 and 1 were cancelled when newer values arrived
}

function main(): void {
  console.log('\n=== operatorMap ===');
  operatorMap();

  console.log('\n=== operatorMergeMap ===');
  operatorMergeMap();

  console.log('\n=== operatorSwitchMap ===');
  operatorSwitchMap();

  console.log('\n=== operatorConcatMap ===');
  operatorConcatMap();

  console.log('\n=== operatorExhaustMap ===');
  operatorExhaustMap();

  console.log('\n=== operatorScan ===');
  operatorScan();

  console.log('\n=== operatorReduce ===');
  operatorReduce();

  console.log('\n=== mergeMapVsSwitchMap ===');
  mergeMapVsSwitchMap();
}

main();

11-operators-error-handling.ts 
/**
 * Operators: Error Handling
 *
 * Error handling operators intercept errors in a pipeline, giving you the
 * chance to recover, retry, or inspect and rethrow. Without them, an unhandled
 * error terminates the Observable and propagates to the subscriber's error callback.
 *
 * @see https://rxjs.dev/guide/operators#error-handling-operators
 */

import { of, throwError, interval, Observable } from 'rxjs';
import { catchError, retry, map, take, delay, tap } from 'rxjs';

/**
 * catchError(selector): intercepts an error and returns a replacement Observable.
 * The stream resumes with the replacement — the subscriber never sees the error.
 */
function operatorCatchError(): void {
  const failing$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.next(2);
    subscriber.error(new Error('source failed'));
  });

  failing$
    .pipe(catchError(err => {
      console.log('caught:', (err as Error).message);
      return of(-1, -2); // replacement Observable
    }))
    .subscribe({
      next:     val => console.log('catchError next:', val),
      complete: ()  => console.log('catchError complete'),
    });

  // Expected output:
  // catchError next: 1
  // catchError next: 2
  // caught: source failed
  // catchError next: -1
  // catchError next: -2
  // catchError complete
}

/**
 * throwError(factory): an Observable that immediately errors.
 * Used to create error sources in examples or to rethrow inside catchError.
 */
function operatorThrowError(): void {
  throwError(() => new Error('immediate error'))
    .pipe(
      catchError(err => {
        console.log('throwError caught:', (err as Error).message);
        return of('recovered');
      })
    )
    .subscribe(val => console.log('throwError result:', val));

  // Expected output:
  // throwError caught: immediate error
  // throwError result: recovered
}

/**
 * retry(count): on error, re-subscribes to the source up to `count` times.
 * If the source still fails after all retries, the error propagates.
 * Useful for transient failures (network timeouts, flaky APIs).
 */
function operatorRetry(): void {
  let attempt = 0;

  const flaky$ = new Observable<string>(subscriber => {
    attempt++;
    console.log(`  attempt #${attempt}`);
    if (attempt < 3) {
      subscriber.error(new Error(`attempt ${attempt} failed`));
    } else {
      subscriber.next('success!');
      subscriber.complete();
    }
  });

  flaky$
    .pipe(retry(3))
    .subscribe({
      next:     val => console.log('retry result:', val),
      error:    err => console.log('retry gave up:', (err as Error).message),
      complete: ()  => console.log('retry complete'),
    });

  // Expected output:
  //   attempt #1
  //   attempt #2
  //   attempt #3
  // retry result: success!
  // retry complete
}

/**
 * retry with delay (RxJS 7+): the modern way to retry with a delay between
 * attempts. Pass a config object with `count` and `delay` (ms or a function).
 * This is preferred over the deprecated retryWhen().
 */
function operatorRetryWithDelay(): void {
  let attempt = 0;

  const flaky$ = new Observable<string>(subscriber => {
    attempt++;
    console.log(`  attempt #${attempt}`);
    if (attempt < 3) {
      subscriber.error(new Error('not ready yet'));
    } else {
      subscriber.next('finally succeeded');
      subscriber.complete();
    }
  });

  flaky$
    .pipe(
      retry({ count: 5, delay: 50 }) // retry up to 5 times, 50 ms between each
    )
    .subscribe({
      next:     val => console.log('retryWithDelay result:', val),
      complete: ()  => console.log('retryWithDelay complete'),
    });

  // Expected output:
  //   attempt #1
  //   attempt #2   ← after 50 ms
  //   attempt #3   ← after 50 ms
  // retryWithDelay result: finally succeeded
  // retryWithDelay complete
}

/**
 * catchError — inspect and rethrow: sometimes you want to log or track an error
 * but still propagate it. Re-throw inside catchError using throwError or `throw`.
 */
function catchErrorRethrow(): void {
  const source$ = throwError(() => new Error('critical failure'));

  source$
    .pipe(
      catchError(err => {
        console.log('logging error before rethrow:', (err as Error).message);
        throw err; // rethrow — propagates to subscriber's error handler
      })
    )
    .subscribe({
      next:  val => console.log('value:', val),
      error: err => console.log('subscriber got error:', (err as Error).message),
    });

  // Expected output:
  // logging error before rethrow: critical failure
  // subscriber got error: critical failure
}

/**
 * Exhausting retries — what happens when all retries are used up.
 * The final error propagates to the subscriber's error callback.
 */
function retryExhausted(): void {
  let attempt = 0;

  const alwaysFails$ = new Observable<number>(subscriber => {
    attempt++;
    console.log(`  attempt #${attempt}`);
    subscriber.error(new Error('always fails'));
  });

  alwaysFails$
    .pipe(retry(2)) // try original + 2 retries = 3 total attempts
    .subscribe({
      next:  val => console.log('value:', val),
      error: err => console.log(`gave up after ${attempt} attempts:`, (err as Error).message),
    });

  // Expected output:
  //   attempt #1
  //   attempt #2
  //   attempt #3
  // gave up after 3 attempts: always fails
}

function main(): void {
  console.log('\n=== operatorCatchError ===');
  operatorCatchError();

  console.log('\n=== operatorThrowError ===');
  operatorThrowError();

  console.log('\n=== operatorRetry ===');
  operatorRetry();

  console.log('\n=== operatorRetryWithDelay ===');
  operatorRetryWithDelay();

  console.log('\n=== catchErrorRethrow ===');
  catchErrorRethrow();

  console.log('\n=== retryExhausted ===');
  retryExhausted();
}

main();

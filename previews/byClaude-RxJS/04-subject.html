/**
 * Subject
 *
 * A Subject is both an Observable and an Observer. It can multicast values
 * to multiple subscribers simultaneously — all subscribers share the same
 * execution. Unlike a plain Observable (which is unicast), a Subject acts
 * as an event bus: you push values into it manually via next(), and all
 * current subscribers receive them.
 *
 * Variants: BehaviorSubject, ReplaySubject, AsyncSubject each differ in
 * how they treat late subscribers.
 *
 * @see https://rxjs.dev/guide/subject
 */

import { Subject, BehaviorSubject, ReplaySubject, AsyncSubject } from 'rxjs';

/**
 * Basic Subject: multicast — multiple subscribers all receive the same emission.
 * Values pushed via next() are delivered to every active subscriber.
 */
function basicSubject(): void {
  const subject$ = new Subject<number>();

  subject$.subscribe(val => console.log('subscriber A:', val));
  subject$.subscribe(val => console.log('subscriber B:', val));

  subject$.next(1);
  subject$.next(2);
  subject$.complete();

  // Expected output:
  // subscriber A: 1
  // subscriber B: 1
  // subscriber A: 2
  // subscriber B: 2
}

/**
 * Late subscriber: a plain Subject does NOT replay past values.
 * A subscriber that joins after emissions have occurred misses those values entirely.
 */
function lateSubscriber(): void {
  const subject$ = new Subject<string>();

  subject$.subscribe(val => console.log('early subscriber:', val));

  subject$.next('first');
  subject$.next('second');

  // Subscribes AFTER the first two values — receives nothing past
  subject$.subscribe(val => console.log('late subscriber:', val));

  subject$.next('third');

  // Expected output:
  // early subscriber: first
  // early subscriber: second
  // early subscriber: third
  // late subscriber: third      ← missed "first" and "second"
}

/**
 * BehaviorSubject: stores the current value and replays it immediately to
 * any new subscriber. Requires an initial value. Useful for state management.
 */
function behaviorSubjectExample(): void {
  const behavior$ = new BehaviorSubject<number>(0); // initial value: 0

  console.log('current value:', behavior$.value); // 0

  behavior$.subscribe(val => console.log('A:', val)); // immediately gets 0

  behavior$.next(1);
  behavior$.next(2);

  behavior$.subscribe(val => console.log('B:', val)); // immediately gets 2 (latest)

  behavior$.next(3);

  // Expected output:
  // current value: 0
  // A: 0
  // A: 1
  // A: 2
  // B: 2    ← replayed on subscribe
  // A: 3
  // B: 3
}

/**
 * ReplaySubject: buffers the last N values and replays all of them to new
 * subscribers. Unlike BehaviorSubject, it works with any number of past values
 * and does not require an initial value.
 */
function replaySubjectExample(): void {
  const replay$ = new ReplaySubject<number>(2); // buffer last 2 values

  replay$.next(10);
  replay$.next(20);
  replay$.next(30);

  // Late subscriber gets the last 2 buffered values (20 and 30)
  replay$.subscribe(val => console.log('late subscriber:', val));

  replay$.next(40); // received in real-time

  // Expected output:
  // late subscriber: 20   ← replayed
  // late subscriber: 30   ← replayed
  // late subscriber: 40   ← real-time
}

/**
 * AsyncSubject: only emits the very last value, and only when complete() is called.
 * Useful for representing a single async operation (similar to a Promise).
 */
function asyncSubjectExample(): void {
  const async$ = new AsyncSubject<number>();

  async$.subscribe(val => console.log('subscriber:', val));

  async$.next(1); // buffered, not emitted yet
  async$.next(2); // buffered, not emitted yet
  async$.next(3); // buffered, not emitted yet

  console.log('completing...');
  async$.complete(); // now emits only the last value: 3

  // Late subscriber also gets the last value (AsyncSubject replays on subscribe after complete)
  async$.subscribe(val => console.log('late subscriber:', val));

  // Expected output:
  // completing...
  // subscriber: 3
  // late subscriber: 3
}

function main(): void {
  console.log('\n=== basicSubject ===');
  basicSubject();

  console.log('\n=== lateSubscriber ===');
  lateSubscriber();

  console.log('\n=== behaviorSubjectExample ===');
  behaviorSubjectExample();

  console.log('\n=== replaySubjectExample ===');
  replaySubjectExample();

  console.log('\n=== asyncSubjectExample ===');
  asyncSubjectExample();
}

main();

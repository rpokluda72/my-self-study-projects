/**
 * Subscription
 *
 * A Subscription represents the execution of an Observable. Calling subscribe()
 * returns a Subscription object that has an unsubscribe() method to cancel the
 * execution and release resources. Subscriptions can also be composed: adding
 * a child subscription to a parent means one unsubscribe() call cleans up both.
 *
 * @see https://rxjs.dev/guide/subscription
 */

import { Observable, Subscription, interval } from 'rxjs';
import { take } from 'rxjs';

/**
 * subscribe() returns a Subscription. Inspecting its closed property tells
 * you whether the subscription is still active.
 */
function basicSubscription(): void {
  const source$ = new Observable<number>(subscriber => {
    subscriber.next(1);
    subscriber.next(2);
    subscriber.complete();
  });

  const sub: Subscription = source$.subscribe({
    next:     val => console.log('next:', val),
    complete: ()  => console.log('complete'),
  });

  console.log('sub.closed after complete:', sub.closed); // true — auto-closed on complete

  // Expected output:
  // next: 1
  // next: 2
  // complete
  // sub.closed after complete: true
}

/**
 * Calling unsubscribe() stops value delivery immediately.
 * Useful for cancelling long-running or infinite streams.
 */
function unsubscribeExample(): void {
  let count = 0;

  const source$ = new Observable<number>(subscriber => {
    const id = setInterval(() => {
      count++;
      subscriber.next(count);
    }, 50);

    // Teardown: runs when unsubscribe() is called
    return () => {
      clearInterval(id);
      console.log('teardown called, count was:', count);
    };
  });

  const sub = source$.subscribe(val => console.log('value:', val));

  setTimeout(() => {
    sub.unsubscribe();
    // Expected output:
    // value: 1
    // value: 2
    // value: 3
    // teardown called, count was: 3
  }, 175); // ~3 ticks at 50 ms each
}

/**
 * Using take(N) is a cleaner alternative to manual unsubscribe for finite streams.
 * The subscription auto-closes after N values are received.
 */
function intervalWithTake(): void {
  interval(100).pipe(take(4)).subscribe({
    next:     val => console.log('tick:', val),
    complete: ()  => console.log('done after 4 ticks'),
  });

  // Expected output:
  // tick: 0
  // tick: 1
  // tick: 2
  // tick: 3
  // done after 4 ticks
}

/**
 * Composite Subscription: sub.add(childSub) makes the child a dependent.
 * Calling sub.unsubscribe() also unsubscribes all added children.
 */
function compositeSubscription(): void {
  const parent = new Subscription();

  const child1 = interval(80).pipe(take(3)).subscribe(
    val => console.log('child1:', val)
  );
  const child2 = interval(120).pipe(take(3)).subscribe(
    val => console.log('child2:', val)
  );

  parent.add(child1);
  parent.add(child2);

  // Unsubscribing the parent automatically unsubscribes child1 and child2
  setTimeout(() => {
    parent.unsubscribe();
    console.log('parent unsubscribed');
    console.log('child1.closed:', child1.closed); // true
    console.log('child2.closed:', child2.closed); // true
  }, 350);

  // Expected output (approximate — async timing):
  // child1: 0
  // child2: 0
  // child1: 1
  // child1: 2
  // child2: 1
  // parent unsubscribed
  // child1.closed: true
  // child2.closed: true
}

/**
 * Teardown function: the function returned from the subscriber body is called
 * when the subscription ends (via unsubscribe, complete, or error).
 * Use it to release any resources acquired during execution.
 */
function subscriptionTeardown(): void {
  const source$ = new Observable<number>(subscriber => {
    console.log('  execution started');
    let n = 0;
    const id = setInterval(() => subscriber.next(++n), 100);

    return () => {
      clearInterval(id);
      console.log('  teardown: interval cleared');
    };
  });

  const sub = source$.subscribe(val => console.log('  value:', val));

  setTimeout(() => sub.unsubscribe(), 250);

  // Expected output:
  //   execution started
  //   value: 1
  //   value: 2
  //   teardown: interval cleared
}

function main(): void {
  console.log('\n=== basicSubscription ===');
  basicSubscription();

  console.log('\n=== unsubscribeExample ===');
  unsubscribeExample();

  console.log('\n=== intervalWithTake ===');
  intervalWithTake();

  console.log('\n=== compositeSubscription ===');
  compositeSubscription();

  console.log('\n=== subscriptionTeardown ===');
  subscriptionTeardown();
}

main();
